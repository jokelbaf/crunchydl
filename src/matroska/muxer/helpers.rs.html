5pub(super) fn validate_tracks(tracks: &[Track]) -> Result<()> {
6    if tracks.is_empty() {
7        return Err(Error::Invalid("no tracks"));
8    }
9    let mut numbers = Vec::new();
10    let mut video_defaults = 0;
11    for (index, track) in tracks.iter().enumerate() {
12        track.validate()?;
13        let number = effective_number(index, track);
14        if number == 0 || numbers.contains(&number) {
15            return Err(Error::Invalid("track numbers must be unique and nonzero"));
16        }
17        numbers.push(number);
18        if track.track_type == TrackType::Video && track.default {
19            video_defaults += 1;
20        }
21    }
22    if video_defaults > 1 {
23        return Err(Error::Invalid("multiple default video tracks"));
24    }
25    Ok(())
26}

28pub(super) fn effective_number(index: usize, track: &Track) -> u64 {
29    if track.number == 0 {
30        index as u64 + 1
31    } else {
32        track.number
33    }
34}

51pub(super) fn write_info<W: Write + Seek>(output: &mut W, options: &MuxOptions) -> Result<u64> {
52    let mut children = vec![
53        uint(0x002a_d7b1, 1_000_000)?,
54        string(0x4d80, &options.muxing_application)?,
55        string(0x5741, &options.writing_application)?,
56    ];
57    if let Some(title) = &options.title {
58        children.push(string(0x7ba9, title)?);
59    }
60    let info_start = output.stream_position()?;
61    let duration = float64(0x4489, 0.0)?;
62    children.push(duration);
63    let encoded = master(0x1549_a966, children)?;
64    let marker = id_bytes(0x4489)?;
65    let offset = encoded
66        .windows(marker.len())
67        .position(|window| window == marker)
68        .ok_or(Error::Invalid("duration marker missing"))?;
69    let payload_position = info_start + offset as u64 + marker.len() as u64 + 1;
70    output.write_all(&encoded)?;
71    Ok(payload_position)
72}

83pub(super) fn write_void<W: Write>(output: &mut W, total: usize) -> Result<()> {
84    let id = id_bytes(0xec)?;
85    for width in 1..=8 {
86        if total > id.len() + width {
87            let payload = total - id.len() - width;
88            if let Ok(size) = size_bytes(payload as u64, Some(width)) {
89                output.write_all(&id)?;
90                output.write_all(&size)?;
91                output.write_all(&vec![0; payload])?;
92                return Ok(());
93            }
94        }
95    }
96    Err(Error::Invalid("void reserve cannot be encoded"))
97}

99pub(super) fn patch_duration<W: Write + Seek>(
100    output: &mut W,
101    position: u64,
102    duration: Duration,
103) -> Result<()> {
104    output.seek(SeekFrom::Start(position))?;
105    output.write_all(&(duration.as_secs_f64() * 1000.0).to_be_bytes())?;
106    Ok(())
107}

109pub(super) fn patch_seek_head<W: Write + Seek>(
110    output: &mut W,
111    position: u64,
112    base: &[Entry],
113    chapters: Option<u64>,
114    attachments: Option<u64>,
115) -> Result<()> {
116    let mut entries = base
117        .iter()
118        .map(|entry| Entry {
119            id: entry.id,
120            position: entry.position,
121        })
122        .collect::<Vec<_>>();
123    if let Some(position) = chapters {
124        entries.push(Entry {
125            id: 0x1043_a770,
126            position,
127        });
128    }
129    if let Some(position) = attachments {
130        entries.push(Entry {
131            id: 0x1941_a469,
132            position,
133        });
134    }
135    let encoded = seek::encode(&entries)?;
136    if encoded.len() + 2 > SEEK_RESERVE {
137        return Err(Error::Overflow("seek head reserve"));
138    }
139    output.seek(SeekFrom::Start(position))?;
140    output.write_all(&encoded)?;
141    write_void(output, SEEK_RESERVE - encoded.len())
142}
