3pub(crate) fn id_bytes(id: u32) -> Result<Vec<u8>> {
4    let width = match id {
5        0x80..=0xff => 1,
6        0x4000..=0x7fff => 2,
7        0x20_0000..=0x3f_ffff => 3,
8        0x1000_0000..=0x1fff_ffff => 4,
9        _ => return Err(Error::Invalid("invalid EBML element id")),
10    };
11    Ok(id.to_be_bytes()[4 - width..].to_vec())
12}

14pub(crate) fn size_bytes(value: u64, width: Option<usize>) -> Result<Vec<u8>> {
15    let width = match width {
16        Some(width @ 1..=8) => width,
17        Some(_) => return Err(Error::Invalid("invalid EBML size width")),
18        None => (1..=8)
19            .find(|width| value < (1_u64 << (7 * width)) - 1)
20            .ok_or(Error::Overflow("EBML size"))?,
21    };
22    let maximum = (1_u64 << (7 * width)) - 1;
23    if value >= maximum {
24        return Err(Error::Overflow("EBML size"));
25    }
26    let encoded = value | (1_u64 << (7 * width));
27    Ok(encoded.to_be_bytes()[8 - width..].to_vec())
28}

34pub(crate) fn unsigned(value: u64) -> Vec<u8> {
35    let bytes = value.to_be_bytes();
36    let first = bytes.iter().position(|byte| *byte != 0).unwrap_or(7);
37    bytes[first..].to_vec()
38}

41pub(crate) fn signed(value: i64) -> Vec<u8> {
42    let bytes = value.to_be_bytes();
43    let mut first = 0;
44    while first < 7 {
45        let redundant_zero = bytes[first] == 0 && bytes[first + 1] & 0x80 == 0;
46        let redundant_one = bytes[first] == 0xff && bytes[first + 1] & 0x80 != 0;
47        if !(redundant_zero || redundant_one) {
48            break;
49        }
50        first += 1;
51    }
52    bytes[first..].to_vec()
53}
