15/// One compressed packet assigned to an output track.
16#[derive(Clone, Debug, PartialEq, Eq)]
17pub struct Packet {
18    /// One-based output track number.
19    pub track_number: u64,
20    /// Signed decode timestamp in milliseconds, used for input interleaving.
21    pub decode_time_ms: i64,
22    /// Signed presentation timestamp in milliseconds.
23    pub presentation_time_ms: i64,
24    /// Packet duration used to derive segment duration.
25    pub duration: Duration,
26    /// Whether this packet is a random-access point.
27    pub keyframe: bool,
28    /// Exact compressed payload.
29    pub data: Vec<u8>,
30}

32/// Segment-level muxing metadata.
33#[derive(Clone, Debug, PartialEq, Eq)]
34pub struct MuxOptions {
35    /// Segment title.

37    /// Muxing application identifier.
38    pub muxing_application: String,

41}

53/// Streaming Matroska writer.
54pub struct Muxer;
55
56impl Muxer {
57    /// Write a complete seekable Matroska segment.
58    ///
59    /// Packets must be ordered by nondecreasing decode time. Presentation time
60    /// may differ, including for AVC B-frames.
61    ///
62    /// # Errors
63    ///
64    /// Returns an error for unsupported metadata, invalid ordering/timestamps,
65    /// overflow, or output I/O failure.
66    pub fn write<W, I>(
67        output: &mut W,
68        tracks: &[Track],
69        packets: I,
70        chapters: &[Chapter],
71        attachments: &[Attachment],
72        options: &MuxOptions,
73    ) -> Result<()>
74    where
75        W: Write + Seek,
76        I: IntoIterator<Item = Packet>,
77    {
78        Self::write_fallible(
79            output,
80            tracks,
81            packets.into_iter().map(Ok),
82            chapters,
83            attachments,
84            options,
85        )
86    }
87
88    /// Write from a fallible streaming packet source.
89    ///
90    /// This variant lets a demuxer surface an input error without first
91    /// collecting the complete media stream.
92    ///
93    /// # Errors
94    ///
95    /// Returns the first packet-source, validation, overflow, or I/O error.
96    pub fn write_fallible<W, I>(
97        output: &mut W,
98        tracks: &[Track],
99        packets: I,
100        chapters: &[Chapter],
101        attachments: &[Attachment],
102        options: &MuxOptions,
103    ) -> Result<()>
104    where
105        W: Write + Seek,
106        I: IntoIterator<Item = Result<Packet>>,
107    {
108        validate_tracks(tracks)?;
109        output.write_all(&ebml_header()?)?;
110        output.write_all(&id_bytes(0x1853_8067)?)?;
111        let segment_size_position = output.stream_position()?;
112        output.write_all(&size_bytes(0, Some(8))?)?;
113        let segment_start = output.stream_position()?;
114        let seek_position = output.stream_position()?;
115        write_void(output, SEEK_RESERVE)?;
116
117        let info_position = relative(output.stream_position()?, segment_start)?;
118        let duration_payload_position = write_info(output, options)?;
119        let tracks_position = relative(output.stream_position()?, segment_start)?;
120        output.write_all(&track::encode(tracks)?)?;
121        let chapters_position = if let Some(encoded) = crate::chapter::encode(chapters)? {
122            let position = relative(output.stream_position()?, segment_start)?;
123            output.write_all(&encoded)?;
124            Some(position)
125        } else {
126            None
127        };
128        let attachments_position = if let Some(encoded) = crate::attachment::encode(attachments)? {
129            let position = relative(output.stream_position()?, segment_start)?;
130            output.write_all(&encoded)?;
131            Some(position)
132        } else {
133            None
134        };
135
136        let primary_video = tracks
137            .iter()
138            .enumerate()
139            .find(|(_, track)| track.track_type == TrackType::Video)
140            .map(|(i, t)| effective_number(i, t));
141        let mut cues = Vec::new();
142        let mut cluster_time = None;
143        let mut last_cluster_time = 0_i64;
144        let mut cluster_payload = Vec::new();
145        let mut cluster_position = 0;
146        let mut last_decode = i64::MIN;
147        let mut maximum_end_ms = 0_i64;
148        let mut saw_packet = false;
149
150        for packet in packets {
151            let packet = packet?;
152            let track = tracks.iter().enumerate().find_map(|(index, track)| {
153                (effective_number(index, track) == packet.track_number).then_some(track)
154            });
155            if packet.track_number == 0 || track.is_none() || packet.data.is_empty() {
156                return Err(Error::Invalid(
157                    "packet references an invalid track or empty payload",
158                ));
159            }
160            if saw_packet && packet.decode_time_ms < last_decode {
161                return Err(Error::Invalid("packets are not ordered by decode time"));
162            }
163            saw_packet = true;
164            last_decode = packet.decode_time_ms;
165            let pts_ms = packet.presentation_time_ms;
166            let needs_cluster = cluster_time.is_none_or(|base| {
167                let relative = i128::from(pts_ms) - i128::from(base);
168                !(-32_768..=32_767).contains(&relative)
169            });
170            if needs_cluster {
171                if cluster_time.is_some() {
172                    write_cluster(output, &cluster_payload)?;
173                }
174                cluster_payload.clear();
175                cluster_position = relative(output.stream_position()?, segment_start)?;
176                let base = pts_ms.max(0).max(last_cluster_time);
177                let relative = i128::from(pts_ms) - i128::from(base);
178                if !(-32_768..=32_767).contains(&relative) {
179                    return Err(Error::Invalid(
180                        "presentation timestamp is too negative for a cluster",
181                    ));
182                }
183                cluster_time = Some(base);
184                last_cluster_time = base;
185                cluster_payload.extend(uint(0xe7, base as u64)?);
186            }
187            let base = cluster_time.expect("cluster initialized");
188            let relative_ms = i16::try_from(i128::from(pts_ms) - i128::from(base))
189                .map_err(|_| Error::Overflow("block relative timecode"))?;
190            let is_primary_keyframe = Some(packet.track_number) == primary_video && packet.keyframe;
191            if is_primary_keyframe && let Ok(time_ms) = u64::try_from(pts_ms) {
192                cues.push(Cue {
193                    time_ms,
194                    track: packet.track_number,
195                    cluster_position,
196                });
197            }
198            cluster_payload.extend(
199                if track.expect("track validated").track_type == TrackType::Subtitle {
200                    crate::block::group(
201                        packet.track_number,
202                        relative_ms,
203                        packet.duration,
204                        &packet.data,
205                    )?
206                } else {
207                    crate::block::simple(
208                        packet.track_number,
209                        relative_ms,
210                        packet.keyframe,
211                        &packet.data,
212                    )?
213                },
214            );
215            let duration_ms = i64::try_from(packet.duration.as_millis())
216                .map_err(|_| Error::Overflow("packet duration"))?;
217            maximum_end_ms = maximum_end_ms.max(
218                pts_ms
219                    .checked_add(duration_ms)
220                    .ok_or(Error::Overflow("packet end"))?,
221            );
222            if cluster_payload.len() >= 4 * 1024 * 1024 {
223                write_cluster(output, &cluster_payload)?;
224                cluster_time = None;
225                cluster_payload.clear();
226            }
227        }
228        if cluster_time.is_some() {
229            write_cluster(output, &cluster_payload)?;
230        }
231        if !saw_packet {
232            return Err(Error::Invalid("no media packets"));
233        }
234
235        let cues_position = relative(output.stream_position()?, segment_start)?;
236        output.write_all(&cue::encode(&cues)?)?;
237        let end = output.stream_position()?;
238        patch_duration(
239            output,
240            duration_payload_position,
241            Duration::from_millis(maximum_end_ms.max(0) as u64),
242        )?;
243        patch_seek_head(
244            output,
245            seek_position,
246            &[
247                Entry {
248                    id: 0x1549_a966,
249                    position: info_position,
250                },
251                Entry {
252                    id: 0x1654_ae6b,
253                    position: tracks_position,
254                },
255                Entry {
256                    id: 0x1c53_bb6b,
257                    position: cues_position,
258                },
259            ],
260            chapters_position,
261            attachments_position,
262        )?;
263        patch_segment_size(
264            output,
265            segment_size_position,
266            end.checked_sub(segment_start)
267                .ok_or(Error::Overflow("segment size"))?,
268        )?;
269        output.seek(SeekFrom::Start(end))?;
270        output.flush()?;
271        Ok(())
272    }
273}
