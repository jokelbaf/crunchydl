1//! Structured progress events and warnings emitted during a download.
2//!
3//! Library code never writes to a terminal. Progress is reported through
4//! [`DownloadEvent`] values delivered to an [`EventSink`], and non-fatal
5//! deviations from the requested plan are reported as [`DownloadWarning`] values.

13/// A non-fatal deviation from the exact requested plan.
14///
15/// The library never silently falls back to a lower quality, a different
16/// language, or a hardsub. Whenever a configured fallback policy is used, the
17/// corresponding warning is recorded and returned to the caller.
18#[non_exhaustive]
19#[derive(Clone, Debug, PartialEq, Eq)]
20pub enum DownloadWarning {
21    /// A requested audio locale was unavailable and a fallback locale was used.
22    AudioFallback {
23        /// The audio locale the caller asked for.
24        requested: Locale,
25        /// The audio locale that was selected instead.
26        used: Locale,
27    },
28    /// A requested audio locale was unavailable and dropped from a multi-locale
29    /// selection.
30    AudioLocaleUnavailable {
31        /// The audio locale the caller asked for.
32        requested: Locale,
33    },
34    /// A requested subtitle locale was not present in any playback version.
35    SubtitleUnavailable {
36        /// The subtitle locale the caller asked for.
37        requested: Locale,
38    },
39    /// A requested hardsub locale was not available for the stream.
40    HardSubUnavailable {
41        /// The hardsub locale the caller asked for.
42        requested: Locale,
43    },
44    /// The requested video quality was unavailable and a fallback was chosen.
45    VideoQualityFallback {
46        /// A human-readable description of the requested quality.
47        requested: String,
48        /// A human-readable description of the quality that was selected.
49        used: String,
50    },
51    /// The requested audio quality was unavailable and a fallback was chosen.
52    AudioQualityFallback {
53        /// A human-readable description of the requested quality.
54        requested: String,
55        /// A human-readable description of the quality that was selected.
56        used: String,
57    },
58    /// A referenced subtitle font could not be resolved under warning policy.
59    MissingFont {
60        /// Canonical family name referenced by the ASS document.
61        family: String,
62    },
63}
64
65impl fmt::Display for DownloadWarning {
66    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
67        match self {
68            Self::AudioFallback { requested, used } => {
69                write!(f, "audio locale {requested} unavailable, using {used}")
70            }
71            Self::AudioLocaleUnavailable { requested } => {
72                write!(f, "audio locale {requested} unavailable")
73            }
74            Self::SubtitleUnavailable { requested } => {
75                write!(f, "subtitle locale {requested} unavailable")
76            }
77            Self::HardSubUnavailable { requested } => {
78                write!(f, "hardsub locale {requested} unavailable")
79            }
80            Self::VideoQualityFallback { requested, used } => {
81                write!(f, "video quality {requested} unavailable, using {used}")
82            }
83            Self::AudioQualityFallback { requested, used } => {
84                write!(f, "audio quality {requested} unavailable, using {used}")
85            }
86            Self::MissingFont { family } => write!(f, "referenced font {family} was not found"),
87        }
88    }
89}

91/// The orchestration state of a download job.
92///
93/// The states mirror the internal job state machine. Every non-terminal state
94/// may transition to [`JobState::Cancelled`] or [`JobState::Failed`].
95#[non_exhaustive]
96#[derive(Clone, Copy, Debug, PartialEq, Eq)]
97pub enum JobState {
98    /// The job has been created but not started.
99    Created,
100    /// Requested media is being resolved and validated.
101    ResolvingMedia,
102    /// Playback sessions are being opened for the selected versions.
103    OpeningPlaybackSessions,
104    /// An immutable track plan is being built.
105    PlanningTracks,
106    /// Content decryption licenses are being acquired.
107    AcquiringLicenses,
108    /// Media segments are being downloaded.
109    Downloading,
110    /// Downloaded fragments are being decrypted and assembled.
111    Decrypting,
112    /// Subtitles and captions are being processed.
113    ProcessingSubtitles,
114    /// Tracks are being muxed into the output container.
115    Muxing,
116    /// The output file is being verified.
117    Verifying,
118    /// The verified output is being committed atomically.
119    Committing,
120    /// The job finished successfully.
121    Completed,
122    /// The job was cancelled by the caller.
123    Cancelled,
124    /// The job failed.
125    Failed,
126}

128/// A structured event describing download progress.
129#[non_exhaustive]
130#[derive(Clone, Debug)]
131pub enum DownloadEvent {
132    /// The job transitioned to a new [`JobState`].
133    StateChanged(JobState),
134    /// A non-fatal [`DownloadWarning`] was recorded.
135    Warning(DownloadWarning),
136    /// One segment was committed atomically to staging.
137    SegmentCompleted {
138        /// Requested media id.
139        media_id: String,
140        /// Playback version that owns this representation.
141        version_id: String,
142        /// Stable redacted representation identity.
143        representation_fingerprint: String,
144        /// Track role when supplied by the orchestrator.
145        track: Option<TransferTrack>,
146        /// Zero-based media-segment index.
147        index: usize,
148        /// Number of media segments now complete.
149        completed: usize,
150        /// Total number of media segments in the representation.
151        total: usize,
152        /// Bytes committed by this segment.
153        bytes: u64,
154        /// Bytes committed for this representation, including initialization.
155        completed_bytes: u64,
156        /// Expected representation bytes when every request declared a length.
157        total_bytes: Option<u64>,
158    },
159    /// A retryable transfer failed and will be attempted again.
160    TransferRetry {
161        /// Requested media id.
162        media_id: String,
163        /// Playback version that owns this representation.
164        version_id: String,
165        /// Stable redacted representation identity.
166        representation_fingerprint: String,
167        /// Track role when supplied by the orchestrator.
168        track: Option<TransferTrack>,
169        /// Zero-based media-segment index, or [`None`] for initialization.
170        index: Option<usize>,
171        /// Next one-based attempt number.
172        attempt: u32,
173        /// Delay before the next attempt.
174        delay: std::time::Duration,
175    },
176    /// Bounded item progress within a non-transfer pipeline stage.
177    StageProgress {
178        /// Active job state.
179        state: JobState,
180        /// Number of items completed in this stage.
181        completed: usize,
182        /// Total number of items in this stage.
183        total: usize,
184    },
185    /// A verified output was committed successfully.
186    OutputCommitted {
187        /// Final output path.
188        output: std::path::PathBuf,
189    },
190}

201/// A sink for [`DownloadEvent`] values.
202///
203/// Any `Fn(DownloadEvent)` closure implements this trait, so callers can pass a
204/// closure directly instead of defining a type.
205pub trait EventSink: Send + Sync {
206    /// Handle a single [`DownloadEvent`].
207    fn emit(&self, event: DownloadEvent);
208}
209
210impl<F> EventSink for F
211where
212    F: Fn(DownloadEvent) + Send + Sync,
213{
214    fn emit(&self, event: DownloadEvent) {
215        self(event);
216    }
217}
218
219impl<T: EventSink + ?Sized> EventSink for std::sync::Arc<T> {
220    fn emit(&self, event: DownloadEvent) {
221        (**self).emit(event);
222    }
223}

225/// An [`EventSink`] that discards every event.
226#[derive(Clone, Copy, Debug, Default)]
227pub struct NoopSink;
228
229impl EventSink for NoopSink {
230    fn emit(&self, _event: DownloadEvent) {}
