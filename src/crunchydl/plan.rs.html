49/// Safe, immutable diagnostics for one selected representation.
50#[derive(Clone, Debug, PartialEq, Eq, serde::Deserialize, serde::Serialize)]
51pub struct PlannedTrack {
52    /// Video or audio.
53    pub kind: PlannedTrackKind,
54    /// Version content id that supplied this track.
55    pub version_id: String,
56    /// Audio locale for audio tracks and the source version for video tracks.
57    pub locale: Locale,
58    /// Codec descriptor from the MPD.
59    pub codec: String,
60    /// Representation bandwidth.
61    pub bandwidth: u64,
62    /// Video dimensions when this is a video track.
63    pub dimensions: Option<(u32, u32)>,
64    /// Audio sampling rate when this is an audio track.
65    pub sampling_rate: Option<u32>,
66    /// Number of media segments, excluding initialization.
67    pub segment_count: usize,
68    /// Stable descriptor that excludes signed URL query values.
69    pub representation_fingerprint: String,
70    /// Whether the representation carries DRM metadata.
71    pub encrypted: bool,
72}

154pub(crate) async fn prepare<A: CrunchyrollApi>(
155    api: &A,
156    platform: &crunchyroll_rs::media::StreamPlatform,
157    media: &ResolvedMedia,
158    options: &PlanningOptions,
159    cancellation: &CancellationToken,
160) -> Result<PreparedPlan, Error> {
161    let mut guard = SessionGuard::new(api);
162    let result = prepare_with_guard(api, platform, media, options, cancellation, &mut guard).await;
163    let finalization = guard.finalize().await;
164    match (result, finalization) {
165        (Ok(plan), Ok(())) => Ok(plan),
166        (Ok(_), Err(error)) => Err(error),
167        (Err(error), _) => Err(error),
168    }
169}

171pub(crate) async fn prepare_with_guard<A: CrunchyrollApi>(
172    api: &A,
173    platform: &crunchyroll_rs::media::StreamPlatform,
174    media: &ResolvedMedia,
175    options: &PlanningOptions,
176    cancellation: &CancellationToken,
177    guard: &mut SessionGuard<'_, A>,
178) -> Result<PreparedPlan, Error> {
179    let audio_plan = select_audio(media, &options.audio, options.fallback)?;
180    let mut warnings = audio_plan.warnings;
181    let mut sources = Vec::new();
182    let mut all_subtitles = Vec::new();
183    let mut selected_audio_locales = Vec::new();
184
185    for version in audio_plan.versions {
186        let session_index = guard
187            .open(&version.content_id, platform, cancellation)
188            .await?;
189        let metadata = guard.metadata(session_index);
190        let hardsub = select_hardsub(&metadata.hardsubs, &options.hardsub)?;
191        let data = guard
192            .stream_data(session_index, hardsub, cancellation)
193            .await?;
194        selected_audio_locales.push(metadata.audio_locale.clone());
195        all_subtitles.extend(metadata.subtitles);
196
197        let video =
198            choose_video(&version.content_id, &metadata.audio_locale, &data, options).await?;
199        if let Some(warning) = video.1 {
200            warnings.push(warning);
201        }
202        let keep_video = options.video == VideoSelection::PerVersion
203            || !sources.iter().any(|source: &PlannedSource| {
204                source.diagnostic.kind == PlannedTrackKind::Video
205                    && source.diagnostic.representation_fingerprint
206                        == video.0.diagnostic.representation_fingerprint
207            });
208        if keep_video {
209            sources.push(video.0);
210        }
211
212        let audio =
213            choose_audio(&version.content_id, &metadata.audio_locale, &data, options).await?;
214        if let Some(warning) = audio.1 {
215            warnings.push(warning);
216        }
217        sources.push(audio.0);
218    }
219
220    let (subtitles, subtitle_sources) = plan_subtitles(
221        all_subtitles,
222        &selected_audio_locales,
223        options,
224        &mut warnings,
225    );
226    let chapters = match options.chapters {
227        ChapterSelection::Disabled => Vec::new(),
228        ChapterSelection::SkipEvents => from_skip_events(
229            api.skip_events(&media.content_id, media.kind).await?,
230            media.duration,
231        ),
232    };
233    let tracks = sources
234        .iter()
235        .map(|source| source.diagnostic.clone())
236        .collect::<Vec<_>>();
237    let plan_fingerprint = plan_fingerprint(&media.content_id, &tracks, &subtitles);
238    Ok(PreparedPlan {
239        public: DownloadPlan {
240            media_id: media.content_id.clone(),
241            tracks,
242            subtitles,
243            chapters,
244            warnings,
245            fingerprint: plan_fingerprint,
246        },
247        sources,
248        subtitles: subtitle_sources,
249    })
250}

252async fn choose_video(
253    version_id: &str,
254    locale: &Locale,
255    data: &StreamData,
256    options: &PlanningOptions,
257) -> Result<(PlannedSource, Option<DownloadWarning>), Error> {
258    let candidates = data
259        .video
260        .iter()
261        .map(|video| VideoQualityCandidate {
262            width: u32::try_from(video.resolution.width).unwrap_or(u32::MAX),
263            height: u32::try_from(video.resolution.height).unwrap_or(u32::MAX),
264            bandwidth: video.bandwidth,
265        })
266        .collect::<Vec<_>>();
267    let choice = select_video_quality(&candidates, &options.video_quality, options.fallback)?;
268    Ok((
269        video_source(version_id, locale, &data.video[choice.index]).await?,
270        choice.warning,
271    ))
272}

335#[allow(clippy::too_many_arguments)]
336fn source_from_segments(
337    kind: PlannedTrackKind,
338    version_id: &str,
339    locale: &Locale,
340    codec: String,
341    bandwidth: u64,
342    dimensions: Option<(u32, u32)>,
343    sampling_rate: Option<u32>,
344    drm: Option<MediaStreamDRM>,
345    segments: Vec<crunchyroll_rs::media::StreamSegment>,
346) -> Result<PlannedSource, Error> {
347    let mut segments = segments.into_iter().map(|segment| SourceSegment {
348        url: segment.url,
349        range: segment.range,
350        duration: segment.length,
351    });
352    let init = segments.next().ok_or_else(|| {
353        Error::Manifest("representation has no initialization segment".to_string())
354    })?;
355    let media = segments.collect::<Vec<_>>();
356    let mut identity = vec![
357        version_id.to_string(),
358        format!("{kind:?}"),
359        codec.clone(),
360        bandwidth.to_string(),
361        format!("{dimensions:?}"),
362        format!("{sampling_rate:?}"),
363        url_identity(&init.url),
364        format!("{:?}", init.range),
365    ];
366    identity.extend(media.iter().flat_map(|segment| {
367        [
368            url_identity(&segment.url),
369            format!("{:?}", segment.range),
370            segment.duration.as_millis().to_string(),
371        ]
372    }));
373    let representation_fingerprint = fingerprint(identity.iter());
374    Ok(PlannedSource {
375        diagnostic: PlannedTrack {
376            kind,
377            version_id: version_id.to_string(),
378            locale: locale.clone(),
379            codec,
380            bandwidth,
381            dimensions,
382            sampling_rate,
383            segment_count: media.len(),
384            representation_fingerprint,
385            encrypted: drm.is_some(),
386        },
387        init,
388        media,
389        drm,
390    })
391}

393fn plan_subtitles(
394    subtitles: Vec<ApiSubtitle>,
395    audio_locales: &[Locale],
396    options: &PlanningOptions,
397    warnings: &mut Vec<DownloadWarning>,
398) -> (Vec<PlannedSubtitle>, Vec<PreparedSubtitle>) {
399    let mut unique = Vec::<ApiSubtitle>::new();
400    for subtitle in subtitles {
401        let identity = url_identity(&subtitle.url);
402        if !unique.iter().any(|current| {
403            current.locale == subtitle.locale
404                && current.format == subtitle.format
405                && current.is_caption == subtitle.is_caption
406                && url_identity(&current.url) == identity
407        }) {
408            unique.push(subtitle);
409        }
410    }
411    let infos = unique
412        .iter()
413        .map(|subtitle| SubtitleTrackInfo {
414            locale: subtitle.locale.clone(),
415            is_caption: subtitle.is_caption,
416            format: subtitle.format.clone(),
417        })
418        .collect::<Vec<_>>();
419    let selected = select_subtitles(&infos, &options.subtitles);
420    warnings.extend(selected.warnings);
421    let mut prepared = selected
422        .tracks
423        .into_iter()
424        .filter_map(|index| unique.get(index))
425        .filter(|subtitle| {
426            options.subtitles.include_signs
427                || subtitle.is_caption
428                || !audio_locales.contains(&subtitle.locale)
429        })
430        .map(|subtitle| {
431            let is_signs = !subtitle.is_caption && audio_locales.contains(&subtitle.locale);
432            let language = crate::locale_display_name(&subtitle.locale);
433            let title = if subtitle.is_caption {
434                format!("{language} (CC)")
435            } else if is_signs {
436                format!("{language} (Signs)")
437            } else {
438                language
439            };
440            let diagnostic = PlannedSubtitle {
441                locale: subtitle.locale.clone(),
442                format: subtitle.format.clone(),
443                is_caption: subtitle.is_caption,
444                is_signs,
445                resource_identity: url_identity(&subtitle.url),
446                title,
447                default: false,
448                forced: is_signs,
449            };
450            PreparedSubtitle {
451                diagnostic,
452                url: subtitle.url.clone(),
453            }
454        })
455        .collect::<Vec<_>>();
456    // When the user asks for a specific subtitle language, make the first
457    // matching non-forced track player-visible by default. Tracks selected via
458    // "all" remain opt-in so multilingual downloads do not pick an arbitrary
459    // language.
460    if matches!(options.subtitles.locales, SubtitleLocales::Explicit(_))
461        && let Some(subtitle) = prepared
462            .iter_mut()
463            .find(|subtitle| !subtitle.diagnostic.forced)
464    {
465        subtitle.diagnostic.default = true;
466    }
467    let diagnostics = prepared
468        .iter()
469        .map(|subtitle| subtitle.diagnostic.clone())
470        .collect();
471    (diagnostics, prepared)
472}

474fn plan_fingerprint(
475    media_id: &str,
476    tracks: &[PlannedTrack],
477    subtitles: &[PlannedSubtitle],
478) -> String {
479    let mut parts = vec![media_id.to_string()];
480    parts.extend(tracks.iter().map(|track| {
481        format!(
482            "{:?}|{}|{}|{}|{}|{:?}|{:?}|{}|{}",
483            track.kind,
484            track.version_id,
485            track.locale,
486            track.codec,
487            track.bandwidth,
488            track.dimensions,
489            track.sampling_rate,
490            track.segment_count,
491            track.representation_fingerprint
492        )
493    }));
494    parts.extend(subtitles.iter().map(|subtitle| {
495        format!(
496            "{}|{}|{}|{}|{}",
497            subtitle.locale,
498            subtitle.format,
499            subtitle.is_caption,
500            subtitle.is_signs,
501            subtitle.resource_identity
502        )
503    }));
504    fingerprint(parts.iter())
505}

512    fn subtitle_resources_deduplicate_without_query_secrets_and_keep_signs_metadata() {
513        let subtitles = vec![
514            ApiSubtitle {
515                locale: Locale::ja_JP,
516                url: "https://cdn.test/signs.ass?token=one".to_string(),
517                format: "ass".to_string(),
518                is_caption: false,
519            },
520            ApiSubtitle {
521                locale: Locale::ja_JP,
522                url: "https://cdn.test/signs.ass?token=two".to_string(),
523                format: "ass".to_string(),
524                is_caption: false,
525            },
526        ];
527        let mut warnings = Vec::new();
528        let (diagnostics, prepared) = plan_subtitles(
529            subtitles,
530            &[Locale::ja_JP],
531            &PlanningOptions::default(),
532            &mut warnings,
533        );
534        assert_eq!(diagnostics.len(), 1);
535        assert_eq!(prepared.len(), 1);
536        assert!(diagnostics[0].is_signs && diagnostics[0].forced);
537        assert_eq!(diagnostics[0].title, "Japanese (Signs)");
538        assert_eq!(
539            diagnostics[0].resource_identity,
540            "https://cdn.test/signs.ass"
541        );
542        assert!(!format!("{:?}", diagnostics).contains("token="));
543    }
