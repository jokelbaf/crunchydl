12/// Catalog media kinds, including non-downloadable parent collections.
13#[non_exhaustive]
14#[derive(Clone, Copy, Debug, PartialEq, Eq, serde::Deserialize, serde::Serialize)]
15pub enum CatalogKind {
16    /// An anime series.
17    Series,
18    /// One season of a series.
19    Season,
20    /// A downloadable episode.
21    Episode,
22    /// A collection containing one or more movies.
23    MovieListing,
24    /// A downloadable movie.
25    Movie,
26    /// A downloadable music video.
27    MusicVideo,
28    /// A concert catalog item not supported by the downloader yet.
29    Concert,
30}

44/// A safe catalog image descriptor.
45#[derive(Clone, Debug, PartialEq, Eq, serde::Deserialize, serde::Serialize)]
46pub struct CatalogImage {

57/// User-rating summary returned by the catalog.
58#[non_exhaustive]
59#[derive(Clone, Debug, PartialEq, serde::Deserialize, serde::Serialize)]
60#[serde(tag = "kind", rename_all = "snake_case")]
61pub enum CatalogRating {
62    /// A one-to-five-star rating used for series and movie listings.
63    Stars {
64        /// Average number of stars.
65        average: f64,
66        /// Number of submitted ratings when reported.
67        total: Option<u32>,
68    },
69    /// Positive-vote percentage used for episodes.
70    Approval {
71        /// Approximate percentage of positive votes.
72        percentage: f64,
73        /// Number of submitted votes when reported.
74        total: Option<u32>,
75    },
76}

80/// Frontend-facing metadata for one catalog result or detail page.
81#[derive(Clone, Debug, PartialEq, serde::Deserialize, serde::Serialize)]
82pub struct CatalogItem {
83    /// Modern Crunchyroll catalog id.
84    pub id: String,
85    /// Media or collection kind.
86    pub kind: CatalogKind,
87    /// Download target when the item itself is downloadable.
88    pub target: Option<MediaTarget>,
89    /// Localized title.
90    pub title: String,
91    /// Localized short description.
92    pub description: String,
93    /// Localized extended description when supplied separately.
94    pub extended_description: Option<String>,
95    /// Artwork ordered as returned by the service.
96    pub images: Vec<CatalogImage>,
97    /// User rating when included by the endpoint or loaded for a detail page.
98    pub rating: Option<CatalogRating>,
99    /// Release or launch year when known.
100    pub release_year: Option<u32>,
101    /// Season number for season and episode rows when known.
102    pub season_number: Option<u32>,
103    /// Service-provided episode label, including decimal and special labels.
104    pub episode_number: Option<String>,
105    /// Number of seasons when this is a series.
106    pub season_count: Option<u32>,
107    /// Number of episodes when reported.
108    pub episode_count: Option<u32>,
109    /// Duration in milliseconds for playable items.
110    pub duration_millis: Option<u64>,
111    /// Whether some or all of the item is premium-only.
112    pub premium_only: bool,
113    /// Whether subtitle metadata is advertised.
114    pub is_subbed: bool,
115    /// Whether dubbed metadata is advertised.
116    pub is_dubbed: bool,
117    /// Audio locales advertised without opening playback.
118    pub audio_locales: Vec<Locale>,
119    /// Subtitle locales advertised without opening playback.
120    pub subtitle_locales: Vec<Locale>,
121}

129fn catalog_images(values: &[Image], kind: CatalogImageKind) -> Vec<CatalogImage> {
130    values
131        .iter()
132        .map(|image| CatalogImage {
133            source: image.source.clone(),
134            width: image.width,
135            height: image.height,
136            kind,
137        })
138        .collect()
139}

141fn largest_image<'a>(images: impl Iterator<Item = &'a CatalogImage>) -> Option<&'a CatalogImage> {
142    images.max_by_key(|image| {
143        (
144            u64::from(image.width) * u64::from(image.height),
145            image.width,
146            image.height,
147        )
148    })
149}

151fn episode_label(episode: &Episode) -> Option<String> {
152    non_empty(&episode.episode)
153        .or_else(|| episode.episode_number.map(|number| number.to_string()))
154        .or_else(|| {
155            (episode.sequence_number > 0.0).then(|| {
156                episode
157                    .sequence_number
158                    .to_string()
159                    .trim_end_matches(".0")
160                    .to_string()
161            })
162        })
163}

165fn unique_locales(locales: impl IntoIterator<Item = Locale>) -> Vec<Locale> {
166    let mut locales = locales
167        .into_iter()
168        .filter(|locale| !locale.to_string().is_empty())
169        .collect::<Vec<_>>();
170    locales.sort_by_key(ToString::to_string);
171    locales.dedup();
172    locales
173}

183fn vote_count(value: &serde_json::Value) -> Option<f64> {
184    let displayed = value.get("displayed")?.as_str()?.parse::<f64>().ok()?;
185    let multiplier = match value.get("unit").and_then(serde_json::Value::as_str) {
186        None | Some("") => 1.0,
187        Some("K") | Some("k") => 1_000.0,
188        Some("M") | Some("m") => 1_000_000.0,
189        Some(_) => return None,
190    };
191    Some(displayed * multiplier)
192}
