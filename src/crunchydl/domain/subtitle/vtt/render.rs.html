8pub(super) fn ass_style(
9    name: &str,
10    font: &str,
11    size: f32,
12    color: &str,
13    bold: bool,
14    italic: bool,
15    underline: bool,
16) -> String {
17    format!(
18        "Style: {name},{font},{size},{color},&H000000FF,&H00000000,&H00000000,{},{},{},0,100,100,0,0,1,2,0,2,20,20,30,1",
19        if bold { -1 } else { 0 },
20        if italic { -1 } else { 0 },
21        if underline { -1 } else { 0 }
22    )
23}

25pub(super) fn style_name(value: &str) -> String {
26    let mut name = String::from("Vtt_");
27    name.extend(value.chars().map(|character| {
28        if character.is_ascii_alphanumeric() {
29            character
30        } else {
31            '_'
32        }
33    }));
34    name
35}

37pub(super) fn cue_text_to_ass(text: &str, styles: &BTreeMap<String, CueStyle>) -> (String, String) {
38    let mut output = String::new();
39    let mut default_style = "Default".to_string();
40    let mut rest = text;
41    while let Some(open) = rest.find('<') {
42        output.push_str(&escape_ass_text(&rest[..open]));
43        let Some(close) = rest[open + 1..].find('>') else {
44            output.push_str(&escape_ass_text(&rest[open..]));
45            rest = "";
46            break;
47        };
48        let tag = &rest[open + 1..open + 1 + close];
49        match tag {
50            "b" => output.push_str("{\\b1}"),
51            "/b" => output.push_str("{\\b0}"),
52            "i" => output.push_str("{\\i1}"),
53            "/i" => output.push_str("{\\i0}"),
54            "u" => output.push_str("{\\u1}"),
55            "/u" => output.push_str("{\\u0}"),
56            "/c" => output.push_str("{\\r}"),
57            _ if tag.starts_with("c.") => {
58                let class = tag[2..].split('.').next().unwrap_or_default();
59                if let Some(style) = styles.get(class) {
60                    if output.is_empty() {
61                        default_style.clone_from(&style.name);
62                    } else {
63                        output.push_str(&format!("{{\\r{}}}", style.name));
64                    }
65                }
66            }
67            _ => {}
68        }
69        rest = &rest[open + close + 2..];
70    }
71    output.push_str(&escape_ass_text(rest));
72    (default_style, output.replace('\n', "\\N"))
73}
74
75fn escape_ass_text(value: &str) -> String {
76    value
77        .replace("&lt;", "<")
78        .replace("&gt;", ">")
79        .replace("&amp;", "&")
80        .replace('{', "\\{")
81        .replace('}', "\\}")
82}

84pub(super) fn cue_position(settings: &BTreeMap<String, String>) -> Result<String, Error> {
85    if settings.contains_key("vertical") || settings.contains_key("size") {
86        return Err(Error::Subtitle(
87            "vertical or sized WebVTT cues are outside the supported conversion profile"
88                .to_string(),
89        ));
90    }
91    if settings
92        .get("line")
93        .is_some_and(|line| !line.ends_with('%'))
94    {
95        return Err(Error::Subtitle(
96            "line-number WebVTT positioning is outside the supported conversion profile"
97                .to_string(),
98        ));
99    }
100    let x = settings
101        .get("position")
102        .map(|value| percentage(value, 1280))
103        .transpose()?
104        .unwrap_or(640);
105    let y = settings
106        .get("line")
107        .filter(|value| value.ends_with('%'))
108        .map(|value| percentage(value, 720))
109        .transpose()?
110        .unwrap_or(648);
111    let alignment = match settings.get("align").map(String::as_str) {
112        Some("start" | "left") => "\\an1",
113        Some("end" | "right") => "\\an3",
114        Some("center") | None => "",
115        Some(_) => {
116            return Err(Error::Subtitle(
117                "unsupported WebVTT cue alignment".to_string(),
118            ));
119        }
120    };
121    if settings.contains_key("position") || settings.contains_key("line") {
122        Ok(format!("{{{alignment}\\pos({x},{y})}}"))
123    } else if alignment.is_empty() {
124        Ok(String::new())
125    } else {
126        Ok(format!("{{{alignment}}}"))
127    }
128}

130fn percentage(value: &str, scale: u32) -> Result<u32, Error> {
131    let number = value
132        .trim_end_matches('%')
133        .split(',')
134        .next()
135        .unwrap_or_default()
136        .parse::<f64>()
137        .map_err(|_| Error::Subtitle("invalid WebVTT position".to_string()))?;
138    if !(0.0..=100.0).contains(&number) {
139        return Err(Error::Subtitle(
140            "WebVTT position outside viewport".to_string(),
141        ));
142    }
143    Ok((number * f64::from(scale) / 100.0).round() as u32)
144}

146pub(super) fn ass_timestamp(milliseconds: u64) -> String {
147    let centiseconds = (milliseconds + 5) / 10;
148    let hours = centiseconds / 360_000;
149    let minutes = centiseconds / 6_000 % 60;
150    let seconds = centiseconds / 100 % 60;
151    let fraction = centiseconds % 100;
152    format!("{hours}:{minutes:02}:{seconds:02}.{fraction:02}")
153}
