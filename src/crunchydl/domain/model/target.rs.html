1//! Stable media targets used by queues and frontends.
2
3use super::{MediaKind, ResolvedMedia};
4
5/// Stable, serializable identity of one downloadable media item.
6///
7/// Unlike [`crate::MediaRequest`], this type does not retain a `crunchyroll-rs`
8/// executor and can therefore be stored safely in a frontend queue. Resolve it
9/// against an authenticated [`crate::Downloader`] immediately before planning
10/// or downloading.
11#[non_exhaustive]
12#[derive(Clone, Debug, PartialEq, Eq, Hash, serde::Deserialize, serde::Serialize)]
13pub enum MediaTarget {
14    /// A single episode id.
15    Episode(String),
16    /// A single movie id.
17    Movie(String),
18    /// A single music-video id.
19    MusicVideo(String),
20}
21
22impl MediaTarget {
23    /// The kind of media identified by this target.
24    #[must_use]
25    pub fn kind(&self) -> MediaKind {
26        match self {
27            Self::Episode(_) => MediaKind::Episode,
28            Self::Movie(_) => MediaKind::Movie,
29            Self::MusicVideo(_) => MediaKind::MusicVideo,
30        }
31    }
32
33    /// The modern Crunchyroll content id.
34    #[must_use]
35    pub fn id(&self) -> &str {
36        match self {
37            Self::Episode(id) | Self::Movie(id) | Self::MusicVideo(id) => id,
38        }
39    }
40}
41
42impl From<&ResolvedMedia> for MediaTarget {
43    fn from(media: &ResolvedMedia) -> Self {
44        match media.kind {
45            MediaKind::Episode => Self::Episode(media.content_id.clone()),
46            MediaKind::Movie => Self::Movie(media.content_id.clone()),
47            MediaKind::MusicVideo => Self::MusicVideo(media.content_id.clone()),
48        }
49    }
50}
