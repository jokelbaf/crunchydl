3use crunchyroll_rs::media::{SkipEvents, SkipEventsEvent};

5/// A service-neutral chapter point.
6#[derive(Clone, Debug, PartialEq, Eq)]
7pub struct Chapter {
8    /// Chapter start relative to the media timeline.
9    pub start: Duration,
10    /// Stable, human-readable chapter name.
11    pub title: String,
12}
13
14pub(crate) fn from_skip_events(events: Option<SkipEvents>, duration: Duration) -> Vec<Chapter> {
15    let Some(events) = events else {
16        return Vec::new();
17    };
18    let mut ranges = Vec::new();
19    add_event(&mut ranges, "Recap", events.recap, duration, 0);
20    add_event(&mut ranges, "Intro", events.intro, duration, 1);
21    add_event(&mut ranges, "Credits", events.credits, duration, 2);
22    add_event(&mut ranges, "Preview", events.preview, duration, 3);
23    ranges.sort_by_key(|range| (range.start, range.priority));
24    let mut points = Vec::new();
25    let mut cursor = Duration::ZERO;
26    for range in ranges {
27        if range.start < cursor {
28            continue;
29        }
30        if range.start > cursor {
31            points.push(Chapter {
32                start: cursor,
33                title: "Episode".to_string(),
34            });
35        }
36        points.push(Chapter {
37            start: range.start,
38            title: range.title.to_string(),
39        });
40        cursor = range.end.max(range.start);
41    }
42    if cursor < duration {
43        points.push(Chapter {
44            start: cursor,
45            title: "Episode".to_string(),
46        });
47    }
48    points.dedup_by(|left, right| left.start == right.start);
49    points
50}

52struct ChapterRange {
53    start: Duration,
54    end: Duration,
55    title: &'static str,
56    priority: u8,
57}
58
59fn add_event(
60    ranges: &mut Vec<ChapterRange>,
61    title: &'static str,
62    event: Option<SkipEventsEvent>,
63    duration: Duration,
64    priority: u8,
65) {
66    if let Some(event) = event {
67        let start = Duration::from_secs_f32(event.start.max(0.0)).min(duration);
68        let end = Duration::from_secs_f32(event.end.max(0.0)).min(duration);
69        ranges.push(ChapterRange {
70            start,
71            end,
72            title,
73            priority,
74        });
75    }
76}
