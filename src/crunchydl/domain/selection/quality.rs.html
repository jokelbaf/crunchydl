7/// A neutral video representation for quality selection.
8#[derive(Clone, Copy, Debug, PartialEq, Eq)]
9pub struct VideoQualityCandidate {
10    /// Frame width in pixels.
11    pub width: u32,
12    /// Frame height in pixels.
13    pub height: u32,
14    /// Bandwidth in bits per second.
15    pub bandwidth: u64,
16}

18/// A neutral audio representation for quality selection.
19#[derive(Clone, Copy, Debug, PartialEq, Eq)]
20pub struct AudioQualityCandidate {
21    /// Bandwidth in bits per second.
22    pub bandwidth: u64,
23    /// Sampling rate in Hz.
24    pub sampling_rate: u32,
25}

27/// The outcome of a quality selection: an index into the candidate slice plus an
28/// optional fallback warning.
29#[derive(Clone, Debug, PartialEq, Eq)]
30pub struct QualityChoice {
31    /// The index of the chosen candidate.
32    pub index: usize,
33    /// A warning if a fallback quality was chosen.
34    pub warning: Option<DownloadWarning>,
35}

37fn video_quality_key(candidate: &VideoQualityCandidate) -> (u32, u64, u32) {
38    (candidate.height, candidate.bandwidth, candidate.width)
39}
40
41fn audio_quality_key(candidate: &AudioQualityCandidate) -> (u64, u32) {
42    (candidate.bandwidth, candidate.sampling_rate)
43}
44
45fn video_describe(candidate: &VideoQualityCandidate) -> String {
46    format!("{}p", candidate.height)
47}
48
49fn audio_describe(candidate: &AudioQualityCandidate) -> String {
50    format!("{} bps", candidate.bandwidth)
51}

53fn best_matching<T, K: Ord>(
54    items: &[T],
55    key: impl Fn(&T) -> K,
56    pred: impl Fn(&T) -> bool,
57) -> Option<usize> {
58    let mut best: Option<usize> = None;
59    for (index, item) in items.iter().enumerate() {
60        if !pred(item) {
61            continue;
62        }
63        match best {
64            Some(current) if key(&items[current]) >= key(item) => {}
65            _ => best = Some(index),
66        }
67    }
68    best
69}

71/// Select a video representation by [`QualitySelection`].
72///
73/// Video is ranked by height, then bandwidth, then width. Exact requests use
74/// the configured fallback policy.
75///
76/// # Errors
77///
78/// Returns [`SelectionError`] when no candidate exists or an exact request
79/// cannot be met under [`FallbackPolicy::Exact`].
80pub fn select_video_quality(
81    candidates: &[VideoQualityCandidate],
82    selection: &QualitySelection,
83    fallback: FallbackPolicy,
84) -> Result<QualityChoice, SelectionError> {
85    if candidates.is_empty() {
86        return Err(SelectionError::NoVideoRepresentations);
87    }
88
89    match *selection {
90        QualitySelection::Best => Ok(QualityChoice {
91            index: argmax(candidates, video_quality_key),
92            warning: None,
93        }),
94        QualitySelection::MaxHeight(max) => {
95            if let Some(index) = best_matching(candidates, video_quality_key, |c| c.height <= max) {
96                Ok(QualityChoice {
97                    index,
98                    warning: None,
99                })
100            } else {
101                let index = argmin(candidates, video_quality_key);
102                Ok(QualityChoice {
103                    index,
104                    warning: Some(DownloadWarning::VideoQualityFallback {
105                        requested: format!("<={max}p"),
106                        used: video_describe(&candidates[index]),
107                    }),
108                })
109            }
110        }
111        QualitySelection::ExactHeight(height) => resolve_video_target(
112            candidates,
113            |c| u64::from(c.height),
114            u64::from(height),
115            format!("{height}p"),
116            fallback,
117        ),
118        QualitySelection::ExactBandwidth(bandwidth) => resolve_video_target(
119            candidates,
120            |c| c.bandwidth,
121            bandwidth,
122            format!("{bandwidth} bps"),
123            fallback,
124        ),
125        QualitySelection::RankedIndex(rank) => {
126            let mut order: Vec<usize> = (0..candidates.len()).collect();
127            order.sort_by(|&a, &b| {
128                video_quality_key(&candidates[b]).cmp(&video_quality_key(&candidates[a]))
129            });
130            match order.get(rank) {
131                Some(&index) => Ok(QualityChoice {
132                    index,
133                    warning: None,
134                }),
135                None if fallback == FallbackPolicy::Exact => Err(SelectionError::IndexOutOfRange {
136                    index: rank,
137                    available: candidates.len(),
138                }),
139                None => {
140                    let index = order[order.len() - 1];
141                    Ok(QualityChoice {
142                        index,
143                        warning: Some(DownloadWarning::VideoQualityFallback {
144                            requested: format!("rank {rank}"),
145                            used: video_describe(&candidates[index]),
146                        }),
147                    })
148                }
149            }
150        }
151    }
152}

154fn resolve_video_target(
155    candidates: &[VideoQualityCandidate],
156    value: impl Fn(&VideoQualityCandidate) -> u64,
157    target: u64,
158    requested: String,
159    fallback: FallbackPolicy,
160) -> Result<QualityChoice, SelectionError> {
161    if let Some(index) = best_matching(candidates, video_quality_key, |c| value(c) == target) {
162        return Ok(QualityChoice {
163            index,
164            warning: None,
165        });
166    }
167
168    let index = match fallback {
169        FallbackPolicy::Exact => return Err(SelectionError::QualityUnavailable { requested }),
170        FallbackPolicy::NearestLower => {
171            best_matching(candidates, video_quality_key, |c| value(c) < target)
172                .unwrap_or_else(|| argmin(candidates, video_quality_key))
173        }
174        FallbackPolicy::BestAvailable => argmax(candidates, video_quality_key),
175    };
176    Ok(QualityChoice {
177        index,
178        warning: Some(DownloadWarning::VideoQualityFallback {
179            requested,
180            used: video_describe(&candidates[index]),
181        }),
182    })
183}

185/// Select an audio representation by [`QualitySelection`].
186///
187/// Audio is ranked by bandwidth, then sampling rate.
188///
189/// # Errors
190///
191/// Returns [`SelectionError`] when no candidate exists or an exact request
192/// cannot be met under [`FallbackPolicy::Exact`].
193pub fn select_audio_quality(
194    candidates: &[AudioQualityCandidate],
195    selection: &QualitySelection,
196    fallback: FallbackPolicy,
197) -> Result<QualityChoice, SelectionError> {
198    if candidates.is_empty() {
199        return Err(SelectionError::NoAudioRepresentations);
200    }
201
202    match *selection {
203        QualitySelection::ExactBandwidth(bandwidth) => {
204            if let Some(index) =
205                best_matching(candidates, audio_quality_key, |c| c.bandwidth == bandwidth)
206            {
207                return Ok(QualityChoice {
208                    index,
209                    warning: None,
210                });
211            }
212            let index = match fallback {
213                FallbackPolicy::Exact => {
214                    return Err(SelectionError::QualityUnavailable {
215                        requested: format!("{bandwidth} bps"),
216                    });
217                }
218                FallbackPolicy::NearestLower => {
219                    best_matching(candidates, audio_quality_key, |c| c.bandwidth < bandwidth)
220                        .unwrap_or_else(|| argmin(candidates, audio_quality_key))
221                }
222                FallbackPolicy::BestAvailable => argmax(candidates, audio_quality_key),
223            };
224            Ok(QualityChoice {
225                index,
226                warning: Some(DownloadWarning::AudioQualityFallback {
227                    requested: format!("{bandwidth} bps"),
228                    used: audio_describe(&candidates[index]),
229                }),
230            })
231        }
232        QualitySelection::RankedIndex(rank) => {
233            let mut order: Vec<usize> = (0..candidates.len()).collect();
234            order.sort_by(|&a, &b| {
235                audio_quality_key(&candidates[b]).cmp(&audio_quality_key(&candidates[a]))
236            });
237            match order.get(rank) {
238                Some(&index) => Ok(QualityChoice {
239                    index,
240                    warning: None,
241                }),
242                None if fallback == FallbackPolicy::Exact => Err(SelectionError::IndexOutOfRange {
243                    index: rank,
244                    available: candidates.len(),
245                }),
246                None => {
247                    let index = order[order.len() - 1];
248                    Ok(QualityChoice {
249                        index,
250                        warning: Some(DownloadWarning::AudioQualityFallback {
251                            requested: format!("rank {rank}"),
252                            used: audio_describe(&candidates[index]),
253                        }),
254                    })
255                }
256            }
257        }
258        _ => Ok(QualityChoice {
259            index: argmax(candidates, audio_quality_key),
260            warning: None,
261        }),
262    }
263}

265fn argmax<T, K: Ord>(items: &[T], key: impl Fn(&T) -> K) -> usize {
266    let mut best = 0;
267    let mut best_key = key(&items[0]);
268    for (index, item) in items.iter().enumerate().skip(1) {
269        let candidate_key = key(item);
270        if candidate_key > best_key {
271            best = index;
272            best_key = candidate_key;
273        }
274    }
275    best
276}
