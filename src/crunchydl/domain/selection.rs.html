1//! Track selection policies and the logic that resolves them against available
2//! options.
3//!
4//! Selection never silently degrades. When a configured [`FallbackPolicy`] causes
5//! a deviation from the exact request, the resolver records a
6//! [`DownloadWarning`]; when no fallback is permitted, it returns a
7//! [`SelectionError`] that lists what was available.

26/// The resolved set of audio versions plus any fallback warnings.
27#[derive(Clone, Debug, Default, PartialEq)]
28pub struct AudioPlan {
29    /// The selected versions, in output order.
30    pub versions: Vec<ResolvedVersion>,
31    /// Warnings recorded while resolving the selection.
32    pub warnings: Vec<DownloadWarning>,
33}
34
35/// Resolve an [`AudioSelection`] against a [`ResolvedMedia`].
36///
37/// Movies and music videos have a single version whose locale is unknown from
38/// metadata; their only version is always returned regardless of `selection`.
39///
40/// # Errors
41///
42/// Returns [`SelectionError`] when the media has no versions, or when an exact
43/// locale request cannot be met under [`FallbackPolicy::Exact`].
44pub fn select_audio(
45    media: &ResolvedMedia,
46    selection: &AudioSelection,
47    fallback: FallbackPolicy,
48) -> Result<AudioPlan, SelectionError> {
49    if media.versions.is_empty() {
50        return Err(SelectionError::NoAudioVersions);
51    }
52
53    if media.versions.iter().all(|v| v.audio_locale.is_none()) {
54        return Ok(AudioPlan {
55            versions: media.versions.clone(),
56            warnings: Vec::new(),
57        });
58    }
59
60    match selection {
61        AudioSelection::Original => Ok(AudioPlan {
62            versions: vec![pick_original(media)],
63            warnings: Vec::new(),
64        }),
65        AudioSelection::All => Ok(AudioPlan {
66            versions: sorted_all(media),
67            warnings: Vec::new(),
68        }),
69        AudioSelection::Locale(locale) => select_single_locale(media, locale, fallback),
70        AudioSelection::Locales(locales) => select_locale_list(media, locales, fallback),
71    }
72}
73
74fn version_locale_matches(version: &ResolvedVersion, locale: &Locale) -> bool {
75    version.audio_locale.as_ref() == Some(locale)
76}
77
78fn available_locales(media: &ResolvedMedia) -> Vec<Locale> {
79    media
80        .versions
81        .iter()
82        .filter_map(|v| v.audio_locale.clone())
83        .collect()
84}
85
86fn pick_original(media: &ResolvedMedia) -> ResolvedVersion {
87    if let Some(original) = media.versions.iter().find(|v| v.original) {
88        return original.clone();
89    }
90    if let Some(matching) = media
91        .versions
92        .iter()
93        .find(|v| v.audio_locale == media.audio_locale)
94    {
95        return matching.clone();
96    }
97    media.versions[0].clone()
98}
99
100fn sorted_all(media: &ResolvedMedia) -> Vec<ResolvedVersion> {
101    let mut versions = media.versions.clone();
102    versions.sort_by(|a, b| {
103        b.original
104            .cmp(&a.original)
105            .then_with(|| a.content_id.cmp(&b.content_id))
106    });
107    versions
108}
109
110fn select_single_locale(
111    media: &ResolvedMedia,
112    locale: &Locale,
113    fallback: FallbackPolicy,
114) -> Result<AudioPlan, SelectionError> {
115    if let Some(version) = media
116        .versions
117        .iter()
118        .find(|v| version_locale_matches(v, locale))
119    {
120        return Ok(AudioPlan {
121            versions: vec![version.clone()],
122            warnings: Vec::new(),
123        });
124    }
125
126    if fallback == FallbackPolicy::Exact {
127        return Err(SelectionError::AudioLocaleUnavailable {
128            requested: locale.clone(),
129            available: available_locales(media),
130        });
131    }
132
133    let used = pick_original(media);
134    let warning = DownloadWarning::AudioFallback {
135        requested: locale.clone(),
136        used: used.audio_locale.clone().unwrap_or_else(|| locale.clone()),
137    };
138    Ok(AudioPlan {
139        versions: vec![used],
140        warnings: vec![warning],
141    })
142}
143
144fn select_locale_list(
145    media: &ResolvedMedia,
146    locales: &[Locale],
147    fallback: FallbackPolicy,
148) -> Result<AudioPlan, SelectionError> {
149    let mut versions = Vec::new();
150    let mut warnings = Vec::new();
151    for locale in locales {
152        if let Some(version) = media
153            .versions
154            .iter()
155            .find(|v| version_locale_matches(v, locale))
156        {
157            if !versions
158                .iter()
159                .any(|v: &ResolvedVersion| v.content_id == version.content_id)
160            {
161                versions.push(version.clone());
162            }
163        } else if fallback == FallbackPolicy::Exact {
164            return Err(SelectionError::AudioLocaleUnavailable {
165                requested: locale.clone(),
166                available: available_locales(media),
167            });
168        } else {
169            warnings.push(DownloadWarning::AudioLocaleUnavailable {
170                requested: locale.clone(),
171            });
172        }
173    }
174    Ok(AudioPlan { versions, warnings })
175}
