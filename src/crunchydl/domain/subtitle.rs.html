21/// A supported subtitle source format.
22#[derive(Clone, Copy, Debug, PartialEq, Eq)]
23#[non_exhaustive]
24pub enum SubtitleFormat {
25    /// Advanced SubStation Alpha.
26    Ass,
27    /// Web Video Text Tracks.
28    WebVtt,
29}
30
31impl SubtitleFormat {
32    pub(crate) fn parse(value: &str) -> Result<Self, Error> {
33        match value.to_ascii_lowercase().as_str() {
34            "ass" | "ssa" => Ok(Self::Ass),
35            "vtt" | "webvtt" => Ok(Self::WebVtt),
36            _ => Err(Error::Subtitle("unsupported subtitle format".to_string())),
37        }
38    }
39}

58/// Explicit structural ASS normalization choices.
59#[derive(Clone, Debug, Default, PartialEq)]
60pub struct AssNormalization {

63    /// Add missing LayoutResX/LayoutResY with these values.
64    pub layout_resolution: Option<(u32, u32)>,
65    /// Add or replace WrapStyle.

67    /// Add or replace Timer.
68    pub timer: Option<f64>,
69    /// Add or replace ScaledBorderAndShadow.
70    pub scaled_border_and_shadow: Option<bool>,
71    /// Remove semicolon comments and the Aegisub Project Garbage section.

86/// A processed ASS subtitle ready for Matroska muxing.
87#[derive(Clone, Debug, PartialEq, Eq)]
88pub struct SubtitleTrack {
89    /// Exact output metadata selected by the caller.
90    pub metadata: SubtitleMetadata,
91    /// ASS document, including script info, styles, and events.
92    pub ass: String,
93    /// Canonical font family names actually referenced by styles or overrides.
94    pub referenced_fonts: Vec<String>,
95}

97/// Convert or normalize a subtitle resource without losing track metadata.
98///
99/// Raw ASS is byte-for-byte preserved when no normalization is requested.
100/// WebVTT is converted to ASS with cue order, basic inline styling, class
101/// styles, line breaks, and percentage positioning retained.
102///
103/// # Errors
104///
105/// Returns a subtitle error for malformed required sections, timestamps, or
106/// unsupported input.
107pub fn process_subtitle(
108    source: &str,
109    format: SubtitleFormat,
110    metadata: SubtitleMetadata,
111    options: &SubtitleProcessingOptions,
112) -> Result<SubtitleTrack, Error> {
113    let mut ass = match format {
114        SubtitleFormat::Ass => source.to_string(),
115        SubtitleFormat::WebVtt => vtt_to_ass(source, &metadata.title)?,
116    };
117    if let Some(normalization) = &options.normalization {
118        ass = normalize_ass(&ass, normalization, options.media_duration)?;
119    }
120    let referenced_fonts = extract_fonts(&ass)?;
121    Ok(SubtitleTrack {
122        metadata,
123        ass,
124        referenced_fonts,
125    })
126}

128#[allow(dead_code)]
129pub(crate) async fn download_selected<A: SubtitleFetcher>(
130    api: &A,
131    subtitles: &[PreparedSubtitle],
132    options: &SubtitleProcessingOptions,
133) -> Result<Vec<SubtitleTrack>, Error> {
134    let mut tracks = Vec::with_capacity(subtitles.len());
135    for subtitle in subtitles {
136        let source = api.fetch(&subtitle.url).await?;
137        let diagnostic = &subtitle.diagnostic;
138        tracks.push(process_subtitle(
139            &source,
140            SubtitleFormat::parse(&diagnostic.format)?,
141            SubtitleMetadata {
142                locale: diagnostic.locale.clone(),
143                title: diagnostic.title.clone(),
144                is_caption: diagnostic.is_caption,
145                is_signs: diagnostic.is_signs,
146                default: diagnostic.default,
147                forced: diagnostic.forced,
148            },
149            options,
150        )?);
151    }
152    Ok(tracks)
153}
