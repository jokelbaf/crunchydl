3use crunchyroll_rs::Locale;

5pub(super) fn iso639_2(locale: &Locale) -> &'static str {
6    match locale.to_string().split('-').next().unwrap_or("und") {
7        "ar" => "ara",
8        "ca" => "cat",
9        "de" => "deu",
10        "en" => "eng",
11        "es" => "spa",
12        "fr" => "fra",
13        "hi" => "hin",
14        "id" => "ind",
15        "it" => "ita",
16        "ja" => "jpn",
17        "ko" => "kor",
18        "ms" => "msa",
19        "pl" => "pol",
20        "pt" => "por",
21        "ru" => "rus",
22        "ta" => "tam",
23        "te" => "tel",
24        "th" => "tha",
25        "tr" => "tur",
26        "vi" => "vie",
27        "zh" => "zho",
28        _ => "und",
29    }
30}
