5pub(crate) struct PlaybackRefresher<'a> {
6    api: &'a ProductionApi,
7    platform: &'a StreamPlatform,
8    media: &'a ResolvedMedia,
9    options: &'a PlanningOptions,
10    cancellation: &'a CancellationToken,
11    sessions: tokio::sync::Mutex<Vec<SessionGuard<'a, ProductionApi>>>,
12}
13
14impl<'a> PlaybackRefresher<'a> {
15    pub(crate) fn new(
16        api: &'a ProductionApi,
17        platform: &'a StreamPlatform,
18        media: &'a ResolvedMedia,
19        options: &'a PlanningOptions,
20        cancellation: &'a CancellationToken,
21    ) -> Self {
22        Self {
23            api,
24            platform,
25            media,
26            options,
27            cancellation,
28            sessions: tokio::sync::Mutex::new(Vec::new()),
29        }
30    }
31
32    pub(crate) async fn finalize(&self) -> Result<(), Error> {
33        let sessions = {
34            let mut sessions = self.sessions.lock().await;
35            std::mem::take(&mut *sessions)
36        };
37        let mut first_error = None;
38        for guard in sessions {
39            if let Err(error) = guard.finalize().await
40                && first_error.is_none()
41            {
42                first_error = Some(error);
43            }
44        }
45        first_error.map_or(Ok(()), Err)
46    }
47}

49impl crate::RepresentationRefresher for PlaybackRefresher<'_> {
50    fn refresh<'a>(
51        &'a self,
52        expired: &'a crate::RepresentationTransferPlan,
53    ) -> Pin<Box<dyn Future<Output = Result<crate::RepresentationTransferPlan, Error>> + Send + 'a>>
54    {
55        Box::pin(async move {
56            let mut guard = SessionGuard::new(self.api);
57            let prepared = match crate::plan::prepare_with_guard(
58                self.api,
59                self.platform,
60                self.media,
61                self.options,
62                self.cancellation,
63                &mut guard,
64            )
65            .await
66            {
67                Ok(prepared) => prepared,
68                Err(error) => {
69                    let _ = guard.finalize().await;
70                    return Err(error);
71                }
72            };
73            let source = prepared
74                .sources
75                .iter()
76                .find(|source| {
77                    source.diagnostic.version_id == expired.version_id
78                        && source.diagnostic.representation_fingerprint
79                            == expired.representation_fingerprint
80                })
81                .ok_or_else(|| {
82                    Error::ResumeMismatch(
83                        "refreshed playback omitted the selected representation".into(),
84                    )
85                })?;
86            let refreshed =
87                source_transfer_plan(source, &expired.plan_fingerprint, &expired.media_id);
88            self.sessions.lock().await.push(guard);
89            Ok(refreshed)
90        })
91    }
92}

94pub(crate) fn select_pssh(
95    drm: &crunchyroll_rs::media::MediaStreamDRM,
96    system: DrmSystem,
97) -> Result<Vec<u8>, Error> {
98    let encoded = drm
99        .types
100        .iter()
101        .find_map(|kind| match (system, kind) {
102            (DrmSystem::PlayReady, MediaStreamDRMType::Playready { pro, pssh }) => pssh
103                .as_ref()
104                .and_then(|values| values.first())
105                .or(pro.as_ref()),
106            (DrmSystem::Widevine, MediaStreamDRMType::Widevine { pssh }) => pssh.first(),
107            _ => None,
108        })
109        .ok_or_else(|| Error::License(drm::Error::License))?;
110    base64::engine::general_purpose::STANDARD
111        .decode(encoded)
112        .map_err(|_| Error::License(drm::Error::License))
113}

115pub(crate) fn license_endpoint(
116    system: DrmSystem,
117    endpoint_override: Option<&str>,
118    playback_drm: &StreamDrm,
119) -> Result<String, Error> {
120    let endpoint = endpoint_override.unwrap_or(&playback_drm.drm_url).trim();
121    if endpoint.is_empty() {
122        return Err(Error::License(drm::Error::License));
123    }
124    if endpoint_override.is_some() || system.matches_name(playback_drm.name.trim()) {
125        return Ok(endpoint.to_string());
126    }
127    let (base, advertised_name) = endpoint
128        .rsplit_once('/')
129        .ok_or_else(|| Error::License(drm::Error::License))?;
130    if playback_drm.name.trim().is_empty()
131        || !advertised_name.eq_ignore_ascii_case(playback_drm.name.trim())
132    {
133        return Err(Error::License(drm::Error::License));
134    }
135    Ok(format!("{base}/{}", system.endpoint_name()))
136}
