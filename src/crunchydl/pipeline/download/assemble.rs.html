1//! Clear and CENC-decrypted MP4 track assembly.
2
3use super::*;
4
5pub(crate) fn assemble_track(
6    init_path: &Path,
7    segments: &[PathBuf],
8    output_path: &Path,
9    key: Option<&ContentKey>,
10    cancellation: &CancellationToken,
11) -> Result<(), Error> {
12    cancellation.check()?;
13    let init = std::fs::read(init_path).map_err(|error| path_error(init_path, error))?;
14    let mut output =
15        BufWriter::new(File::create(output_path).map_err(|error| path_error(output_path, error))?);
16    if let Some(key) = key {
17        let decrypter = crate::CencDecrypter::new(&init, key)
18            .map_err(|error| Error::Decrypt(error.to_string()))?;
19        decrypter
20            .assemble(&init, std::iter::empty(), &mut output)
21            .map_err(|error| Error::Decrypt(error.to_string()))?;
22        for segment in segments {
23            cancellation.check()?;
24            let encrypted = std::fs::read(segment).map_err(|error| path_error(segment, error))?;
25            output
26                .write_all(
27                    &decrypter
28                        .decrypt_fragment(encrypted)
29                        .map_err(|error| Error::Decrypt(error.to_string()))?,
30                )
31                .map_err(|error| path_error(output_path, error))?;
32        }
33    } else {
34        output
35            .write_all(&init)
36            .map_err(|error| path_error(output_path, error))?;
37        for segment in segments {
38            cancellation.check()?;
39            let mut input = File::open(segment).map_err(|error| path_error(segment, error))?;
40            std::io::copy(&mut input, &mut output)
41                .map_err(|error| path_error(output_path, error))?;
42        }
43    }
44    output
45        .flush()
46        .map_err(|error| path_error(output_path, error))?;
47    output
48        .get_ref()
49        .sync_all()
50        .map_err(|error| path_error(output_path, error))
51}
