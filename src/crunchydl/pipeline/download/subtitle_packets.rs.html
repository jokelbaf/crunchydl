5pub(crate) fn ass_packets(
6    ass: &str,
7    track_number: u64,
8) -> Result<(String, Vec<mkv::Packet>), Error> {
9    let normalized = ass.replace("\r\n", "\n").replace('\r', "\n");
10    let mut header = Vec::new();
11    let mut packets = Vec::new();
12    let mut read_order = 0_u64;
13    for line in normalized.lines() {
14        let Some(dialogue) = line.strip_prefix("Dialogue:") else {
15            header.push(line);
16            continue;
17        };
18        let fields = dialogue.trim_start().splitn(10, ',').collect::<Vec<_>>();
19        if fields.len() != 10 {
20            return Err(Error::Subtitle("malformed ASS dialogue".into()));
21        }
22        let start = ass_time(fields[1])?;
23        let end = ass_time(fields[2])?;
24        if end < start {
25            return Err(Error::Subtitle("ASS dialogue ends before it starts".into()));
26        }
27        let payload = format!(
28            "{read_order},{},{},{},{},{},{},{},{}",
29            fields[0], fields[3], fields[4], fields[5], fields[6], fields[7], fields[8], fields[9]
30        );
31        packets.push(mkv::Packet {
32            track_number,
33            decode_time_ms: i64::try_from(start.as_millis())
34                .map_err(|_| Error::Subtitle("subtitle timestamp overflow".into()))?,
35            presentation_time_ms: i64::try_from(start.as_millis())
36                .map_err(|_| Error::Subtitle("subtitle timestamp overflow".into()))?,
37            duration: end - start,
38            keyframe: true,
39            data: payload.into_bytes(),
40        });
41        read_order += 1;
42    }
43    packets.sort_by_key(|packet| packet.decode_time_ms);
44    Ok((header.join("\r\n") + "\r\n", packets))
45}

47fn ass_time(value: &str) -> Result<Duration, Error> {
48    let (hours, rest) = value
49        .trim()
50        .split_once(':')
51        .ok_or_else(|| Error::Subtitle("invalid ASS timestamp".into()))?;
52    let (minutes, seconds) = rest
53        .split_once(':')
54        .ok_or_else(|| Error::Subtitle("invalid ASS timestamp".into()))?;
55    let hours: u64 = hours
56        .parse()
57        .map_err(|_| Error::Subtitle("invalid ASS timestamp".into()))?;
58    let minutes: u64 = minutes
59        .parse()
60        .map_err(|_| Error::Subtitle("invalid ASS timestamp".into()))?;
61    let seconds: f64 = seconds
62        .parse()
63        .map_err(|_| Error::Subtitle("invalid ASS timestamp".into()))?;
64    if minutes >= 60 || !seconds.is_finite() || !(0.0..60.0).contains(&seconds) {
65        return Err(Error::Subtitle("invalid ASS timestamp".into()));
66    }
67    Ok(Duration::from_millis(
68        hours
69            .saturating_mul(3_600_000)
70            .saturating_add(minutes * 60_000)
71            .saturating_add((seconds * 1000.0).round() as u64),
72    ))
73}
