5pub(crate) fn mux_inputs(
6    paths: &[(PathBuf, crate::PlannedTrack)],
7    subtitles: &[crate::SubtitleTrack],
8    synchronization: &SynchronizationOptions,
9    cancellation: &CancellationToken,
10) -> Result<(Vec<mkv::Track>, PacketStreams), Error> {
11    let mut tracks = Vec::new();
12    let mut streams = Vec::new();
13    let mut audio_default = true;
14    let mut video_default = true;
15    let mut reference_audio_duration: Option<Duration> = None;
16    for (path, diagnostic) in paths {
17        let parsed = FragmentedMp4::open(BufReader::new(
18            File::open(path).map_err(|error| path_error(path, error))?,
19        ))?;
20        let media_track = parsed.tracks()[0].clone();
21        let number = tracks.len() as u64 + 1;
22        let offset = synchronization
23            .offsets
24            .iter()
25            .find_map(|(locale, offset)| (locale == &diagnostic.locale).then_some(*offset))
26            .unwrap_or(0);
27        let (track_type, codec, settings, default) = match (&media_track.kind, &media_track.codec) {
28            (TrackKind::Video, Codec::Avc { configuration }) => (
29                mkv::TrackType::Video,
30                mkv::TrackCodec::Avc(configuration.clone()),
31                mkv::TrackSettings::Video(mkv::VideoSettings {
32                    width: media_track
33                        .dimensions
34                        .ok_or_else(|| Error::Mux("video dimensions missing".into()))?
35                        .0,
36                    height: media_track
37                        .dimensions
38                        .ok_or_else(|| Error::Mux("video dimensions missing".into()))?
39                        .1,
40                }),
41                {
42                    let default = video_default;
43                    video_default = false;
44                    default
45                },
46            ),
47            (
48                TrackKind::Audio,
49                Codec::Aac {
50                    audio_specific_config,
51                },
52            ) => {
53                let actual_duration = FragmentedMp4::open(BufReader::new(
54                    File::open(path).map_err(|error| path_error(path, error))?,
55                ))?
56                .probe()?
57                .duration;
58                if let Some(reference) = reference_audio_duration {
59                    let difference = reference.abs_diff(actual_duration);
60                    if synchronization.strict && difference > synchronization.tolerance {
61                        return Err(Error::IncompatibleTimelines {
62                            locale: diagnostic.locale.to_string(),
63                        });
64                    }
65                } else {
66                    reference_audio_duration = Some(actual_duration);
67                }
68                let value = (
69                    mkv::TrackType::Audio,
70                    mkv::TrackCodec::Aac(audio_specific_config.clone()),
71                    mkv::TrackSettings::Audio(mkv::AudioSettings {
72                        sampling_frequency: f64::from(
73                            media_track
74                                .sample_rate
75                                .ok_or_else(|| Error::Mux("audio sampling rate missing".into()))?,
76                        ),
77                        channels: media_track
78                            .channels
79                            .ok_or_else(|| Error::Mux("audio channels missing".into()))?,
80                    }),
81                    audio_default,
82                );
83                audio_default = false;
84                value
85            }
86            _ => return Err(Error::Mux("unsupported media codec".into())),
87        };
88        tracks.push(mkv::Track {
89            number,
90            uid: 0,
91            track_type,
92            codec,
93            settings,
94            name: (track_type == mkv::TrackType::Audio)
95                .then(|| crate::locale_display_name(&diagnostic.locale)),
96            language: (track_type == mkv::TrackType::Audio).then(|| language(&diagnostic.locale)),
97            default,
98            forced: false,
99            hearing_impaired: false,
100            visual_impaired: false,
101            original: default,
102            commentary: false,
103        });
104        let packets = parsed.packets().map(move |packet| {
105            let packet = packet.map_err(|error| mkv::Error::PacketSource(error.to_string()))?;
106            let decode_time_ms = timestamp_millis(packet.dts)
107                .map_err(|error| mkv::Error::PacketSource(error.to_string()))?;
108            let presentation_time_ms = timestamp_millis(packet.pts)
109                .map_err(|error| mkv::Error::PacketSource(error.to_string()))?;
110            Ok(mkv::Packet {
111                track_number: number,
112                decode_time_ms: decode_time_ms
113                    .checked_add(offset)
114                    .ok_or(mkv::Error::Overflow("explicit track decode offset"))?,
115                presentation_time_ms: presentation_time_ms
116                    .checked_add(offset)
117                    .ok_or(mkv::Error::Overflow("explicit track presentation offset"))?,
118                duration: packet
119                    .pts
120                    .time_base()
121                    .duration(u64::from(packet.duration))
122                    .map_err(|error| mkv::Error::PacketSource(error.to_string()))?,
123                keyframe: packet.is_keyframe,
124                data: packet.data,
125            })
126        });
127        streams.push(PacketStream::new(Box::new(packets)));
128    }
129    for subtitle in subtitles {
130        let number = tracks.len() as u64 + 1;
131        let (header, packets) = ass_packets(&subtitle.ass, number)?;
132        tracks.push(mkv::Track {
133            number,
134            uid: 0,
135            track_type: mkv::TrackType::Subtitle,
136            codec: mkv::TrackCodec::Ass(header),
137            settings: mkv::TrackSettings::Subtitle,
138            name: Some(subtitle.metadata.title.clone()),
139            language: Some(language(&subtitle.metadata.locale)),
140            default: subtitle.metadata.default,
141            forced: subtitle.metadata.forced,
142            hearing_impaired: subtitle.metadata.is_caption,
143            visual_impaired: false,
144            original: false,
145            commentary: false,
146        });
147        streams.push(PacketStream::new(Box::new(packets.into_iter().map(Ok))));
148    }
149    Ok((
150        tracks,
151        PacketStreams {
152            streams,
153            cancellation: cancellation.clone(),
154            cancellation_reported: false,
155        },
156    ))
157}

159type FalliblePackets = Box<dyn Iterator<Item = Result<mkv::Packet, mkv::Error>>>;

161struct PacketStream {
162    packets: FalliblePackets,
163    head: Option<Result<mkv::Packet, mkv::Error>>,
164}
165
166impl PacketStream {
167    fn new(mut packets: FalliblePackets) -> Self {
168        let head = packets.next();
169        Self { packets, head }
170    }
171}

173pub(crate) struct PacketStreams {
174    streams: Vec<PacketStream>,
175    cancellation: CancellationToken,
176    cancellation_reported: bool,
177}
178
179impl Iterator for PacketStreams {
180    type Item = Result<mkv::Packet, mkv::Error>;
181
182    fn next(&mut self) -> Option<Self::Item> {
183        if self.cancellation.is_cancelled() && !self.cancellation_reported {
184            self.cancellation_reported = true;
185            return Some(Err(mkv::Error::Cancelled));
186        }
187        let index = self
188            .streams
189            .iter()
190            .position(|stream| stream.head.as_ref().is_some_and(Result::is_err))
191            .or_else(|| {
192                self.streams
193                    .iter()
194                    .enumerate()
195                    .filter_map(|(index, stream)| {
196                        stream
197                            .head
198                            .as_ref()
199                            .and_then(|packet| packet.as_ref().ok())
200                            .map(|packet| (index, packet.decode_time_ms, packet.track_number))
201                    })
202                    .min_by_key(|(_, decode_time_ms, track_number)| {
203                        (*decode_time_ms, *track_number)
204                    })
205                    .map(|(index, _, _)| index)
206            })?;
207        let result = self.streams[index].head.take();
208        self.streams[index].head = self.streams[index].packets.next();
209        result
210    }
211}

213fn timestamp_millis(timestamp: media::Timestamp) -> Result<i64, media::Error> {
214    let numerator = i128::from(timestamp.ticks())
215        .checked_mul(1_000)
216        .ok_or(media::Error::Overflow("timestamp milliseconds"))?;
217    i64::try_from(numerator / i128::from(timestamp.time_base().ticks_per_second()))
218        .map_err(|_| media::Error::Overflow("timestamp milliseconds"))
219}

221pub(crate) fn language(locale: &Locale) -> mkv::Language {
222    let ietf = locale.to_string();
223    let primary = ietf.split('-').next().unwrap_or("und");
224    let legacy = match primary {
225        "ar" => "ara",
226        "ca" => "cat",
227        "de" => "ger",
228        "en" => "eng",
229        "es" => "spa",
230        "fr" => "fre",
231        "hi" => "hin",
232        "id" => "ind",
233        "it" => "ita",
234        "ja" => "jpn",
235        "ko" => "kor",
236        "ms" => "may",
237        "pl" => "pol",
238        "pt" => "por",
239        "ru" => "rus",
240        "ta" => "tam",
241        "te" => "tel",
242        "th" => "tha",
243        "tr" => "tur",
244        "vi" => "vie",
245        "zh" => "chi",
246        _ => "und",
247    };
248    mkv::Language {
249        legacy: legacy.into(),
250        ietf,
251    }
