12pub(super) fn verify(
13    path: &Path,
14    track_count: usize,
15    attachment_count: usize,
16    chapter_count: usize,
17) -> Result<Vec<OutputTrack>, Error> {
18    let parsed =
19        matroska_reader::open(path).map_err(|error| Error::Verification(error.to_string()))?;
20    if parsed.tracks.len() != track_count
21        || parsed.attachments.len() != attachment_count
22        || parsed
23            .chapters
24            .iter()
25            .map(|edition| edition.chapters.len())
26            .sum::<usize>()
27            != chapter_count
28        || parsed
29            .info
30            .duration
31            .is_none_or(|duration| duration.is_zero())
32    {
33        return Err(Error::Verification(
34            "Matroska structural verification failed".into(),
35        ));
36    }
37    let markers = scan_markers(path)?;
38    if !markers.cluster || !markers.block {
39        return Err(Error::Verification(
40            "Matroska contains no media blocks".into(),
41        ));
42    }
43    Ok(parsed
44        .tracks
45        .into_iter()
46        .map(|track| OutputTrack {
47            codec: track.codec_id,
48            language: track.language.map(|language| match language {
49                matroska_reader::Language::ISO639(value)
50                | matroska_reader::Language::IETF(value) => value,
51            }),
52            name: track.name,
53            default: track.default,
54            forced: track.forced,
55        })
56        .collect())
57}

59pub(super) struct StructuralMarkers {
60    pub(super) cluster: bool,
61    pub(super) block: bool,
62}
63
64pub(super) fn scan_markers(path: &Path) -> Result<StructuralMarkers, Error> {
65    let mut reader = BufReader::new(File::open(path).map_err(|error| path_error(path, error))?);
66    let mut carry = Vec::new();
67    let mut markers = StructuralMarkers {
68        cluster: false,
69        block: false,
70    };
71    loop {
72        let mut chunk = vec![0; 64 * 1024];
73        let read = reader
74            .read(&mut chunk)
75            .map_err(|error| path_error(path, error))?;
76        if read == 0 {
77            break;
78        }
79        chunk.truncate(read);
80        carry.extend(chunk);
81        markers.cluster |= carry
82            .windows(4)
83            .any(|window| window == [0x1f, 0x43, 0xb6, 0x75]);
84        markers.block |= carry.contains(&0xa3);
85        if carry.len() > 7 {
86            carry.drain(..carry.len() - 7);
87        }
88    }
89    Ok(markers)
90}

92pub(super) fn commit(
93    temporary: &Path,
94    destination: &Path,
95    overwrite: OverwritePolicy,
96) -> Result<(), Error> {
97    if destination.exists() && overwrite != OverwritePolicy::Replace {
98        return Err(Error::Filesystem(format!(
99            "output already exists: {}",
100            destination.display()
101        )));
102    }
103    match std::fs::rename(temporary, destination) {
104        Ok(()) => {}
105        Err(error)
106            if overwrite == OverwritePolicy::Replace
107                && error.kind() == std::io::ErrorKind::AlreadyExists =>
108        {
109            std::fs::remove_file(destination).map_err(|error| path_error(destination, error))?;
110            std::fs::rename(temporary, destination)
111                .map_err(|error| path_error(destination, error))?;
112        }
113        Err(error) => return Err(path_error(destination, error)),
114    }
115    if let Some(parent) = destination.parent()
116        && let Ok(directory) = File::open(parent)
117    {
118        let _ = directory.sync_all();
119    }
120    Ok(())
121}

123pub(super) fn commit_output(
124    temporary: &Path,
125    destination: &Path,
126    overwrite: OverwritePolicy,
127    archive: Option<&dyn Archive>,
128    archive_key: &ArchiveKey,
129    tracks: &[OutputTrack],
130) -> Result<(), Error> {
131    commit(temporary, destination, overwrite)?;
132    if let Some(archive) = archive {
133        archive.record(&ArchiveEntry {
134            key: archive_key.clone(),
135            output: destination.to_path_buf(),
136            tracks: tracks.to_vec(),
137        })?;
138    }
139    Ok(())
140}

142pub(super) fn cleanup_staging(path: &Path, retention: RetentionPolicy, succeeded: bool) {
143    let retain = matches!(
144        (retention, succeeded),
145        (RetentionPolicy::KeepAlways, _) | (RetentionPolicy::KeepOnFailure, false)
146    );
147    if !retain {
148        let _ = std::fs::remove_dir_all(path);
149    }
150}

152pub(super) fn validate_media(media: &ResolvedMedia) -> Result<(), Error> {
153    if media.content_id.is_empty()
154        || media
155            .availability_status
156            .eq_ignore_ascii_case("unavailable")
157    {
158        return Err(Error::Unavailable(media.content_id.clone()));
159    }
160    Ok(())
161}

163pub(super) fn safe_component(value: &str) -> String {
164    let value = value
165        .chars()
166        .map(|character| {
167            if character.is_ascii_alphanumeric() || matches!(character, '-' | '_') {
168                character
169            } else {
170                '_'
171            }
172        })
173        .collect::<String>();
174    if value.is_empty() {
175        "media".into()
176    } else {
177        value
178    }
179}
