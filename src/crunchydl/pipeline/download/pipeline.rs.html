5#[allow(clippy::too_many_arguments)]
6pub(crate) async fn execute_pipeline(
7    api: &ProductionApi,
8    crunchyroll: &crunchyroll_rs::Crunchyroll,
9    platform: &StreamPlatform,
10    runtime: &RuntimeConfiguration,
11    request: &DownloadRequest,
12    media: &ResolvedMedia,
13    archive_key: &ArchiveKey,
14    prepared: &PreparedPlan,
15    staging: &Path,
16    destination: &Path,
17    states: &StateEmitter,
18) -> Result<DownloadResult, Error> {
19    let refresher = PlaybackRefresher::new(
20        api,
21        platform,
22        media,
23        &request.planning,
24        &request.cancellation,
25    );
26    let result = execute_pipeline_inner(
27        api,
28        crunchyroll,
29        runtime,
30        request,
31        media,
32        archive_key,
33        prepared,
34        staging,
35        destination,
36        states,
37        &refresher,
38    )
39    .await;
40    let finalization = refresher.finalize().await;
41    match (result, finalization) {
42        (Ok(value), Ok(())) => Ok(value),
43        (Ok(_), Err(error)) | (Err(error), _) => Err(error),
44    }
45}

47#[allow(clippy::too_many_arguments)]
48pub(crate) async fn execute_pipeline_inner(
49    api: &ProductionApi,
50    crunchyroll: &crunchyroll_rs::Crunchyroll,
51    runtime: &RuntimeConfiguration,
52    request: &DownloadRequest,
53    media: &ResolvedMedia,
54    archive_key: &ArchiveKey,
55    prepared: &PreparedPlan,
56    staging: &Path,
57    destination: &Path,
58    states: &StateEmitter,
59    refresher: &PlaybackRefresher<'_>,
60) -> Result<DownloadResult, Error> {
61    let transfer = TransferEngine::with_client(crunchyroll.client(), request.transfer.clone())?
62        .event_sink(runtime.events.clone());
63    let transport = CrunchyrollLicenseTransport::new(crunchyroll.clone());
64    let mut warnings = prepared.public.warnings.clone();
65    states.enter(JobState::AcquiringLicenses)?;
66    let mut licensed = Vec::with_capacity(prepared.sources.len());
67    for (index, source) in prepared.sources.iter().enumerate() {
68        let transfer_plan =
69            source_transfer_plan(source, &prepared.public.fingerprint, &media.content_id);
70        let init_path = transfer
71            .transfer_init(&transfer_plan, staging, &request.cancellation)
72            .await?;
73        let init = std::fs::read(&init_path).map_err(|error| path_error(&init_path, error))?;
74        let key = if let Some(drm) = &source.drm {
75            let config = runtime
76                .drm
77                .as_ref()
78                .ok_or_else(|| Error::License(drm::Error::License))?;
79            let info = inspect_encryption(&init)?;
80            let pssh = select_pssh(drm, config.system)?;
81            let endpoint = license_endpoint(
82                config.system,
83                config.endpoint_override.as_deref(),
84                &source.playback_drm,
85            )?;
86            let request_data = DrmRequest {
87                endpoint,
88                content_id: source.diagnostic.version_id.clone(),
89                playback_token: drm.token.clone(),
90                content_type: match source.diagnostic.kind {
91                    PlannedTrackKind::Video => ContentType::Video,
92                    PlannedTrackKind::Audio => ContentType::Audio,
93                },
94                pssh,
95            };
96            let keys = config
97                .provider
98                .acquire_keys(request_data, &transport)
99                .await?;
100            Some(keys.require(info.default_kid)?.clone())
101        } else {
102            None
103        };
104        licensed.push((source, transfer_plan, key));
105        runtime.events.emit(DownloadEvent::StageProgress {
106            state: JobState::AcquiringLicenses,
107            completed: index + 1,
108            total: prepared.sources.len(),
109        });
110    }
111
112    states.enter(JobState::Downloading)?;
113    let mut transferred = Vec::with_capacity(licensed.len());
114    let licensed_count = licensed.len();
115    for (index, (source, transfer_plan, key)) in licensed.into_iter().enumerate() {
116        let result = transfer
117            .transfer_with_refresh(transfer_plan, staging, &request.cancellation, refresher)
118            .await?;
119        transferred.push((source, result, key));
120        runtime.events.emit(DownloadEvent::StageProgress {
121            state: JobState::Downloading,
122            completed: index + 1,
123            total: licensed_count,
124        });
125    }
126
127    states.enter(JobState::Decrypting)?;
128    let mut assembled = Vec::with_capacity(transferred.len());
129    let transferred_count = transferred.len();
130    for (index, (source, transferred, key)) in transferred.into_iter().enumerate() {
131        let output = staging.join(format!("track-{index:03}.mp4"));
132        assemble_track(
133            &transferred.init,
134            &transferred.segments,
135            &output,
136            key.as_ref(),
137            &request.cancellation,
138        )?;
139        let file = File::open(&output).map_err(|error| path_error(&output, error))?;
140        let parsed = FragmentedMp4::open(BufReader::new(file))?;
141        if parsed.tracks().len() != 1 {
142            return Err(Error::MediaParse(media::Error::Unsupported(
143                "assembled representation must contain one track",
144            )));
145        }
146        assembled.push((output, source.diagnostic.clone()));
147        runtime.events.emit(DownloadEvent::StageProgress {
148            state: JobState::Decrypting,
149            completed: index + 1,
150            total: transferred_count,
151        });
152    }
153
154    states.enter(JobState::ProcessingSubtitles)?;
155    let subtitles = download_selected(api, &prepared.subtitles, &request.subtitles).await?;
156    runtime.events.emit(DownloadEvent::StageProgress {
157        state: JobState::ProcessingSubtitles,
158        completed: subtitles.len(),
159        total: prepared.subtitles.len(),
160    });
161    let missing = MissingFontResolver;
162    let resolver: &dyn FontResolver = runtime.font_resolver.as_deref().unwrap_or(&missing);
163    let (fonts, font_warnings) =
164        crate::resolve_referenced_fonts(&subtitles, resolver, runtime.font_policy)?;
165    for warning in &font_warnings {
166        runtime.events.emit(DownloadEvent::Warning(warning.clone()));
167    }
168    warnings.extend(font_warnings);
169
170    states.enter(JobState::Muxing)?;
171    let temporary = staging.join(match request.output.container {
172        crate::Container::Matroska => "output.mkv.part",
173        crate::Container::Mp4 => "output.mp4.part",
174    });
175    let output_tracks = match request.output.container {
176        crate::Container::Matroska => mux_matroska(
177            &temporary,
178            &assembled,
179            &subtitles,
180            &fonts,
181            &prepared.public.chapters,
182            media,
183            &request.synchronization,
184            &request.cancellation,
185        )?,
186        crate::Container::Mp4 => crate::mp4_output::write_and_verify(
187            &temporary,
188            &assembled,
189            &request.synchronization,
190            &request.cancellation,
191        )?,
192    };
193    runtime.events.emit(DownloadEvent::StageProgress {
194        state: JobState::Muxing,
195        completed: 1,
196        total: 1,
197    });
198
199    states.enter(JobState::Verifying)?;
200    runtime.events.emit(DownloadEvent::StageProgress {
201        state: JobState::Verifying,
202        completed: 1,
203        total: 1,
204    });
205    states.enter(JobState::Committing)?;
206    commit_output(
207        &temporary,
208        destination,
209        request.output.overwrite,
210        runtime.archive.as_deref(),
211        archive_key,
212        &output_tracks,
213    )?;
214    runtime.events.emit(DownloadEvent::StageProgress {
215        state: JobState::Committing,
216        completed: 1,
217        total: 1,
218    });
219    runtime.events.emit(DownloadEvent::OutputCommitted {
220        output: destination.to_path_buf(),
221    });
222    Ok(DownloadResult {
223        output: destination.to_path_buf(),
224        media_id: media.content_id.clone(),
225        tracks: output_tracks,
226        warnings,
227    })
228}
