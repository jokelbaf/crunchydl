23    /// Construct an engine with a dedicated rustls HTTP client and no events.
24    ///
25    /// # Errors
26    ///
27    /// Returns [`Error::Transfer`] if the HTTP client cannot be constructed or
28    /// if an option invariant is invalid.
29    pub fn new(options: TransferOptions) -> Result<Self, Error> {
30        Self::with_client(reqwest::Client::new(), options)

33    /// Construct an engine around a caller-configured HTTP client.
34    ///
35    /// # Errors
36    ///
37    /// Returns [`Error::Transfer`] if concurrency or attempts is zero.
38    pub fn with_client(client: reqwest::Client, options: TransferOptions) -> Result<Self, Error> {
39        if options.concurrency == 0 || options.max_attempts == 0 {
40            return Err(Error::Transfer(
41                "transfer concurrency and max attempts must be nonzero".to_string(),
42            ));
43        }
44        Ok(Self {
45            client,
46            options,
47            events: Arc::new(NoopSink),
48        })
49    }

51    /// Deliver structured progress and retry events to `sink`.
52    #[must_use]
53    pub fn event_sink(mut self, sink: impl EventSink + 'static) -> Self {
54        self.events = Arc::new(sink);
55        self
56    }

58    /// Download or resume a representation beneath `staging_root`.
59    ///
60    /// # Errors
61    ///
62    /// Returns typed cancellation, transfer, filesystem, or resume mismatch
63    /// errors. Existing staging is reused only after complete validation.
64    pub async fn transfer(
65        &self,
66        plan: RepresentationTransferPlan,
67        staging_root: &Path,
68        cancellation: &CancellationToken,
69    ) -> Result<TransferResult, Error> {
70        self.transfer_with_refresh(plan, staging_root, cancellation, &NoRefresh)
71            .await
72    }

74    pub(crate) async fn transfer_init(
75        &self,
76        plan: &RepresentationTransferPlan,
77        staging_root: &Path,
78        cancellation: &CancellationToken,
79    ) -> Result<PathBuf, Error> {
80        let layout = StagingLayout::new(
81            staging_root,
82            &plan.media_id,
83            &plan.representation_fingerprint,
84        );
85        layout.create().await?;
86        let mut journal = load_or_create_journal(&layout, plan).await?;
87        validate_completed_files(&layout, &mut journal).await?;
88        self.transfer_initialization(plan, &layout, &mut journal, cancellation)
89            .await
90            .map_err(|error| match error {
91                AttemptError::Expired => {
92                    Error::Transfer("initialization URL expired before licensing".to_string())
93                }
94                AttemptError::Fatal(error) => error,
95            })?;
96        Ok(layout.init)
97    }

99    /// Download or resume with stable representation refresh after URL expiry.
100    ///
101    /// # Errors
102    ///
103    /// In addition to [`TransferEngine::transfer`] errors, returns
104    /// [`Error::ResumeMismatch`] when refreshed playback no longer contains the
105    /// exact stable representation.
106    pub async fn transfer_with_refresh<R: RepresentationRefresher>(
107        &self,
108        mut plan: RepresentationTransferPlan,
109        staging_root: &Path,
110        cancellation: &CancellationToken,
111        refresher: &R,
112    ) -> Result<TransferResult, Error> {
113        let layout = StagingLayout::new(
114            staging_root,
115            &plan.media_id,
116            &plan.representation_fingerprint,
117        );
118        layout.create().await?;
119        let mut journal = load_or_create_journal(&layout, &plan).await?;
120        validate_completed_files(&layout, &mut journal).await?;
121
122        loop {
123            cancellation.check()?;
124            match self
125                .transfer_current(&plan, &layout, &mut journal, cancellation)
126                .await
127            {
128                Ok(()) => break,
129                Err(AttemptError::Expired) => {
130                    let refreshed = refresher.refresh(&plan).await?;
131                    validate_refresh(&plan, &refreshed)?;
132                    plan = refreshed;
133                }
134                Err(AttemptError::Fatal(error)) => return Err(error),
135            }
136        }
137
138        Ok(TransferResult {
139            init: layout.init.clone(),
140            segments: (0..plan.segments.len())
141                .map(|index| layout.segment(index))
142                .collect(),
143            journal: layout.journal.clone(),
144        })
145    }

147    async fn transfer_current(
148        &self,
149        plan: &RepresentationTransferPlan,
150        layout: &StagingLayout,
151        journal: &mut ResumeJournal,
152        cancellation: &CancellationToken,
153    ) -> Result<(), AttemptError> {
154        self.transfer_initialization(plan, layout, journal, cancellation)
155            .await?;
156
157        let completed = journal
158            .completed
159            .iter()
160            .map(|segment| segment.index)
161            .collect::<BTreeSet<_>>();
162        let pending = (0..plan.segments.len())
163            .filter(|index| !completed.contains(index))
164            .collect::<Vec<_>>();
165        for batch in pending.chunks(self.options.concurrency) {
166            cancellation.check().map_err(AttemptError::Fatal)?;
167            let mut tasks = JoinSet::new();
168            for &index in batch {
169                let engine = self.clone();
170                let event_context = TransferEventContext::from(plan);
171                let request = plan.segments[index].clone();
172                let path = layout.segment(index);
173                let cancellation = cancellation.clone();
174                tasks.spawn(async move {
175                    let bytes = engine
176                        .fetch(&event_context, &request, Some(index), &cancellation)
177                        .await?;
178                    cancellation.check().map_err(FetchError::Fatal)?;
179                    atomic_write(&path, &bytes)
180                        .await
181                        .map_err(FetchError::Fatal)?;
182                    Ok::<_, FetchError>((index, bytes.len() as u64))
183                });
184            }
185
186            let mut successes = BTreeMap::new();
187            let mut first_error = None;
188            while let Some(result) = tasks.join_next().await {
189                match result {
190                    Ok(Ok((index, bytes))) => {
191                        successes.insert(index, bytes);
192                    }
193                    Ok(Err(error)) if first_error.is_none() => first_error = Some(error),
194                    Err(error) if first_error.is_none() => {
195                        first_error = Some(FetchError::Fatal(Error::Transfer(format!(
196                            "segment task failed: {error}"
197                        ))));
198                    }
199                    _ => {}
200                }
201            }
202            for (index, bytes) in successes {
203                journal.completed.push(CompletedSegment { index, bytes });
204                journal.completed.sort_by_key(|segment| segment.index);
205                journal
206                    .save(&layout.journal)
207                    .await
208                    .map_err(AttemptError::Fatal)?;
209                self.events.emit(DownloadEvent::SegmentCompleted {
210                    media_id: plan.media_id.clone(),
211                    version_id: plan.version_id.clone(),
212                    representation_fingerprint: plan.representation_fingerprint.clone(),
213                    track: plan.track.clone(),
214                    index,
215                    completed: journal.completed.len(),
216                    total: plan.segments.len(),
217                    bytes,
218                    completed_bytes: journal.init_bytes.unwrap_or_default().saturating_add(
219                        journal.completed.iter().map(|segment| segment.bytes).sum(),
220                    ),
221                    total_bytes: representation_bytes(plan),
222                });
223            }
224            if let Some(error) = first_error {
225                return Err(error.into_attempt_error());
226            }
227        }
228        Ok(())
229    }

231    async fn transfer_initialization(
232        &self,
233        plan: &RepresentationTransferPlan,
234        layout: &StagingLayout,
235        journal: &mut ResumeJournal,
236        cancellation: &CancellationToken,
237    ) -> Result<(), AttemptError> {
238        if journal.init_bytes.is_none() {
239            let event_context = TransferEventContext::from(plan);
240            let bytes = self
241                .fetch(&event_context, &plan.init, None, cancellation)
242                .await
243                .map_err(|error| error.into_attempt_error())?;
244            cancellation.check().map_err(AttemptError::Fatal)?;
245            atomic_write(&layout.init, &bytes)
246                .await
247                .map_err(AttemptError::Fatal)?;
248            journal.init_bytes = Some(bytes.len() as u64);
249            journal
250                .save(&layout.journal)
251                .await
252                .map_err(AttemptError::Fatal)?;
253        }
254        Ok(())
255    }
