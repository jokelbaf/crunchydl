3use std::path::{Path, PathBuf};
4use std::sync::Mutex;
5
6use serde::{Deserialize, Serialize};

8use crate::{Error, OutputTrack};

10const SCHEMA_VERSION: u32 = 1;
11
12/// Stable archive identity for one completed selection.
13#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
14pub struct ArchiveKey {
15    /// Crunchyroll media id.
16    pub media_id: String,
17    /// Deterministic requested-selection fingerprint.
18    pub selection_fingerprint: String,
19}
20
21/// A committed download stored in an archive.
22#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
23pub struct ArchiveEntry {
24    /// Stable selection identity.
25    pub key: ArchiveKey,
26    /// Committed output path.
27    pub output: PathBuf,
28    /// Output track diagnostics.
29    pub tracks: Vec<OutputTrack>,
30}
31
32/// Persistence seam for opt-in download archives.
33pub trait Archive: Send + Sync {
34    /// Find an entry. A stale entry whose output no longer exists must return
35    /// `None` so it cannot suppress a new download.
36    ///
37    /// # Errors
38    ///
39    /// Returns a filesystem or schema error.
40    fn find(&self, key: &ArchiveKey) -> Result<Option<ArchiveEntry>, Error>;
41
42    /// Persist a committed entry.
43    ///
44    /// # Errors
45    ///
46    /// Returns a filesystem or serialization error.
47    fn record(&self, entry: &ArchiveEntry) -> Result<(), Error>;
48}
49
50/// A versioned JSON archive using atomic replacement.
51pub struct JsonArchive {
52    path: PathBuf,
53    access: Mutex<()>,
54}
55
56impl JsonArchive {
57    /// Create an archive stored at `path`.
58    #[must_use]
59    pub fn new(path: impl Into<PathBuf>) -> Self {
60        Self {
61            path: path.into(),
62            access: Mutex::new(()),
63        }
64    }
65}
66
67#[derive(Default, Serialize, Deserialize)]
68struct Document {
69    schema_version: u32,
70    entries: Vec<ArchiveEntry>,
71}
72
73impl Archive for JsonArchive {
74    fn find(&self, key: &ArchiveKey) -> Result<Option<ArchiveEntry>, Error> {
75        let _guard = self
76            .access
77            .lock()
78            .map_err(|_| Error::Filesystem("archive lock poisoned".into()))?;
79        let document = load(&self.path)?;
80        Ok(document
81            .entries
82            .into_iter()
83            .find(|entry| &entry.key == key && entry.output.is_file()))
84    }
85
86    fn record(&self, entry: &ArchiveEntry) -> Result<(), Error> {
87        let _guard = self
88            .access
89            .lock()
90            .map_err(|_| Error::Filesystem("archive lock poisoned".into()))?;
91        let mut document = load(&self.path)?;
92        document
93            .entries
94            .retain(|existing| existing.key != entry.key);
95        document.entries.push(entry.clone());
96        document.entries.sort_by(|left, right| {
97            left.key.media_id.cmp(&right.key.media_id).then_with(|| {
98                left.key
99                    .selection_fingerprint
100                    .cmp(&right.key.selection_fingerprint)
101            })
102        });
103        let bytes = serde_json::to_vec_pretty(&document)
104            .map_err(|error| Error::Filesystem(format!("serialize archive: {error}")))?;
105        if let Some(parent) = self.path.parent() {
106            std::fs::create_dir_all(parent).map_err(|error| io_error(parent, error))?;
107        }
108        atomic_write(&self.path, &bytes)
109    }
110}
111
112fn load(path: &Path) -> Result<Document, Error> {
113    let bytes = match std::fs::read(path) {
114        Ok(bytes) => bytes,
115        Err(error) if error.kind() == std::io::ErrorKind::NotFound => {
116            return Ok(Document {
117                schema_version: SCHEMA_VERSION,
118                entries: Vec::new(),
119            });
120        }
121        Err(error) => return Err(io_error(path, error)),
122    };
123    let document: Document = serde_json::from_slice(&bytes).map_err(|error| {
124        Error::Filesystem(format!(
125            "invalid archive schema at {}: {error}",
126            path.display()
127        ))
128    })?;
129    if document.schema_version != SCHEMA_VERSION {
130        return Err(Error::Filesystem(format!(
131            "unsupported archive schema version {}",
132            document.schema_version
133        )));
134    }
135    Ok(document)
136}
137
138fn atomic_write(path: &Path, bytes: &[u8]) -> Result<(), Error> {
139    let temporary = path.with_extension("tmp");
140    let mut file =
141        std::fs::File::create(&temporary).map_err(|error| io_error(&temporary, error))?;
142    use std::io::Write as _;
143    file.write_all(bytes)
144        .map_err(|error| io_error(&temporary, error))?;
145    file.sync_all()
146        .map_err(|error| io_error(&temporary, error))?;
147    drop(file);
148    std::fs::rename(&temporary, path).map_err(|error| io_error(path, error))
149}
150
151fn io_error(path: &Path, error: std::io::Error) -> Error {
152    Error::Filesystem(format!("{}: {error}", path.display()))
