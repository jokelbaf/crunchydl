16/// Options used to turn playback manifests into an immutable plan.
17#[derive(Clone, Debug, Default)]
18pub struct PlanningOptions {
19    /// Audio versions to open.
20    pub audio: AudioSelection,
21    /// Whether equivalent video is shared between dubs.
22    pub video: VideoSelection,
23    /// Raw video or one exact hardsub locale.
24    pub hardsub: HardSubSelection,
25    /// Subtitle and caption selection.
26    pub subtitles: SubtitleSelection,
27    /// Video quality policy.
28    pub video_quality: QualitySelection,
29    /// Audio quality policy.
30    pub audio_quality: QualitySelection,
31    /// Quality and locale fallback policy.
32    pub fallback: FallbackPolicy,
33    /// Chapter source.
34    pub chapters: ChapterSelection,
35}

37/// The kind of a planned media track.
38#[derive(Clone, Copy, Debug, PartialEq, Eq, serde::Deserialize, serde::Serialize)]
39pub enum PlannedTrackKind {
40    /// AVC or another selected video representation.
41    Video,
42    /// An audio representation.
43    Audio,
44}
45
46/// Safe, immutable diagnostics for one selected representation.
47#[derive(Clone, Debug, PartialEq, Eq, serde::Deserialize, serde::Serialize)]
48pub struct PlannedTrack {
49    /// Video or audio.
50    pub kind: PlannedTrackKind,
51    /// Version content id that supplied this track.
52    pub version_id: String,
53    /// Audio locale for audio tracks and the source version for video tracks.
54    pub locale: Locale,
55    /// Codec descriptor from the MPD.
56    pub codec: String,
57    /// Representation bandwidth.
58    pub bandwidth: u64,
59    /// Video dimensions when this is a video track.
60    pub dimensions: Option<(u32, u32)>,
61    /// Audio sampling rate when this is an audio track.
62    pub sampling_rate: Option<u32>,
63    /// Number of media segments, excluding initialization.
64    pub segment_count: usize,
65    /// Stable descriptor that excludes signed URL query values.
66    pub representation_fingerprint: String,
67    /// Whether the representation carries DRM metadata.
68    pub encrypted: bool,
69}
70
71/// A selected subtitle or caption resource without its signed URL.
72#[derive(Clone, Debug, PartialEq, Eq, serde::Deserialize, serde::Serialize)]
73pub struct PlannedSubtitle {
74    /// Subtitle locale.
75    pub locale: Locale,
76    /// Source format, such as `ass` or `vtt`.
77    pub format: String,
78    /// Whether this resource is a closed caption.
79    pub is_caption: bool,
80    /// Whether this is a signs track for one selected audio locale.
81    pub is_signs: bool,
82    /// Stable resource identity with query and fragment removed.
83    pub resource_identity: String,
84    /// Stable output track title.
85    pub title: String,
86    /// Whether players should enable this track by default.
87    pub default: bool,
88    /// Whether players should force this track on.
89    pub forced: bool,
90}

97/// A redacted immutable playback plan suitable for diagnostics and persistence.
98#[derive(Clone, Debug, PartialEq, Eq, serde::Deserialize, serde::Serialize)]
99pub struct DownloadPlan {
100    /// Requested media id.
101    pub media_id: String,
102    /// Exact selected tracks in output order.
103    pub tracks: Vec<PlannedTrack>,
104    /// Deduplicated subtitle and caption resources.
105    pub subtitles: Vec<PlannedSubtitle>,
106    /// Neutral chapter points.
107    #[serde(skip)]
108    pub chapters: Vec<Chapter>,
109    /// Every configured fallback used during planning.
110    #[serde(skip)]
111    pub warnings: Vec<DownloadWarning>,
112    /// Stable SHA-256 fingerprint over only redacted plan descriptors.
113    pub fingerprint: String,
114}

132impl fmt::Debug for PlannedSource {
133    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
134        formatter
135            .debug_struct("PlannedSource")
136            .field("diagnostic", &self.diagnostic)
137            .field("media_segments", &self.media.len())
138            .finish_non_exhaustive()
139    }
140}
