33/// DRM system whose PSSH should be sent to the configured provider.
34#[derive(Clone, Copy, Debug, PartialEq, Eq)]
35#[non_exhaustive]
36pub enum DrmSystem {
37    /// Microsoft PlayReady.
38    PlayReady,
39    /// Google Widevine.
40    Widevine,
41}
42
43impl DrmSystem {
44    /// Return the production Crunchyroll license endpoint for this DRM system.
45    ///
46    /// Callers normally do not need to configure an endpoint. An explicit
47    /// override remains available through
48    /// [`DownloaderBuilder::drm_with_license_endpoint`](crate::DownloaderBuilder::drm_with_license_endpoint)
49    /// for testing, proxies, or future service changes.
50    #[must_use]
51    pub const fn default_license_endpoint(self) -> &'static str {
52        match self {
53            Self::PlayReady => {
54                "https://cr-license-proxy.prd.crunchyrollsvc.com/v1/license/playReady"
55            }
56            Self::Widevine => "https://cr-license-proxy.prd.crunchyrollsvc.com/v1/license/widevine",
57        }
58    }
59}

61/// Whether staging is retained after success or failure.
62#[derive(Clone, Copy, Debug, Default, PartialEq, Eq)]
63#[non_exhaustive]
64pub enum RetentionPolicy {
65    /// Remove staging after every terminal result.
66    #[default]
67    Delete,
68    /// Retain staging after failure or cancellation, but remove it after success.
69    KeepOnFailure,
70    /// Always retain staging.
71    KeepAlways,
72}

74/// Filesystem and metadata options for a completed output.
75#[derive(Clone, Debug)]
76pub struct OutputOptions {
77    /// Root beneath which the output is committed.
78    pub root: PathBuf,
79    /// Compiled filename template.
80    pub filename: FilenameTemplate,
81    /// Optional hierarchical layout. When set, this takes precedence over
82    /// [`OutputOptions::filename`].
83    pub layout: Option<OutputLayoutTemplate>,
84    /// Native output container.
85    pub container: crate::Container,
86    /// Existing-output policy.
87    pub overwrite: OverwritePolicy,
88    /// Staging retention policy.
89    pub retention: RetentionPolicy,
90    /// Maximum sanitized filename component length before the `.mkv` suffix.
91    pub max_component_length: usize,
92}
93
94impl OutputOptions {
95    /// Create Matroska output options rooted at `root`.
96    ///
97    /// # Errors
98    ///
99    /// Returns an error only if the built-in filename template is invalid.
100    pub fn new(root: impl Into<PathBuf>) -> Result<Self, Error> {
101        Ok(Self {
102            root: root.into(),
103            filename: FilenameTemplate::compile("{series} - {episode} - {title} [{media_id}]")?,
104            layout: None,
105            container: crate::Container::Matroska,
106            overwrite: OverwritePolicy::Fail,
107            retention: RetentionPolicy::Delete,
108            max_component_length: 180,
109        })
110    }
111}

194pub(crate) async fn run(
195    api: &ProductionApi,
196    crunchyroll: &crunchyroll_rs::Crunchyroll,
197    platform: &StreamPlatform,
198    runtime: &RuntimeConfiguration,
199    request: DownloadRequest,
200) -> Result<DownloadResult, Error> {
201    let cancellation = request.cancellation.clone();
202    let states = StateEmitter::new(runtime.events.clone(), cancellation.clone());
203    let result = run_inner(api, crunchyroll, platform, runtime, request, &states).await;
204    match result {
205        Ok(value) => {
206            runtime
207                .events
208                .emit(DownloadEvent::StateChanged(JobState::Completed));
209            Ok(value)
210        }
211        Err(Error::Cancelled) => {
212            runtime
213                .events
214                .emit(DownloadEvent::StateChanged(JobState::Cancelled));
215            Err(Error::Cancelled)
216        }
217        Err(error) => {
218            runtime
219                .events
220                .emit(DownloadEvent::StateChanged(JobState::Failed));
221            Err(error)
222        }
223    }
224}

226async fn run_inner(
227    api: &ProductionApi,
228    crunchyroll: &crunchyroll_rs::Crunchyroll,
229    platform: &StreamPlatform,
230    runtime: &RuntimeConfiguration,
231    request: DownloadRequest,
232    states: &StateEmitter,
233) -> Result<DownloadResult, Error> {
234    let cancellation = request.cancellation.clone();
235    states.enter(JobState::Created)?;
236    states.enter(JobState::ResolvingMedia)?;
237    let media = request.media.resolve();
238    validate_media(&media)?;
239    let archive_key = ArchiveKey {
240        media_id: media.content_id.clone(),
241        selection_fingerprint: crate::redaction::fingerprint([
242            format!("{:?}", request.planning),
243            format!("{:?}", request.subtitles),
244            format!("{:?}", request.synchronization),
245            format!("{:?}", request.output.container),
246        ]),
247    };
248    if let Some(archive) = &runtime.archive
249        && let Some(entry) = archive.find(&archive_key)?
250    {
251        return Ok(DownloadResult {
252            output: entry.output,
253            media_id: media.content_id,
254            tracks: entry.tracks,
255            warnings: Vec::new(),
256        });
257    }
258    std::fs::create_dir_all(&request.output.root)
259        .map_err(|error| path_error(&request.output.root, error))?;
260
261    states.enter(JobState::OpeningPlaybackSessions)?;
262    let mut guard = SessionGuard::new(api);
263    states.enter(JobState::PlanningTracks)?;
264    let prepared = crate::plan::prepare_with_guard(
265        api,
266        platform,
267        &media,
268        &request.planning,
269        &cancellation,
270        &mut guard,
271    )
272    .await;
273    let result = match prepared {
274        Ok(prepared) => {
275            run_prepared(
276                api,
277                crunchyroll,
278                platform,
279                runtime,
280                &request,
281                &media,
282                &archive_key,
283                prepared,
284                states,
285            )
286            .await
287        }
288        Err(error) => Err(error),
289    };
290    let finalization = guard.finalize().await;
291    match (result, finalization) {
292        (Ok(value), Ok(())) => Ok(value),
293        (Ok(_), Err(error)) | (Err(error), _) => Err(error),
294    }
295}

297#[allow(clippy::too_many_arguments)]
298async fn run_prepared(
299    api: &ProductionApi,
300    crunchyroll: &crunchyroll_rs::Crunchyroll,
301    platform: &StreamPlatform,
302    runtime: &RuntimeConfiguration,
303    request: &DownloadRequest,
304    media: &ResolvedMedia,
305    archive_key: &ArchiveKey,
306    prepared: PreparedPlan,
307    states: &StateEmitter,
308) -> Result<DownloadResult, Error> {
309    validate_container_contract(request, &prepared)?;
310    for warning in &prepared.public.warnings {
311        runtime.events.emit(DownloadEvent::Warning(warning.clone()));
312    }
313    let height = prepared
314        .public
315        .tracks
316        .iter()
317        .filter_map(|track| track.dimensions.map(|(_, height)| height))
318        .max();
319    let audio = prepared
320        .public
321        .tracks
322        .iter()
323        .filter(|track| track.kind == PlannedTrackKind::Audio)
324        .map(|track| track.locale.to_string())
325        .collect::<Vec<_>>();
326    let mut relative = request.output.layout.as_ref().map_or_else(
327        || {
328            request
329                .output
330                .filename
331                .render(media, height, &audio, request.output.max_component_length)
332                .into()
333        },
334        |layout| layout.render(media, height, &audio, request.output.max_component_length),
335    );
336    relative.set_extension(match request.output.container {
337        crate::Container::Matroska => "mkv",
338        crate::Container::Mp4 => "mp4",
339    });
340    let destination = output_path(&request.output.root, &relative)?;
341    if destination.exists() && request.output.overwrite != OverwritePolicy::Replace {
342        return Err(Error::Filesystem(format!(
343            "output already exists: {}",
344            destination.display()
345        )));
346    }
347    if let Some(parent) = destination.parent() {
348        std::fs::create_dir_all(parent).map_err(|error| path_error(parent, error))?;
349    }
350    let staging = request
351        .output
352        .root
353        .join(".crunchydl-staging")
354        .join(safe_component(&media.content_id))
355        .join(&prepared.public.fingerprint[..16]);
356    std::fs::create_dir_all(&staging).map_err(|error| path_error(&staging, error))?;
357    let outcome = execute_pipeline(
358        api,
359        crunchyroll,
360        platform,
361        runtime,
362        request,
363        media,
364        archive_key,
365        &prepared,
366        &staging,
367        &destination,
368        states,
369    )
370    .await;
371    cleanup_staging(&staging, request.output.retention, outcome.is_ok());
372    outcome
373}

375#[allow(clippy::too_many_arguments)]
376async fn execute_pipeline(
377    api: &ProductionApi,
378    crunchyroll: &crunchyroll_rs::Crunchyroll,
379    platform: &StreamPlatform,
380    runtime: &RuntimeConfiguration,
381    request: &DownloadRequest,
382    media: &ResolvedMedia,
383    archive_key: &ArchiveKey,
384    prepared: &PreparedPlan,
385    staging: &Path,
386    destination: &Path,
387    states: &StateEmitter,
388) -> Result<DownloadResult, Error> {
389    let refresher = PlaybackRefresher::new(
390        api,
391        platform,
392        media,
393        &request.planning,
394        &request.cancellation,
395    );
396    let result = execute_pipeline_inner(
397        api,
398        crunchyroll,
399        runtime,
400        request,
401        media,
402        archive_key,
403        prepared,
404        staging,
405        destination,
406        states,
407        &refresher,
408    )
409    .await;
410    let finalization = refresher.finalize().await;
411    match (result, finalization) {
412        (Ok(value), Ok(())) => Ok(value),
413        (Ok(_), Err(error)) | (Err(error), _) => Err(error),
414    }
415}

417#[allow(clippy::too_many_arguments)]
418async fn execute_pipeline_inner(
419    api: &ProductionApi,
420    crunchyroll: &crunchyroll_rs::Crunchyroll,
421    runtime: &RuntimeConfiguration,
422    request: &DownloadRequest,
423    media: &ResolvedMedia,
424    archive_key: &ArchiveKey,
425    prepared: &PreparedPlan,
426    staging: &Path,
427    destination: &Path,
428    states: &StateEmitter,
429    refresher: &PlaybackRefresher<'_>,
430) -> Result<DownloadResult, Error> {
431    let transfer = TransferEngine::with_client(crunchyroll.client(), request.transfer.clone())?
432        .event_sink(runtime.events.clone());
433    let transport = CrunchyrollLicenseTransport::new(crunchyroll.clone());
434    let mut warnings = prepared.public.warnings.clone();
435    states.enter(JobState::AcquiringLicenses)?;
436    let mut licensed = Vec::with_capacity(prepared.sources.len());
437    for (index, source) in prepared.sources.iter().enumerate() {
438        let transfer_plan =
439            source_transfer_plan(source, &prepared.public.fingerprint, &media.content_id);
440        let init_path = transfer
441            .transfer_init(&transfer_plan, staging, &request.cancellation)
442            .await?;
443        let init = std::fs::read(&init_path).map_err(|error| path_error(&init_path, error))?;
444        let key = if let Some(drm) = &source.drm {
445            let config = runtime
446                .drm
447                .as_ref()
448                .ok_or_else(|| Error::License(drm::Error::License))?;
449            let info = inspect_encryption(&init)?;
450            let pssh = select_pssh(drm, config.system)?;
451            let request_data = DrmRequest {
452                endpoint: config.endpoint.clone(),
453                content_id: source.diagnostic.version_id.clone(),
454                playback_token: drm.token.clone(),
455                content_type: match source.diagnostic.kind {
456                    PlannedTrackKind::Video => ContentType::Video,
457                    PlannedTrackKind::Audio => ContentType::Audio,
458                },
459                pssh,
460            };
461            let keys = config
462                .provider
463                .acquire_keys(request_data, &transport)
464                .await?;
465            Some(keys.require(info.default_kid)?.clone())
466        } else {
467            None
468        };
469        licensed.push((source, transfer_plan, key));
470        runtime.events.emit(DownloadEvent::StageProgress {
471            state: JobState::AcquiringLicenses,
472            completed: index + 1,
473            total: prepared.sources.len(),
474        });
475    }
476
477    states.enter(JobState::Downloading)?;
478    let mut transferred = Vec::with_capacity(licensed.len());
479    let licensed_count = licensed.len();
480    for (index, (source, transfer_plan, key)) in licensed.into_iter().enumerate() {
481        let result = transfer
482            .transfer_with_refresh(transfer_plan, staging, &request.cancellation, refresher)
483            .await?;
484        transferred.push((source, result, key));
485        runtime.events.emit(DownloadEvent::StageProgress {
486            state: JobState::Downloading,
487            completed: index + 1,
488            total: licensed_count,
489        });
490    }
491
492    states.enter(JobState::Decrypting)?;
493    let mut assembled = Vec::with_capacity(transferred.len());
494    let transferred_count = transferred.len();
495    for (index, (source, transferred, key)) in transferred.into_iter().enumerate() {
496        let output = staging.join(format!("track-{index:03}.mp4"));
497        assemble_track(
498            &transferred.init,
499            &transferred.segments,
500            &output,
501            key.as_ref(),
502            &request.cancellation,
503        )?;
504        let file = File::open(&output).map_err(|error| path_error(&output, error))?;
505        let parsed = FragmentedMp4::open(BufReader::new(file))?;
506        if parsed.tracks().len() != 1 {
507            return Err(Error::MediaParse(media::Error::Unsupported(
508                "assembled representation must contain one track",
509            )));
510        }
511        assembled.push((output, source.diagnostic.clone()));
512        runtime.events.emit(DownloadEvent::StageProgress {
513            state: JobState::Decrypting,
514            completed: index + 1,
515            total: transferred_count,
516        });
517    }
518
519    states.enter(JobState::ProcessingSubtitles)?;
520    let subtitles = download_selected(api, &prepared.subtitles, &request.subtitles).await?;
521    runtime.events.emit(DownloadEvent::StageProgress {
522        state: JobState::ProcessingSubtitles,
523        completed: subtitles.len(),
524        total: prepared.subtitles.len(),
525    });
526    let missing = MissingFontResolver;
527    let resolver: &dyn FontResolver = runtime.font_resolver.as_deref().unwrap_or(&missing);
528    let (fonts, font_warnings) =
529        crate::resolve_referenced_fonts(&subtitles, resolver, runtime.font_policy)?;
530    for warning in &font_warnings {
531        runtime.events.emit(DownloadEvent::Warning(warning.clone()));
532    }
533    warnings.extend(font_warnings);
534
535    states.enter(JobState::Muxing)?;
536    let temporary = staging.join(match request.output.container {
537        crate::Container::Matroska => "output.mkv.part",
538        crate::Container::Mp4 => "output.mp4.part",
539    });
540    let output_tracks = match request.output.container {
541        crate::Container::Matroska => mux_matroska(
542            &temporary,
543            &assembled,
544            &subtitles,
545            &fonts,
546            &prepared.public.chapters,
547            media,
548            &request.synchronization,
549            &request.cancellation,
550        )?,
551        crate::Container::Mp4 => crate::mp4_output::write_and_verify(
552            &temporary,
553            &assembled,
554            &request.synchronization,
555            &request.cancellation,
556        )?,
557    };
558    runtime.events.emit(DownloadEvent::StageProgress {
559        state: JobState::Muxing,
560        completed: 1,
561        total: 1,
562    });
563
564    states.enter(JobState::Verifying)?;
565    runtime.events.emit(DownloadEvent::StageProgress {
566        state: JobState::Verifying,
567        completed: 1,
568        total: 1,
569    });
570    states.enter(JobState::Committing)?;
571    commit_output(
572        &temporary,
573        destination,
574        request.output.overwrite,
575        runtime.archive.as_deref(),
576        archive_key,
577        &output_tracks,
578    )?;
579    runtime.events.emit(DownloadEvent::StageProgress {
580        state: JobState::Committing,
581        completed: 1,
582        total: 1,
583    });
584    runtime.events.emit(DownloadEvent::OutputCommitted {
585        output: destination.to_path_buf(),
586    });
587    Ok(DownloadResult {
588        output: destination.to_path_buf(),
589        media_id: media.content_id.clone(),
590        tracks: output_tracks,
591        warnings,
592    })
593}

595#[allow(clippy::too_many_arguments)]
596fn mux_matroska(
597    temporary: &Path,
598    assembled: &[(PathBuf, crate::PlannedTrack)],
599    subtitles: &[crate::SubtitleTrack],
600    fonts: &[crate::ResolvedFont],
601    source_chapters: &[crate::Chapter],
602    media: &ResolvedMedia,
603    synchronization: &SynchronizationOptions,
604    cancellation: &CancellationToken,
605) -> Result<Vec<OutputTrack>, Error> {
606    let (tracks, packets) = mux_inputs(assembled, subtitles, synchronization, cancellation)?;
607    let chapters = source_chapters
608        .iter()
609        .map(|chapter| mkv::Chapter {
610            start: chapter.start,
611            title: chapter.title.clone(),
612            language: language(&Locale::en_US),
613        })
614        .collect::<Vec<_>>();
615    let attachments = fonts
616        .iter()
617        .enumerate()
618        .map(|(index, font)| mkv::Attachment {
619            filename: font.filename.clone(),
620            mime_type: font.mime_type.clone(),
621            uid: index as u64 + 1,
622            data: font.data.clone(),
623        })
624        .collect::<Vec<_>>();
625    let mut output =
626        BufWriter::new(File::create(temporary).map_err(|error| path_error(temporary, error))?);
627    mkv::Muxer::write_fallible(
628        &mut output,
629        &tracks,
630        packets,
631        &chapters,
632        &attachments,
633        &mkv::MuxOptions {
634            title: Some(media.title.clone()),
635            ..mkv::MuxOptions::default()
636        },
637    )
638    .map_err(|error| match error {
639        mkv::Error::Cancelled => Error::Cancelled,
640        error => Error::Mux(error.to_string()),
641    })?;
642    output
643        .get_ref()
644        .sync_all()
645        .map_err(|error| path_error(temporary, error))?;
646    drop(output);
647    verify(temporary, tracks.len(), attachments.len(), chapters.len())
648}

650fn validate_container_contract(
651    request: &DownloadRequest,
652    prepared: &PreparedPlan,
653) -> Result<(), Error> {
654    if request.output.container != crate::Container::Mp4 {
655        return Ok(());
656    }
657    if !prepared.subtitles.is_empty() {
658        return Err(Error::Mux(
659            "MP4 output does not silently discard subtitles; select no subtitles or use Matroska"
660                .into(),
661        ));
662    }
663    if !prepared.public.chapters.is_empty() {
664        return Err(Error::Mux(
665            "MP4 output does not silently discard chapters; disable chapters or use Matroska"
666                .into(),
667        ));
668    }
669    if !request.synchronization.offsets.is_empty() {
670        return Err(Error::Mux(
671            "MP4 output does not support explicit track offsets".into(),
672        ));
673    }
674    let video = prepared
675        .public
676        .tracks
677        .iter()
678        .filter(|track| track.kind == PlannedTrackKind::Video)
679        .count();
680    let audio = prepared
681        .public
682        .tracks
683        .iter()
684        .filter(|track| track.kind == PlannedTrackKind::Audio)
685        .count();
686    if video != 1 || audio == 0 {
687        return Err(Error::Mux(
688            "MP4 output requires exactly one AVC video and at least one AAC audio track".into(),
689        ));
690    }
691    Ok(())
692}

694fn source_transfer_plan(
695    source: &PlannedSource,
696    fingerprint: &str,
697    media_id: &str,
698) -> crate::RepresentationTransferPlan {
699    let request = |segment: &crate::plan::SourceSegment| {
700        SegmentRequest::new(
701            segment.url.clone(),
702            segment.range,
703            url_identity(&segment.url),
704            segment
705                .range
706                .map(|(start, end)| end.saturating_sub(start).saturating_add(1)),
707        )
708    };
709    crate::RepresentationTransferPlan {
710        media_id: media_id.to_string(),
711        version_id: source.diagnostic.version_id.clone(),
712        plan_fingerprint: fingerprint.to_string(),
713        representation_fingerprint: source.diagnostic.representation_fingerprint.clone(),
714        track: Some(crate::TransferTrack {
715            kind: source.diagnostic.kind,
716            locale: source.diagnostic.locale.clone(),
717        }),
718        init: request(&source.init),
719        segments: source.media.iter().map(request).collect(),
720    }
721}

723struct PlaybackRefresher<'a> {
724    api: &'a ProductionApi,
725    platform: &'a StreamPlatform,
726    media: &'a ResolvedMedia,
727    options: &'a PlanningOptions,
728    cancellation: &'a CancellationToken,
729    sessions: tokio::sync::Mutex<Vec<SessionGuard<'a, ProductionApi>>>,
730}
731
732impl<'a> PlaybackRefresher<'a> {
733    fn new(
734        api: &'a ProductionApi,
735        platform: &'a StreamPlatform,
736        media: &'a ResolvedMedia,
737        options: &'a PlanningOptions,
738        cancellation: &'a CancellationToken,
739    ) -> Self {
740        Self {
741            api,
742            platform,
743            media,
744            options,
745            cancellation,
746            sessions: tokio::sync::Mutex::new(Vec::new()),
747        }
748    }
749
750    async fn finalize(&self) -> Result<(), Error> {
751        let sessions = {
752            let mut sessions = self.sessions.lock().await;
753            std::mem::take(&mut *sessions)
754        };
755        let mut first_error = None;
756        for guard in sessions {
757            if let Err(error) = guard.finalize().await
758                && first_error.is_none()
759            {
760                first_error = Some(error);
761            }
762        }
763        first_error.map_or(Ok(()), Err)
764    }
765}
766
767impl crate::RepresentationRefresher for PlaybackRefresher<'_> {
768    fn refresh<'a>(
769        &'a self,
770        expired: &'a crate::RepresentationTransferPlan,
771    ) -> Pin<Box<dyn Future<Output = Result<crate::RepresentationTransferPlan, Error>> + Send + 'a>>
772    {
773        Box::pin(async move {
774            let mut guard = SessionGuard::new(self.api);
775            let prepared = match crate::plan::prepare_with_guard(
776                self.api,
777                self.platform,
778                self.media,
779                self.options,
780                self.cancellation,
781                &mut guard,
782            )
783            .await
784            {
785                Ok(prepared) => prepared,
786                Err(error) => {
787                    let _ = guard.finalize().await;
788                    return Err(error);
789                }
790            };
791            let source = prepared
792                .sources
793                .iter()
794                .find(|source| {
795                    source.diagnostic.version_id == expired.version_id
796                        && source.diagnostic.representation_fingerprint
797                            == expired.representation_fingerprint
798                })
799                .ok_or_else(|| {
800                    Error::ResumeMismatch(
801                        "refreshed playback omitted the selected representation".into(),
802                    )
803                })?;
804            let refreshed =
805                source_transfer_plan(source, &expired.plan_fingerprint, &expired.media_id);
806            self.sessions.lock().await.push(guard);
807            Ok(refreshed)
808        })
809    }
810}

833fn assemble_track(
834    init_path: &Path,
835    segments: &[PathBuf],
836    output_path: &Path,
837    key: Option<&ContentKey>,
838    cancellation: &CancellationToken,
839) -> Result<(), Error> {
840    cancellation.check()?;
841    let init = std::fs::read(init_path).map_err(|error| path_error(init_path, error))?;
842    let mut output =
843        BufWriter::new(File::create(output_path).map_err(|error| path_error(output_path, error))?);
844    if let Some(key) = key {
845        let decrypter = crate::CencDecrypter::new(&init, key)
846            .map_err(|error| Error::Decrypt(error.to_string()))?;
847        decrypter
848            .assemble(&init, std::iter::empty(), &mut output)
849            .map_err(|error| Error::Decrypt(error.to_string()))?;
850        for segment in segments {
851            cancellation.check()?;
852            let encrypted = std::fs::read(segment).map_err(|error| path_error(segment, error))?;
853            output
854                .write_all(
855                    &decrypter
856                        .decrypt_fragment(encrypted)
857                        .map_err(|error| Error::Decrypt(error.to_string()))?,
858                )
859                .map_err(|error| path_error(output_path, error))?;
860        }
861    } else {
862        output
863            .write_all(&init)
864            .map_err(|error| path_error(output_path, error))?;
865        for segment in segments {
866            cancellation.check()?;
867            let mut input = File::open(segment).map_err(|error| path_error(segment, error))?;
868            std::io::copy(&mut input, &mut output)
869                .map_err(|error| path_error(output_path, error))?;
870        }
871    }
872    output
873        .flush()
874        .map_err(|error| path_error(output_path, error))?;
875    output
876        .get_ref()
877        .sync_all()
878        .map_err(|error| path_error(output_path, error))
879}

881fn mux_inputs(
882    paths: &[(PathBuf, crate::PlannedTrack)],
883    subtitles: &[crate::SubtitleTrack],
884    synchronization: &SynchronizationOptions,
885    cancellation: &CancellationToken,
886) -> Result<(Vec<mkv::Track>, PacketStreams), Error> {
887    let mut tracks = Vec::new();
888    let mut streams = Vec::new();
889    let mut audio_default = true;
890    let mut video_default = true;
891    let mut reference_audio_duration: Option<Duration> = None;
892    for (path, diagnostic) in paths {
893        let parsed = FragmentedMp4::open(BufReader::new(
894            File::open(path).map_err(|error| path_error(path, error))?,
895        ))?;
896        let media_track = parsed.tracks()[0].clone();
897        let number = tracks.len() as u64 + 1;
898        let offset = synchronization
899            .offsets
900            .iter()
901            .find_map(|(locale, offset)| (locale == &diagnostic.locale).then_some(*offset))
902            .unwrap_or(0);
903        let (track_type, codec, settings, default) = match (&media_track.kind, &media_track.codec) {
904            (TrackKind::Video, Codec::Avc { configuration }) => (
905                mkv::TrackType::Video,
906                mkv::TrackCodec::Avc(configuration.clone()),
907                mkv::TrackSettings::Video(mkv::VideoSettings {
908                    width: media_track
909                        .dimensions
910                        .ok_or_else(|| Error::Mux("video dimensions missing".into()))?
911                        .0,
912                    height: media_track
913                        .dimensions
914                        .ok_or_else(|| Error::Mux("video dimensions missing".into()))?
915                        .1,
916                }),
917                {
918                    let default = video_default;
919                    video_default = false;
920                    default
921                },
922            ),
923            (
924                TrackKind::Audio,
925                Codec::Aac {
926                    audio_specific_config,
927                },
928            ) => {
929                let actual_duration = FragmentedMp4::open(BufReader::new(
930                    File::open(path).map_err(|error| path_error(path, error))?,
931                ))?
932                .probe()?
933                .duration;
934                if let Some(reference) = reference_audio_duration {
935                    let difference = reference.abs_diff(actual_duration);
936                    if synchronization.strict && difference > synchronization.tolerance {
937                        return Err(Error::IncompatibleTimelines {
938                            locale: diagnostic.locale.to_string(),
939                        });
940                    }
941                } else {
942                    reference_audio_duration = Some(actual_duration);
943                }
944                let value = (
945                    mkv::TrackType::Audio,
946                    mkv::TrackCodec::Aac(audio_specific_config.clone()),
947                    mkv::TrackSettings::Audio(mkv::AudioSettings {
948                        sampling_frequency: f64::from(
949                            media_track
950                                .sample_rate
951                                .ok_or_else(|| Error::Mux("audio sampling rate missing".into()))?,
952                        ),
953                        channels: media_track
954                            .channels
955                            .ok_or_else(|| Error::Mux("audio channels missing".into()))?,
956                    }),
957                    audio_default,
958                );
959                audio_default = false;
960                value
961            }
962            _ => return Err(Error::Mux("unsupported media codec".into())),
963        };
964        tracks.push(mkv::Track {
965            number,
966            uid: 0,
967            track_type,
968            codec,
969            settings,
970            name: (track_type == mkv::TrackType::Audio)
971                .then(|| crate::locale_display_name(&diagnostic.locale)),
972            language: (track_type == mkv::TrackType::Audio).then(|| language(&diagnostic.locale)),
973            default,
974            forced: false,
975            hearing_impaired: false,
976            visual_impaired: false,
977            original: default,
978            commentary: false,
979        });
980        let packets = parsed.packets().map(move |packet| {
981            let packet = packet.map_err(|error| mkv::Error::PacketSource(error.to_string()))?;
982            let decode_time_ms = timestamp_millis(packet.dts)
983                .map_err(|error| mkv::Error::PacketSource(error.to_string()))?;
984            let presentation_time_ms = timestamp_millis(packet.pts)
985                .map_err(|error| mkv::Error::PacketSource(error.to_string()))?;
986            Ok(mkv::Packet {
987                track_number: number,
988                decode_time_ms: decode_time_ms
989                    .checked_add(offset)
990                    .ok_or(mkv::Error::Overflow("explicit track decode offset"))?,
991                presentation_time_ms: presentation_time_ms
992                    .checked_add(offset)
993                    .ok_or(mkv::Error::Overflow("explicit track presentation offset"))?,
994                duration: packet
995                    .pts
996                    .time_base()
997                    .duration(u64::from(packet.duration))
998                    .map_err(|error| mkv::Error::PacketSource(error.to_string()))?,
999                keyframe: packet.is_keyframe,
1000                data: packet.data,
1001            })
1002        });
1003        streams.push(PacketStream::new(Box::new(packets)));
1004    }
1005    for subtitle in subtitles {
1006        let number = tracks.len() as u64 + 1;
1007        let (header, packets) = ass_packets(&subtitle.ass, number)?;
1008        tracks.push(mkv::Track {
1009            number,
1010            uid: 0,
1011            track_type: mkv::TrackType::Subtitle,
1012            codec: mkv::TrackCodec::Ass(header),
1013            settings: mkv::TrackSettings::Subtitle,
1014            name: Some(subtitle.metadata.title.clone()),
1015            language: Some(language(&subtitle.metadata.locale)),
1016            default: subtitle.metadata.default,
1017            forced: subtitle.metadata.forced,
1018            hearing_impaired: subtitle.metadata.is_caption,
1019            visual_impaired: false,
1020            original: false,
1021            commentary: false,
1022        });
1023        streams.push(PacketStream::new(Box::new(packets.into_iter().map(Ok))));
1024    }
1025    Ok((
1026        tracks,
1027        PacketStreams {
1028            streams,
1029            cancellation: cancellation.clone(),
1030            cancellation_reported: false,
1031        },
1032    ))
1033}

1035type FalliblePackets = Box<dyn Iterator<Item = Result<mkv::Packet, mkv::Error>>>;
1036
1037struct PacketStream {
1038    packets: FalliblePackets,
1039    head: Option<Result<mkv::Packet, mkv::Error>>,
1040}
1041
1042impl PacketStream {
1043    fn new(mut packets: FalliblePackets) -> Self {
1044        let head = packets.next();
1045        Self { packets, head }
1046    }
1047}
1048
1049struct PacketStreams {
1050    streams: Vec<PacketStream>,
1051    cancellation: CancellationToken,
1052    cancellation_reported: bool,
1053}
1054
1055impl Iterator for PacketStreams {
1056    type Item = Result<mkv::Packet, mkv::Error>;
1057
1058    fn next(&mut self) -> Option<Self::Item> {
1059        if self.cancellation.is_cancelled() && !self.cancellation_reported {
1060            self.cancellation_reported = true;
1061            return Some(Err(mkv::Error::Cancelled));
1062        }
1063        let index = self
1064            .streams
1065            .iter()
1066            .position(|stream| stream.head.as_ref().is_some_and(Result::is_err))
1067            .or_else(|| {
1068                self.streams
1069                    .iter()
1070                    .enumerate()
1071                    .filter_map(|(index, stream)| {
1072                        stream
1073                            .head
1074                            .as_ref()
1075                            .and_then(|packet| packet.as_ref().ok())
1076                            .map(|packet| (index, packet.decode_time_ms, packet.track_number))
1077                    })
1078                    .min_by_key(|(_, decode_time_ms, track_number)| {
1079                        (*decode_time_ms, *track_number)
1080                    })
1081                    .map(|(index, _, _)| index)
1082            })?;
1083        let result = self.streams[index].head.take();
1084        self.streams[index].head = self.streams[index].packets.next();
1085        result
1086    }
1087}

1089fn timestamp_millis(timestamp: media::Timestamp) -> Result<i64, media::Error> {
1090    let numerator = i128::from(timestamp.ticks())
1091        .checked_mul(1_000)
1092        .ok_or(media::Error::Overflow("timestamp milliseconds"))?;
1093    i64::try_from(numerator / i128::from(timestamp.time_base().ticks_per_second()))
1094        .map_err(|_| media::Error::Overflow("timestamp milliseconds"))
1095}

1097fn ass_packets(ass: &str, track_number: u64) -> Result<(String, Vec<mkv::Packet>), Error> {
1098    let normalized = ass.replace("\r\n", "\n").replace('\r', "\n");
1099    let mut header = Vec::new();
1100    let mut packets = Vec::new();
1101    let mut read_order = 0_u64;
1102    for line in normalized.lines() {
1103        let Some(dialogue) = line.strip_prefix("Dialogue:") else {
1104            header.push(line);
1105            continue;
1106        };
1107        let fields = dialogue.trim_start().splitn(10, ',').collect::<Vec<_>>();
1108        if fields.len() != 10 {
1109            return Err(Error::Subtitle("malformed ASS dialogue".into()));
1110        }
1111        let start = ass_time(fields[1])?;
1112        let end = ass_time(fields[2])?;
1113        if end < start {
1114            return Err(Error::Subtitle("ASS dialogue ends before it starts".into()));
1115        }
1116        let payload = format!(
1117            "{read_order},{},{},{},{},{},{},{},{}",
1118            fields[0], fields[3], fields[4], fields[5], fields[6], fields[7], fields[8], fields[9]
1119        );
1120        packets.push(mkv::Packet {
1121            track_number,
1122            decode_time_ms: i64::try_from(start.as_millis())
1123                .map_err(|_| Error::Subtitle("subtitle timestamp overflow".into()))?,
1124            presentation_time_ms: i64::try_from(start.as_millis())
1125                .map_err(|_| Error::Subtitle("subtitle timestamp overflow".into()))?,
1126            duration: end - start,
1127            keyframe: true,
1128            data: payload.into_bytes(),
1129        });
1130        read_order += 1;
1131    }
1132    packets.sort_by_key(|packet| packet.decode_time_ms);
1133    Ok((header.join("\r\n") + "\r\n", packets))
1134}

1136fn ass_time(value: &str) -> Result<Duration, Error> {
1137    let (hours, rest) = value
1138        .trim()
1139        .split_once(':')
1140        .ok_or_else(|| Error::Subtitle("invalid ASS timestamp".into()))?;
1141    let (minutes, seconds) = rest
1142        .split_once(':')
1143        .ok_or_else(|| Error::Subtitle("invalid ASS timestamp".into()))?;
1144    let hours: u64 = hours
1145        .parse()
1146        .map_err(|_| Error::Subtitle("invalid ASS timestamp".into()))?;
1147    let minutes: u64 = minutes
1148        .parse()
1149        .map_err(|_| Error::Subtitle("invalid ASS timestamp".into()))?;
1150    let seconds: f64 = seconds
1151        .parse()
1152        .map_err(|_| Error::Subtitle("invalid ASS timestamp".into()))?;
1153    if minutes >= 60 || !seconds.is_finite() || !(0.0..60.0).contains(&seconds) {
1154        return Err(Error::Subtitle("invalid ASS timestamp".into()));
1155    }
1156    Ok(Duration::from_millis(
1157        hours
1158            .saturating_mul(3_600_000)
1159            .saturating_add(minutes * 60_000)
1160            .saturating_add((seconds * 1000.0).round() as u64),
1161    ))
1162}

1197fn verify(
1198    path: &Path,
1199    track_count: usize,
1200    attachment_count: usize,
1201    chapter_count: usize,
1202) -> Result<Vec<OutputTrack>, Error> {
1203    let parsed =
1204        matroska_reader::open(path).map_err(|error| Error::Verification(error.to_string()))?;
1205    if parsed.tracks.len() != track_count
1206        || parsed.attachments.len() != attachment_count
1207        || parsed
1208            .chapters
1209            .iter()
1210            .map(|edition| edition.chapters.len())
1211            .sum::<usize>()
1212            != chapter_count
1213        || parsed
1214            .info
1215            .duration
1216            .is_none_or(|duration| duration.is_zero())
1217    {
1218        return Err(Error::Verification(
1219            "Matroska structural verification failed".into(),
1220        ));
1221    }
1222    let markers = scan_markers(path)?;
1223    if !markers.cluster || !markers.block {
1224        return Err(Error::Verification(
1225            "Matroska contains no media blocks".into(),
1226        ));
1227    }
1228    Ok(parsed
1229        .tracks
1230        .into_iter()
1231        .map(|track| OutputTrack {
1232            codec: track.codec_id,
1233            language: track.language.map(|language| match language {
1234                matroska_reader::Language::ISO639(value)
1235                | matroska_reader::Language::IETF(value) => value,
1236            }),
1237            name: track.name,
1238            default: track.default,
1239            forced: track.forced,
1240        })
1241        .collect())
1242}

1244struct StructuralMarkers {
1245    cluster: bool,
1246    block: bool,
1247}

1249fn scan_markers(path: &Path) -> Result<StructuralMarkers, Error> {
1250    let mut reader = BufReader::new(File::open(path).map_err(|error| path_error(path, error))?);
1251    let mut carry = Vec::new();
1252    let mut markers = StructuralMarkers {
1253        cluster: false,
1254        block: false,
1255    };
1256    loop {
1257        let mut chunk = vec![0; 64 * 1024];
1258        let read = reader
1259            .read(&mut chunk)
1260            .map_err(|error| path_error(path, error))?;
1261        if read == 0 {
1262            break;
1263        }
1264        chunk.truncate(read);
1265        carry.extend(chunk);
1266        markers.cluster |= carry
1267            .windows(4)
1268            .any(|window| window == [0x1f, 0x43, 0xb6, 0x75]);
1269        markers.block |= carry.contains(&0xa3);
1270        if carry.len() > 7 {
1271            carry.drain(..carry.len() - 7);
1272        }
1273    }
1274    Ok(markers)
1275}

1277fn commit(temporary: &Path, destination: &Path, overwrite: OverwritePolicy) -> Result<(), Error> {
1278    if destination.exists() && overwrite != OverwritePolicy::Replace {
1279        return Err(Error::Filesystem(format!(
1280            "output already exists: {}",
1281            destination.display()
1282        )));
1283    }
1284    match std::fs::rename(temporary, destination) {
1285        Ok(()) => {}
1286        Err(error)
1287            if overwrite == OverwritePolicy::Replace
1288                && error.kind() == std::io::ErrorKind::AlreadyExists =>
1289        {
1290            std::fs::remove_file(destination).map_err(|error| path_error(destination, error))?;
1291            std::fs::rename(temporary, destination)
1292                .map_err(|error| path_error(destination, error))?;
1293        }
1294        Err(error) => return Err(path_error(destination, error)),
1295    }
1296    if let Some(parent) = destination.parent()
1297        && let Ok(directory) = File::open(parent)
1298    {
1299        let _ = directory.sync_all();
1300    }
1301    Ok(())
1302}

1344fn safe_component(value: &str) -> String {
1345    let value = value
1346        .chars()
1347        .map(|character| {
1348            if character.is_ascii_alphanumeric() || matches!(character, '-' | '_') {
1349                character
1350            } else {
1351                '_'
1352            }
1353        })
1354        .collect::<String>();
1355    if value.is_empty() {
1356        "media".into()
1357    } else {
