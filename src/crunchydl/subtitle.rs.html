20/// A supported subtitle source format.
21#[derive(Clone, Copy, Debug, PartialEq, Eq)]
22#[non_exhaustive]
23pub enum SubtitleFormat {
24    /// Advanced SubStation Alpha.
25    Ass,
26    /// Web Video Text Tracks.
27    WebVtt,
28}
29
30impl SubtitleFormat {
31    pub(crate) fn parse(value: &str) -> Result<Self, Error> {
32        match value.to_ascii_lowercase().as_str() {
33            "ass" | "ssa" => Ok(Self::Ass),
34            "vtt" | "webvtt" => Ok(Self::WebVtt),
35            _ => Err(Error::Subtitle("unsupported subtitle format".to_string())),
36        }
37    }
38}

96/// Convert or normalize a subtitle resource without losing track metadata.
97///
98/// Raw ASS is byte-for-byte preserved when no normalization is requested.
99/// WebVTT is converted to ASS with cue order, basic inline styling, class
100/// styles, line breaks, and percentage positioning retained.
101///
102/// # Errors
103///
104/// Returns a subtitle error for malformed required sections, timestamps, or
105/// unsupported input.
106pub fn process_subtitle(
107    source: &str,
108    format: SubtitleFormat,
109    metadata: SubtitleMetadata,
110    options: &SubtitleProcessingOptions,
111) -> Result<SubtitleTrack, Error> {
112    let mut ass = match format {
113        SubtitleFormat::Ass => source.to_string(),
114        SubtitleFormat::WebVtt => vtt_to_ass(source, &metadata.title)?,
115    };
116    if let Some(normalization) = &options.normalization {
117        ass = normalize_ass(&ass, normalization, options.media_duration)?;
118    }
119    let referenced_fonts = extract_fonts(&ass)?;
120    Ok(SubtitleTrack {
121        metadata,
122        ass,
123        referenced_fonts,
124    })
125}

127#[allow(dead_code)]
128pub(crate) async fn download_selected<A: SubtitleFetcher>(
129    api: &A,
130    subtitles: &[PreparedSubtitle],
131    options: &SubtitleProcessingOptions,
132) -> Result<Vec<SubtitleTrack>, Error> {
133    let mut tracks = Vec::with_capacity(subtitles.len());
134    for subtitle in subtitles {
135        let source = api.fetch(&subtitle.url).await?;
136        let diagnostic = &subtitle.diagnostic;
137        tracks.push(process_subtitle(
138            &source,
139            SubtitleFormat::parse(&diagnostic.format)?,
140            SubtitleMetadata {
141                locale: diagnostic.locale.clone(),
142                title: diagnostic.title.clone(),
143                is_caption: diagnostic.is_caption,
144                is_signs: diagnostic.is_signs,
145                default: diagnostic.default,
146                forced: diagnostic.forced,
147            },
148            options,
149        )?);
150    }
151    Ok(tracks)
152}

173fn vtt_to_ass(source: &str, title: &str) -> Result<String, Error> {
174    let normalized = source
175        .trim_start_matches('\u{feff}')
176        .replace("\r\n", "\n")
177        .replace('\r', "\n");
178    let blocks = text_blocks(&normalized);
179    let Some(first) = blocks.first() else {
180        return Err(Error::Subtitle("empty WebVTT document".to_string()));
181    };
182    if !first
183        .first()
184        .is_some_and(|line| line.trim_start().starts_with("WEBVTT"))
185    {
186        return Err(Error::Subtitle("missing WebVTT header".to_string()));
187    }
188    let mut styles = BTreeMap::new();
189    let mut cues = Vec::new();
190    for block in blocks.iter().skip(1) {
191        let Some(first_line) = block.first() else {
192            continue;
193        };
194        if first_line.trim_start().starts_with("NOTE") {
195            continue;
196        }
197        if first_line.trim() == "REGION" {
198            return Err(Error::Subtitle(
199                "WebVTT regions are outside the supported conversion profile".to_string(),
200            ));
201        }
202        if first_line.trim() == "STYLE" {
203            parse_css(&block[1..].join("\n"), &mut styles);
204            continue;
205        }
206        cues.push(parse_cue(block)?);
207    }
208    let mut output = vec![
209        "[Script Info]".to_string(),
210        format!("Title: {title}"),
211        "ScriptType: v4.00+".to_string(),
212        "PlayResX: 1280".to_string(),
213        "PlayResY: 720".to_string(),
214        "WrapStyle: 0".to_string(),
215        "ScaledBorderAndShadow: yes".to_string(),
216        String::new(),
217        "[V4+ Styles]".to_string(),
218        "Format: Name, Fontname, Fontsize, PrimaryColour, SecondaryColour, OutlineColour, BackColour, Bold, Italic, Underline, StrikeOut, ScaleX, ScaleY, Spacing, Angle, BorderStyle, Outline, Shadow, Alignment, MarginL, MarginR, MarginV, Encoding".to_string(),
219        ass_style("Default", "Arial", 42.0, "&H00FFFFFF", false, false, false),
220    ];
221    for style in styles.values() {
222        output.push(ass_style(
223            &style.name,
224            style.font_family.as_deref().unwrap_or("Arial"),
225            style.font_size.unwrap_or(42.0),
226            style.color.as_deref().unwrap_or("&H00FFFFFF"),
227            style.bold,
228            style.italic,
229            style.underline,
230        ));
231    }
232    output.extend([
233        String::new(),
234        "[Events]".to_string(),
235        "Format: Layer, Start, End, Style, Name, MarginL, MarginR, MarginV, Effect, Text"
236            .to_string(),
237    ]);
238    for cue in cues {
239        let (style, text) = cue_text_to_ass(&cue.text, &styles);
240        let position = cue_position(&cue.settings)?;
241        output.push(format!(
242            "Dialogue: 0,{},{},{style},,0,0,0,,{position}{text}",
243            ass_timestamp(cue.start_ms),
244            ass_timestamp(cue.end_ms)
245        ));
246    }
247    Ok(output.join("\r\n") + "\r\n")
248}

250fn text_blocks(source: &str) -> Vec<Vec<&str>> {
251    let mut blocks = Vec::new();
252    let mut current = Vec::new();
253    for line in source.lines() {
254        if line.trim().is_empty() {
255            if !current.is_empty() {
256                blocks.push(std::mem::take(&mut current));
257            }
258        } else {
259            current.push(line);
260        }
261    }
262    if !current.is_empty() {
263        blocks.push(current);
264    }
265    blocks
266}

268fn parse_cue(block: &[&str]) -> Result<Cue, Error> {
269    let time_index = block
270        .iter()
271        .position(|line| line.contains("-->"))
272        .ok_or_else(|| Error::Subtitle("WebVTT cue has no timing line".to_string()))?;
273    let timing = block[time_index];
274    let (start, remainder) = timing
275        .split_once("-->")
276        .ok_or_else(|| Error::Subtitle("invalid WebVTT timing line".to_string()))?;
277    let mut right = remainder.split_whitespace();
278    let end = right
279        .next()
280        .ok_or_else(|| Error::Subtitle("WebVTT cue has no end timestamp".to_string()))?;
281    let settings = right
282        .filter_map(|setting| setting.split_once(':'))
283        .map(|(key, value)| (key.to_ascii_lowercase(), value.to_string()))
284        .collect();
285    let start_ms = parse_vtt_timestamp(start.trim())?;
286    let end_ms = parse_vtt_timestamp(end)?;
287    if end_ms < start_ms {
288        return Err(Error::Subtitle(
289            "WebVTT cue ends before it starts".to_string(),
290        ));
291    }
292    Ok(Cue {
293        start_ms,
294        end_ms,
295        settings,
296        text: block[time_index + 1..].join("\n"),
297    })
298}

300fn parse_vtt_timestamp(value: &str) -> Result<u64, Error> {
301    let parts = value.split(':').collect::<Vec<_>>();
302    let (hours, minutes, seconds) = match parts.as_slice() {
303        [minutes, seconds] => (0_u64, parse_u64(minutes)?, *seconds),
304        [hours, minutes, seconds] => (parse_u64(hours)?, parse_u64(minutes)?, *seconds),
305        _ => return Err(Error::Subtitle("invalid WebVTT timestamp".to_string())),
306    };
307    let (seconds, millis) = seconds
308        .split_once('.')
309        .ok_or_else(|| Error::Subtitle("WebVTT timestamp lacks milliseconds".to_string()))?;
310    let seconds = parse_u64(seconds)?;
311    let millis = match millis.len() {
312        1 => parse_u64(millis)? * 100,
313        2 => parse_u64(millis)? * 10,
314        3 => parse_u64(millis)?,
315        _ => {
316            return Err(Error::Subtitle(
317                "invalid WebVTT timestamp precision".to_string(),
318            ));
319        }
320    };
321    hours
322        .checked_mul(3_600_000)
323        .and_then(|value| value.checked_add(minutes.checked_mul(60_000)?))
324        .and_then(|value| value.checked_add(seconds.checked_mul(1000)?))
325        .and_then(|value| value.checked_add(millis))
326        .ok_or_else(|| Error::Subtitle("WebVTT timestamp overflow".to_string()))
327}

335fn parse_css(css: &str, styles: &mut BTreeMap<String, CueStyle>) {
336    let mut rest = css;
337    while let Some(cue) = rest.find("::cue") {
338        rest = &rest[cue + 5..];
339        let Some(open) = rest.find('{') else { break };
340        let selector = rest[..open].trim().trim_matches(['(', ')']).trim();
341        let Some(close) = rest[open + 1..].find('}') else {
342            break;
343        };
344        let declarations = &rest[open + 1..open + 1 + close];
345        rest = &rest[open + 2 + close..];
346        let class = selector.trim_start_matches('.');
347        if class.is_empty() {
348            continue;
349        }
350        let name = style_name(class);
351        let mut style = CueStyle {
352            name: name.clone(),
353            ..CueStyle::default()
354        };
355        for declaration in declarations.split(';') {
356            let Some((key, value)) = declaration.split_once(':') else {
357                continue;
358            };
359            let value = value.trim();
360            match key.trim().to_ascii_lowercase().as_str() {
361                "font-family" => {
362                    style.font_family = Some(value.trim_matches(['\'', '"']).to_string())
363                }
364                "font-size" => style.font_size = value.trim_end_matches("px").parse().ok(),
365                "color" => style.color = css_color(value),
366                "font-weight" => {
367                    style.bold = value.eq_ignore_ascii_case("bold")
368                        || value.parse::<u16>().is_ok_and(|weight| weight >= 600)
369                }
370                "font-style" => style.italic = value.eq_ignore_ascii_case("italic"),
371                "text-decoration" => style.underline = value.contains("underline"),
372                _ => {}
373            }
374        }
375        styles.insert(class.to_string(), style);
376    }
377}

379fn css_color(value: &str) -> Option<String> {
380    let hex = value.strip_prefix('#')?;
381    if hex.len() != 6 || !hex.bytes().all(|byte| byte.is_ascii_hexdigit()) {
382        return None;
383    }
384    Some(format!("&H00{}{}{}", &hex[4..6], &hex[2..4], &hex[0..2]).to_ascii_uppercase())
385}

416fn cue_text_to_ass(text: &str, styles: &BTreeMap<String, CueStyle>) -> (String, String) {
417    let mut output = String::new();
418    let mut default_style = "Default".to_string();
419    let mut rest = text;
420    while let Some(open) = rest.find('<') {
421        output.push_str(&escape_ass_text(&rest[..open]));
422        let Some(close) = rest[open + 1..].find('>') else {
423            output.push_str(&escape_ass_text(&rest[open..]));
424            rest = "";
425            break;
426        };
427        let tag = &rest[open + 1..open + 1 + close];
428        match tag {
429            "b" => output.push_str("{\\b1}"),
430            "/b" => output.push_str("{\\b0}"),
431            "i" => output.push_str("{\\i1}"),
432            "/i" => output.push_str("{\\i0}"),
433            "u" => output.push_str("{\\u1}"),
434            "/u" => output.push_str("{\\u0}"),
435            "/c" => output.push_str("{\\r}"),
436            _ if tag.starts_with("c.") => {
437                let class = tag[2..].split('.').next().unwrap_or_default();
438                if let Some(style) = styles.get(class) {
439                    if output.is_empty() {
440                        default_style.clone_from(&style.name);
441                    } else {
442                        output.push_str(&format!("{{\\r{}}}", style.name));
443                    }
444                }
445            }
446            _ => {}
447        }
448        rest = &rest[open + close + 2..];
449    }
450    output.push_str(&escape_ass_text(rest));
451    (default_style, output.replace('\n', "\\N"))
452}

454fn escape_ass_text(value: &str) -> String {
455    value
456        .replace("&lt;", "<")
457        .replace("&gt;", ">")
458        .replace("&amp;", "&")
459        .replace('{', "\\{")
460        .replace('}', "\\}")
461}

463fn cue_position(settings: &BTreeMap<String, String>) -> Result<String, Error> {
464    if settings.contains_key("vertical") || settings.contains_key("size") {
465        return Err(Error::Subtitle(
466            "vertical or sized WebVTT cues are outside the supported conversion profile"
467                .to_string(),
468        ));
469    }
470    if settings
471        .get("line")
472        .is_some_and(|line| !line.ends_with('%'))
473    {
474        return Err(Error::Subtitle(
475            "line-number WebVTT positioning is outside the supported conversion profile"
476                .to_string(),
477        ));
478    }
479    let x = settings
480        .get("position")
481        .map(|value| percentage(value, 1280))
482        .transpose()?
483        .unwrap_or(640);
484    let y = settings
485        .get("line")
486        .filter(|value| value.ends_with('%'))
487        .map(|value| percentage(value, 720))
488        .transpose()?
489        .unwrap_or(648);
490    let alignment = match settings.get("align").map(String::as_str) {
491        Some("start" | "left") => "\\an1",
492        Some("end" | "right") => "\\an3",
493        Some("center") | None => "",
494        Some(_) => {
495            return Err(Error::Subtitle(
496                "unsupported WebVTT cue alignment".to_string(),
497            ));
498        }
499    };
500    if settings.contains_key("position") || settings.contains_key("line") {
501        Ok(format!("{{{alignment}\\pos({x},{y})}}"))
502    } else if alignment.is_empty() {
503        Ok(String::new())
504    } else {
505        Ok(format!("{{{alignment}}}"))
506    }
507}

509fn percentage(value: &str, scale: u32) -> Result<u32, Error> {
510    let number = value
511        .trim_end_matches('%')
512        .split(',')
513        .next()
514        .unwrap_or_default()
515        .parse::<f64>()
516        .map_err(|_| Error::Subtitle("invalid WebVTT position".to_string()))?;
517    if !(0.0..=100.0).contains(&number) {
518        return Err(Error::Subtitle(
519            "WebVTT position outside viewport".to_string(),
520        ));
521    }
522    Ok((number * f64::from(scale) / 100.0).round() as u32)
523}

525fn ass_timestamp(milliseconds: u64) -> String {
526    let centiseconds = (milliseconds + 5) / 10;
527    let hours = centiseconds / 360_000;
528    let minutes = centiseconds / 6_000 % 60;
529    let seconds = centiseconds / 100 % 60;
530    let fraction = centiseconds % 100;
531    format!("{hours}:{minutes:02}:{seconds:02}.{fraction:02}")
532}

534fn normalize_ass(
535    source: &str,
536    options: &AssNormalization,
537    duration: Option<Duration>,
538) -> Result<String, Error> {
539    let crlf = source.contains("\r\n");
540    let trailing_newline = source.ends_with('\n') || source.ends_with('\r');
541    let normalized = source.replace("\r\n", "\n").replace('\r', "\n");
542    let mut lines = normalized.lines().map(str::to_string).collect::<Vec<_>>();
543    if options.remove_project_garbage {
544        lines.retain(|line| !line.trim_start().starts_with(';'));
545        if let Some(start) = section_index(&lines, "Aegisub Project Garbage") {
546            let end = lines[start + 1..]
547                .iter()
548                .position(|line| is_section(line))
549                .map_or(lines.len(), |offset| start + 1 + offset);
550            lines.drain(start..end);
551        }
552    }
553    let script_start = section_index(&lines, "Script Info")
554        .ok_or_else(|| Error::Subtitle("ASS document has no Script Info section".to_string()))?;
555    let script_end = lines[script_start + 1..]
556        .iter()
557        .position(|line| is_section(line))
558        .map_or(lines.len(), |offset| script_start + 1 + offset);
559    let mut additions = Vec::new();
560    if let Some((x, y)) = options.play_resolution {
561        set_or_add(
562            &mut lines,
563            script_start,
564            script_end,
565            "PlayResX",
566            x.to_string(),
567            &mut additions,
568        );
569        set_or_add(
570            &mut lines,
571            script_start,
572            script_end,
573            "PlayResY",
574            y.to_string(),
575            &mut additions,
576        );
577    }
578    if let Some((x, y)) = options.layout_resolution {
579        set_or_add(
580            &mut lines,
581            script_start,
582            script_end,
583            "LayoutResX",
584            x.to_string(),
585            &mut additions,
586        );
587        set_or_add(
588            &mut lines,
589            script_start,
590            script_end,
591            "LayoutResY",
592            y.to_string(),
593            &mut additions,
594        );
595    }
596    if let Some(wrap_style) = options.wrap_style {
597        set_or_add(
598            &mut lines,
599            script_start,
600            script_end,
601            "WrapStyle",
602            wrap_style.to_string(),
603            &mut additions,
604        );
605    }
606    if let Some(timer) = options.timer {
607        set_or_add(
608            &mut lines,
609            script_start,
610            script_end,
611            "Timer",
612            format!("{timer:.4}"),
613            &mut additions,
614        );
615    }
616    if let Some(scaled) = options.scaled_border_and_shadow {
617        set_or_add(
618            &mut lines,
619            script_start,
620            script_end,
621            "ScaledBorderAndShadow",
622            if scaled { "yes" } else { "no" }.to_string(),
623            &mut additions,
624        );
625    }
626    lines.splice(script_end..script_end, additions);
627    if options.clamp_to_duration {
628        let duration = duration.ok_or_else(|| {
629            Error::Subtitle("duration clamping requested without media duration".to_string())
630        })?;
631        lines = clamp_dialogues(lines, duration)?;
632    }
633    let separator = if crlf { "\r\n" } else { "\n" };
634    let mut output = lines.join(separator);
635    if trailing_newline {
636        output.push_str(separator);
637    }
638    Ok(output)
639}

641fn set_or_add(
642    lines: &mut [String],
643    start: usize,
644    end: usize,
645    key: &str,
646    value: String,
647    additions: &mut Vec<String>,
648) {
649    let prefix = format!("{}:", key.to_ascii_lowercase());
650    if let Some(line) = lines[start + 1..end]
651        .iter_mut()
652        .find(|line| line.trim_start().to_ascii_lowercase().starts_with(&prefix))
653    {
654        *line = format!("{key}: {value}");
655    } else {
656        additions.push(format!("{key}: {value}"));
657    }
658}

660fn clamp_dialogues(lines: Vec<String>, duration: Duration) -> Result<Vec<String>, Error> {
661    let limit = u64::try_from(duration.as_millis())
662        .map_err(|_| Error::Subtitle("media duration overflow".to_string()))?;
663    let mut output = Vec::with_capacity(lines.len());
664    for line in lines {
665        if !line.trim_start().starts_with("Dialogue:") {
666            output.push(line);
667            continue;
668        }
669        let Some((prefix, body)) = line.split_once(':') else {
670            continue;
671        };
672        let mut fields = body
673            .trim_start()
674            .splitn(10, ',')
675            .map(str::to_string)
676            .collect::<Vec<_>>();
677        if fields.len() != 10 {
678            return Err(Error::Subtitle("malformed ASS Dialogue event".to_string()));
679        }
680        let start = parse_ass_timestamp(&fields[1])?;
681        let end = parse_ass_timestamp(&fields[2])?;
682        if start >= limit {
683            continue;
684        }
685        if end > limit {
686            fields[2] = ass_timestamp(limit);
687        }
688        output.push(format!("{prefix}: {}", fields.join(",")));
689    }
690    Ok(output)
691}
692
693fn parse_ass_timestamp(value: &str) -> Result<u64, Error> {
694    let (hours, rest) = value
695        .trim()
696        .split_once(':')
697        .ok_or_else(|| Error::Subtitle("invalid ASS timestamp".to_string()))?;
698    let (minutes, seconds) = rest
699        .split_once(':')
700        .ok_or_else(|| Error::Subtitle("invalid ASS timestamp".to_string()))?;
701    let (seconds, fraction) = seconds
702        .replace(',', ".")
703        .split_once('.')
704        .map(|(seconds, fraction)| (seconds.to_string(), fraction.to_string()))
705        .unwrap_or_else(|| (seconds.to_string(), "0".to_string()));
706    let fraction_ms = match fraction.len() {
707        0 => 0,
708        1 => parse_u64(&fraction)? * 100,
709        2 => parse_u64(&fraction)? * 10,
710        _ => parse_u64(&fraction.chars().take(3).collect::<String>())?,
711    };
712    parse_u64(hours)?
713        .checked_mul(3_600_000)
714        .and_then(|value| value.checked_add(parse_u64(minutes).ok()? * 60_000))
715        .and_then(|value| value.checked_add(parse_u64(&seconds).ok()? * 1000))
716        .and_then(|value| value.checked_add(fraction_ms))
717        .ok_or_else(|| Error::Subtitle("ASS timestamp overflow".to_string()))
718}

720fn extract_fonts(ass: &str) -> Result<Vec<String>, Error> {
721    let lines = ass.replace("\r\n", "\n").replace('\r', "\n");
722    let lines = lines.lines().collect::<Vec<_>>();
723    let style_start =
724        section_index_str(&lines, "V4+ Styles").or_else(|| section_index_str(&lines, "V4 Styles"));
725    let mut fonts = Vec::new();
726    if let Some(start) = style_start {
727        let end = lines[start + 1..]
728            .iter()
729            .position(|line| is_section(line))
730            .map_or(lines.len(), |offset| start + 1 + offset);
731        let format = lines[start + 1..end]
732            .iter()
733            .find_map(|line| line.trim_start().strip_prefix("Format:"))
734            .map(|format| {
735                format
736                    .split(',')
737                    .map(|field| field.trim())
738                    .collect::<Vec<_>>()
739            })
740            .ok_or_else(|| Error::Subtitle("ASS styles section has no Format line".to_string()))?;
741        let font_index = format
742            .iter()
743            .position(|field| field.eq_ignore_ascii_case("Fontname"))
744            .ok_or_else(|| Error::Subtitle("ASS style format has no Fontname".to_string()))?;
745        for line in &lines[start + 1..end] {
746            if let Some(style) = line.trim_start().strip_prefix("Style:")
747                && let Some(font) = style.split(',').nth(font_index)
748            {
749                push_font(&mut fonts, font);
750            }
751        }
752    }
753    let mut rest = ass;
754    while let Some(index) = rest.find("\\fn") {
755        rest = &rest[index + 3..];
756        let end = rest.find(['\\', '}']).unwrap_or(rest.len());
757        push_font(&mut fonts, &rest[..end]);
758        rest = &rest[end..];
759    }
760    Ok(fonts)
761}
762
763fn push_font(fonts: &mut Vec<String>, value: &str) {
764    let canonical = value.trim().trim_matches(['\'', '"']);
765    if canonical.is_empty() {
766        return;
767    }
768    if !fonts
769        .iter()
770        .any(|font| font.eq_ignore_ascii_case(canonical))
771    {
772        fonts.push(canonical.to_string());
773    }
774}

776fn section_index(lines: &[String], name: &str) -> Option<usize> {
777    lines.iter().position(|line| {
778        section_name(line).is_some_and(|section| section.eq_ignore_ascii_case(name))
779    })
780}
781
782fn section_index_str(lines: &[&str], name: &str) -> Option<usize> {
783    lines.iter().position(|line| {
784        section_name(line).is_some_and(|section| section.eq_ignore_ascii_case(name))
785    })
786}
787
788fn is_section(line: &str) -> bool {
789    section_name(line).is_some()
790}
791
792fn section_name(line: &str) -> Option<&str> {
793    line.trim().strip_prefix('[')?.strip_suffix(']')
794}
