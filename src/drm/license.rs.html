10/// A boxed asynchronous operation used by object-safe DRM traits.
11pub type BoxFuture<'a, T> = Pin<Box<dyn Future<Output = T> + Send + 'a>>;

13/// The kind of media content named in a license request.
14#[non_exhaustive]
15#[derive(Clone, Copy, Debug, PartialEq, Eq)]
16pub enum ContentType {
17    /// Video content.
18    Video,
19    /// Audio content.
20    Audio,
21}
22
23/// Inputs required to acquire keys for one protection-system initialization.
24///
25/// This type deliberately does not implement `Debug` because it owns PSSH and
26/// playback-token material.
27pub struct DrmRequest {
28    /// Configured Crunchyroll license endpoint.
29    pub endpoint: String,
30    /// Modern content identifier.
31    pub content_id: String,
32    /// Playback session token.
33    pub playback_token: String,
34    /// Media kind associated with the protected track.
35    pub content_type: ContentType,
36    /// Binary PSSH box or protection-system payload.
37    pub pssh: Vec<u8>,
38}

40/// An outbound license request generated by a DRM provider.
41///
42/// This type deliberately does not implement `Debug` or serialization traits.
43pub struct LicenseRequest {
44    /// Configured license endpoint.
45    pub endpoint: String,
46    /// Modern content identifier for `x-cr-content-id`.
47    pub content_id: String,
48    /// Playback token for `x-cr-video-token`.
49    pub playback_token: String,
50    /// Whether the challenge belongs to video or audio.
51    pub content_type: ContentType,
52    /// Binary or XML license challenge.
53    pub challenge: Vec<u8>,
54}
55
56/// Raw license bytes returned by a transport.
57///
58/// This type deliberately does not implement `Debug` or serialization traits.
59pub struct LicenseResponse(Vec<u8>);
60
61impl LicenseResponse {
62    /// Wrap a binary license response.
63    #[must_use]
64    pub fn binary(bytes: Vec<u8>) -> Self {
65        Self(bytes)
66    }
67
68    /// Decode a binary response or a JSON object containing a base64 license.
69    ///
70    /// Accepted JSON field names are `license`, `data`, and `payload`.
71    ///
72    /// # Errors
73    ///
74    /// Returns [`Error::License`] for malformed JSON or base64 data.
75    pub fn from_http(content_type: Option<&str>, body: Vec<u8>) -> Result<Self, Error> {
76        if !content_type.is_some_and(|value| value.to_ascii_lowercase().contains("json")) {
77            return Ok(Self::binary(body));
78        }
79        let value: serde_json::Value = serde_json::from_slice(&body).map_err(|_| Error::License)?;
80        let encoded = ["license", "data", "payload"]
81            .iter()
82            .find_map(|field| value.get(field).and_then(serde_json::Value::as_str))
83            .ok_or(Error::License)?;
84        let bytes = base64::engine::general_purpose::STANDARD
85            .decode(encoded)
86            .map_err(|_| Error::License)?;
87        Ok(Self(bytes))
88    }
89
90    /// Borrow the raw response for backend-specific license parsing.
91    #[must_use]
92    pub fn as_bytes(&self) -> &[u8] {
93        &self.0
94    }
95}

97/// Application-provided transport for DRM license challenges.
98pub trait LicenseTransport: Send + Sync {
99    /// Send one license request and return its response body.
100    fn send<'a>(&'a self, request: LicenseRequest)
101    -> BoxFuture<'a, Result<LicenseResponse, Error>>;
102}
103
104/// A DRM backend that turns PSSH and a license response into content keys.
105pub trait DrmProvider: Send + Sync {
106    /// Generate a challenge, exchange it through `transport`, and parse keys.
107    fn acquire_keys<'a>(
108        &'a self,
109        request: DrmRequest,
110        transport: &'a dyn LicenseTransport,
111    ) -> BoxFuture<'a, Result<KeySet, Error>>;
112}
113
114/// Acquire and match keys for independently licensed tracks.
115///
116/// Results preserve request order. Each returned key is checked against that
117/// track's declared default KID before the next license is requested.
118///
119/// # Errors
120///
121/// Returns the first provider, transport, or missing-key error.
122pub fn acquire_track_keys<'a>(
123    provider: &'a dyn DrmProvider,
124    transport: &'a dyn LicenseTransport,
125    tracks: Vec<(KeyId, DrmRequest)>,
126) -> BoxFuture<'a, Result<Vec<ContentKey>, Error>> {
127    Box::pin(async move {
128        let mut matched = Vec::with_capacity(tracks.len());
129        for (kid, request) in tracks {
130            let keys = provider.acquire_keys(request, transport).await?;
131            matched.push(keys.require(kid)?.clone());
132        }
133        Ok(matched)
134    })
135}

145            &'a self,

152    struct FixtureProvider;
153
154    impl DrmProvider for FixtureProvider {
155        fn acquire_keys<'a>(
156            &'a self,
157            request: DrmRequest,
158            transport: &'a dyn LicenseTransport,
159        ) -> BoxFuture<'a, Result<KeySet, Error>> {
160            Box::pin(async move {
161                let response = transport
162                    .send(LicenseRequest {
163                        endpoint: request.endpoint,
164                        content_id: request.content_id,
165                        playback_token: request.playback_token,
166                        content_type: request.content_type,
167                        challenge: request.pssh,
168                    })
169                    .await?;
170                let marker = *response.as_bytes().first().ok_or(Error::License)?;
171                let mut keys = KeySet::new();
172                keys.insert(
173                    KeyId::new([marker; 16]),
174                    ContentKey::try_from_slice(&[marker.wrapping_add(1); 16])?,
175                );
176                Ok(keys)
177            })
178        }
179    }

195        assert_eq!(response.as_bytes(), &[0, 255, 17]);
