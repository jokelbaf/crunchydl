10/// A supported ISO Common Encryption scheme.
11#[derive(Clone, Copy, Debug, PartialEq, Eq)]
12pub enum EncryptionScheme {
13    /// AES-CTR full-sample encryption.
14    Cenc,
15    /// AES-CTR pattern encryption.
16    Cens,
17    /// AES-CBC full-sample encryption.
18    Cbc1,
19    /// AES-CBC pattern encryption.
20    Cbcs,
21}
22
23/// Encryption defaults parsed from an initialization segment before licensing.
24#[derive(Clone, Copy, Debug, PartialEq, Eq)]
25pub struct EncryptionInfo {
26    /// Default content-key identifier declared by `tenc`.
27    pub default_kid: KeyId,
28    /// Common Encryption scheme declared by `schm`.
29    pub scheme: EncryptionScheme,
30}
31
32/// Inspect an encrypted initialization segment without requiring a content key.
33///
34/// # Errors
35///
36/// Returns a typed layout error when required `tenc` or `schm` metadata is
37/// absent, malformed, or unsupported.
38pub fn inspect_encryption(init: &[u8]) -> Result<EncryptionInfo, Error> {
39    validate_boxes(init, 0)?;
40    let tenc = TencBox::from_init(init)
41        .map_err(|_| Error::UnsupportedLayout("malformed tenc box".to_string()))?
42        .ok_or_else(|| Error::UnsupportedLayout("missing tenc box".to_string()))?;
43    if !tenc.is_protected {
44        return Err(Error::UnsupportedLayout(
45            "tenc does not mark the track protected".to_string(),
46        ));
47    }
48    let scheme_bytes = find_box_payload(init, b"schm", 0)?
49        .ok_or_else(|| Error::UnsupportedLayout("missing schm box".to_string()))?;
50    if scheme_bytes.len() < 12 {
51        return Err(Error::UnsupportedLayout("truncated schm box".to_string()));
52    }
53    Ok(EncryptionInfo {
54        default_kid: KeyId::new(tenc.default_kid),
55        scheme: EncryptionScheme::parse(&scheme_bytes[4..8])?,
56    })
57}
58
59impl EncryptionScheme {
60    fn parse(value: &[u8]) -> Result<Self, Error> {
61        match value {
62            b"cenc" => Ok(Self::Cenc),
63            b"cens" => Ok(Self::Cens),
64            b"cbc1" => Ok(Self::Cbc1),
65            b"cbcs" => Ok(Self::Cbcs),
66            _ => Err(Error::UnsupportedLayout(format!(
67                "unknown protection scheme {}",
68                String::from_utf8_lossy(value)
69            ))),
70        }
71    }
72}

74/// A local replaceable wrapper around `vsd-mp4` decryption.
75pub struct CencDecrypter {
76    inner: VsdDecrypter,
77    default_kid: KeyId,
78    scheme: EncryptionScheme,
79}
80
81impl CencDecrypter {
82    /// Parse encryption defaults from one initialization segment.
83    ///
84    /// # Errors
85    ///
86    /// Returns a typed layout error when required `tenc` or `schm` metadata is
87    /// absent, malformed, or unsupported.
88    pub fn new(init: &[u8], key: &ContentKey) -> Result<Self, Error> {
89        let info = inspect_encryption(init)?;
90        let inner = VsdDecrypter::with_init(&hex(key.expose()), init)
91            .map_err(|_| Error::UnsupportedLayout("unsupported init box layout".to_string()))?;
92        Ok(Self {
93            inner,
94            default_kid: info.default_kid,
95            scheme: info.scheme,
96        })
97    }
98
99    /// Default content-key identifier declared by the init segment.
100    #[must_use]
101    pub const fn default_kid(&self) -> KeyId {
102        self.default_kid
103    }
104
105    /// Encryption scheme declared by the init segment.
106    #[must_use]
107    pub const fn scheme(&self) -> EncryptionScheme {
108        self.scheme
109    }
110
111    /// Decrypt one single-track `moof`/`mdat` fragment and neutralize auxiliary
112    /// encryption signaling.
113    ///
114    /// # Errors
115    ///
116    /// Returns [`Error::UnsupportedLayout`] for malformed, multi-track, or
117    /// incomplete fragments and [`Error::Decrypt`] for cryptographic failure.
118    pub fn decrypt_fragment(&self, fragment: Vec<u8>) -> Result<Vec<u8>, Error> {
119        validate_fragment(&fragment)?;
120        let mut output = self
121            .inner
122            .decrypt_fragment(fragment, None)
123            .map_err(|_| Error::Decrypt)?;
124        neutralize_fragment(&mut output)?;
125        Ok(output)
126    }
127
128    /// Write a clear init segment followed by ordered decrypted fragments.
129    ///
130    /// Fragment indices must be contiguous and start at zero.
131    ///
132    /// # Errors
133    ///
134    /// Returns a typed error for an ordering, MP4 layout, decryption, or output
135    /// failure.
136    pub fn assemble<W, I>(&self, init: &[u8], fragments: I, writer: &mut W) -> Result<(), Error>
137    where
138        W: Write,
139        I: IntoIterator<Item = (usize, Vec<u8>)>,
140    {
141        let mut clear_init = init.to_vec();
142        neutralize_init(&mut clear_init)?;
143        writer.write_all(&clear_init)?;
144        for (expected, (index, fragment)) in fragments.into_iter().enumerate() {
145            if index != expected {
146                return Err(Error::UnsupportedLayout(format!(
147                    "fragment index {index} appeared where {expected} was required"
148                )));
149            }
150            writer.write_all(&self.decrypt_fragment(fragment)?)?;
151        }
152        writer.flush()?;
153        Ok(())
154    }
155}

157fn validate_fragment(data: &[u8]) -> Result<(), Error> {
158    validate_boxes(data, 0)?;
159    let moof = count_box(data, b"moof", 0)?;
160    let mdat = count_box(data, b"mdat", 0)?;
161    let traf = count_box(data, b"traf", 0)?;
162    let trun = count_box(data, b"trun", 0)?;
163    let senc = count_box(data, b"senc", 0)?;
164    if (moof, mdat, traf, trun, senc) != (1, 1, 1, 1, 1) {
165        return Err(Error::UnsupportedLayout(format!(
166            "expected one moof/mdat/traf/trun/senc, found {moof}/{mdat}/{traf}/{trun}/{senc}"
167        )));
168    }
169    Ok(())
170}
171
172fn neutralize_init(data: &mut [u8]) -> Result<(), Error> {
173    walk_boxes_mut(data, 0, &mut |name| match &*name {
174        b"encv" => *name = *b"avc1",
175        b"enca" => *name = *b"mp4a",
176        _ => {}
177    })
178}
179
180fn neutralize_fragment(data: &mut [u8]) -> Result<(), Error> {
181    walk_boxes_mut(data, 0, &mut |name| {
182        if matches!(&*name, b"senc" | b"saiz" | b"saio") {
183            *name = *b"free";
184        }
185    })
186}

188fn validate_boxes(data: &[u8], child_offset: usize) -> Result<(), Error> {
189    walk_boxes(data, child_offset, &mut |_, _| Ok(()))
190}
191
192fn count_box(data: &[u8], target: &[u8; 4], child_offset: usize) -> Result<usize, Error> {
193    let mut count = 0;
194    walk_boxes(data, child_offset, &mut |name, _| {
195        if name == target {
196            count += 1;
197        }
198        Ok(())
199    })?;
200    Ok(count)
201}

203fn find_box_payload<'a>(
204    data: &'a [u8],
205    target: &[u8; 4],
206    child_offset: usize,
207) -> Result<Option<&'a [u8]>, Error> {
208    let mut found = None;
209    walk_boxes(data, child_offset, &mut |name, payload| {
210        if name == target && found.is_none() {
211            found = Some(payload);
212        }
213        Ok(())
214    })?;
215    Ok(found)
216}

218fn walk_boxes<'a>(
219    data: &'a [u8],
220    child_offset: usize,
221    visit: &mut impl FnMut(&[u8; 4], &'a [u8]) -> Result<(), Error>,
222) -> Result<(), Error> {
223    if child_offset > data.len() {
224        return Err(Error::UnsupportedLayout("invalid child offset".to_string()));
225    }
226    let mut position = child_offset;
227    while position < data.len() {
228        let (size, header) = box_size(data, position)?;
229        let end = position
230            .checked_add(size)
231            .ok_or_else(|| Error::UnsupportedLayout("box size overflow".to_string()))?;
232        if end > data.len() {
233            return Err(Error::UnsupportedLayout(
234                "box exceeds parent bounds".to_string(),
235            ));
236        }
237        let name: &[u8; 4] = data[position + 4..position + 8]
238            .try_into()
239            .expect("box type is four bytes");
240        let payload = &data[position + header..end];
241        visit(name, payload)?;
242        if let Some(offset) = children_offset(name) {
243            walk_boxes(payload, offset, visit)?;
244        }
245        position = end;
246    }
247    Ok(())
248}

250fn walk_boxes_mut(
251    data: &mut [u8],
252    child_offset: usize,
253    visit: &mut impl FnMut(&mut [u8; 4]),
254) -> Result<(), Error> {
255    if child_offset > data.len() {
256        return Err(Error::UnsupportedLayout("invalid child offset".to_string()));
257    }
258    let mut position = child_offset;
259    while position < data.len() {
260        let (size, header) = box_size(data, position)?;
261        let end = position
262            .checked_add(size)
263            .ok_or_else(|| Error::UnsupportedLayout("box size overflow".to_string()))?;
264        if end > data.len() {
265            return Err(Error::UnsupportedLayout(
266                "box exceeds parent bounds".to_string(),
267            ));
268        }
269        let original: [u8; 4] = data[position + 4..position + 8]
270            .try_into()
271            .expect("box type is four bytes");
272        let offset = children_offset(&original);
273        let name: &mut [u8; 4] = (&mut data[position + 4..position + 8])
274            .try_into()
275            .expect("box type is four bytes");
276        visit(name);
277        if let Some(offset) = offset {
278            walk_boxes_mut(&mut data[position + header..end], offset, visit)?;
279        }
280        position = end;
281    }
282    Ok(())
283}

285fn box_size(data: &[u8], position: usize) -> Result<(usize, usize), Error> {
286    if data.len().saturating_sub(position) < 8 {
287        return Err(Error::UnsupportedLayout("truncated box header".to_string()));
288    }
289    let short = u32::from_be_bytes(
290        data[position..position + 4]
291            .try_into()
292            .expect("box size is four bytes"),
293    );
294    match short {
295        0 => Ok((data.len() - position, 8)),
296        1 => {
297            if data.len().saturating_sub(position) < 16 {
298                return Err(Error::UnsupportedLayout(
299                    "truncated extended box header".to_string(),
300                ));
301            }
302            let size = u64::from_be_bytes(
303                data[position + 8..position + 16]
304                    .try_into()
305                    .expect("extended box size is eight bytes"),
306            );
307            let size = usize::try_from(size)
308                .map_err(|_| Error::UnsupportedLayout("box size overflow".to_string()))?;
309            if size < 16 {
310                return Err(Error::UnsupportedLayout(
311                    "invalid extended box size".to_string(),
312                ));
313            }
314            Ok((size, 16))
315        }
316        size if size < 8 => Err(Error::UnsupportedLayout("invalid box size".to_string())),
317        size => Ok((size as usize, 8)),
318    }
319}

321fn children_offset(name: &[u8; 4]) -> Option<usize> {
322    match name {
323        b"moov" | b"trak" | b"mdia" | b"minf" | b"stbl" | b"sinf" | b"schi" | b"moof" | b"traf" => {
324            Some(0)
325        }
326        b"stsd" => Some(8),
327        b"encv" => Some(78),
328        b"enca" => Some(28),
329        _ => None,
330    }
331}

423    fn init(scheme: [u8; 4], audio: bool) -> Vec<u8> {
424        let mut schm = vec![0, 0, 0, 0];
425        schm.extend_from_slice(&scheme);
426        schm.extend_from_slice(&0x0001_0000_u32.to_be_bytes());
427        let schm = mp4_box(*b"schm", schm);
428
429        let version = u8::from(matches!(&scheme, b"cens" | b"cbcs"));
430        let mut tenc = vec![version, 0, 0, 0, 0];
431        tenc.push(0);
432        tenc.push(1);
433        tenc.push(16);
434        tenc.extend_from_slice(&[3; 16]);
435        let tenc = mp4_box(*b"tenc", tenc);
436        let sinf = mp4_box(*b"sinf", [schm, mp4_box(*b"schi", tenc)].concat());
437        let (kind, prefix) = if audio {
438            (*b"enca", vec![0; 28])
439        } else {
440            (*b"encv", vec![0; 78])
441        };
442        let entry = mp4_box(kind, [prefix, sinf].concat());
443        let mut stsd = vec![0, 0, 0, 0];
444        stsd.extend_from_slice(&1_u32.to_be_bytes());
445        stsd.extend_from_slice(&entry);
446        mp4_box(
447            *b"moov",
448            mp4_box(
449                *b"trak",
450                mp4_box(
451                    *b"mdia",
452                    mp4_box(*b"minf", mp4_box(*b"stbl", mp4_box(*b"stsd", stsd))),
453                ),
454            ),
455        )
456    }

458    fn fragment(ciphertext: &[u8], subsample: bool, iv: [u8; 16]) -> Vec<u8> {
459        let mut senc = vec![0, 0, 0, if subsample { 2 } else { 0 }];
460        senc.extend_from_slice(&1_u32.to_be_bytes());
461        senc.extend_from_slice(&iv);
462        if subsample {
463            senc.extend_from_slice(&1_u16.to_be_bytes());
464            senc.extend_from_slice(&5_u16.to_be_bytes());
465            senc.extend_from_slice(&16_u32.to_be_bytes());
466        }
467        let senc = mp4_box(*b"senc", senc);
468
469        let mut trun = vec![0, 0, 2, 1];
470        trun.extend_from_slice(&1_u32.to_be_bytes());
471        let data_offset_position = trun.len();
472        trun.extend_from_slice(&0_u32.to_be_bytes());
473        trun.extend_from_slice(&(ciphertext.len() as u32).to_be_bytes());
474        let trun = mp4_box(*b"trun", trun);
475        let traf = mp4_box(*b"traf", [senc, trun].concat());
476        let mut moof = mp4_box(*b"moof", traf);
477        let data_offset = u32::try_from(moof.len() + 8).unwrap();
478        let trun_type = moof.windows(4).position(|value| value == b"trun").unwrap();
479        let field = trun_type + 4 + data_offset_position;
480        moof[field..field + 4].copy_from_slice(&data_offset.to_be_bytes());
481        [moof, mp4_box(*b"mdat", ciphertext.to_vec())].concat()
482    }
483
484    fn mp4_box(name: [u8; 4], payload: Vec<u8>) -> Vec<u8> {
485        let size = u32::try_from(payload.len() + 8).unwrap();
486        [size.to_be_bytes().as_slice(), &name, &payload].concat()
487    }
