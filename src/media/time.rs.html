6/// A media timestamp scale expressed as ticks per second.
7#[derive(Clone, Copy, Debug, PartialEq, Eq)]
8pub struct TimeBase(NonZeroU32);
9
10impl TimeBase {
11    /// Construct a nonzero media time base.
12    ///
13    /// # Errors
14    ///
15    /// Returns an error when `ticks_per_second` is zero.
16    pub fn new(ticks_per_second: u32) -> Result<Self, Error> {
17        NonZeroU32::new(ticks_per_second)
18            .map(Self)
19            .ok_or(Error::Invalid("zero media timescale"))
20    }
21
22    /// Return the number of ticks per second.
23    #[must_use]
24    pub fn ticks_per_second(self) -> u32 {
25        self.0.get()
26    }
27
28    /// Convert unsigned ticks to a duration without floating-point rounding.
29    ///
30    /// # Errors
31    ///
32    /// Returns an error if nanosecond conversion overflows.
33    pub fn duration(self, ticks: u64) -> Result<Duration, Error> {
34        let nanos = u128::from(ticks)
35            .checked_mul(1_000_000_000)
36            .ok_or(Error::Overflow("duration"))?
37            / u128::from(self.0.get());
38        let nanos = u64::try_from(nanos).map_err(|_| Error::Overflow("duration"))?;
39        Ok(Duration::from_nanos(nanos))
40    }
41}
