5impl<R: Read + Seek> PacketIter<R> {
6    pub(super) fn parse_traf(
7        &mut self,
8        moof: BoxHeader,
9        traf: BoxHeader,
10        output: &mut Vec<Sample>,
11    ) -> Result<()> {
12        let mut defaults = None;
13        let mut decode_time = None;
14        let mut truns = Vec::new();
15        seek_to(&mut self.reader, traf.content_start)?;
16        while let Some(child) = next_box(&mut self.reader, traf.end)? {
17            match &child.kind {
18                b"tfhd" => defaults = Some(self.parse_tfhd(moof, child)?),
19                b"tfdt" => decode_time = Some(parse_tfdt(&mut self.reader, child)?),
20                b"trun" => {
21                    if truns.len() >= 1024 {
22                        return Err(Error::Unsupported("excessive trun boxes in one fragment"));
23                    }
24                    truns.push(child);
25                }
26                _ => {}
27            }
28            seek_to(&mut self.reader, child.end)?;
29        }
30        let defaults = defaults.ok_or(Error::Invalid("traf has no tfhd"))?;
31        let mut dts = decode_time.unwrap_or_else(|| self.running_dts(defaults.track_id));
32        let edit_offset = edit_offset(
33            self.tracks
34                .iter()
35                .find(|track| track.id == defaults.track_id)
36                .ok_or(Error::Invalid("tfhd references unknown track"))?,
37        )?;
38        let mut next_data_offset = None;
39        for trun in truns {
40            next_data_offset = Some(self.parse_trun(
41                trun,
42                defaults,
43                &mut dts,
44                edit_offset,
45                next_data_offset,
46                output,
47            )?);
48        }
49        self.set_running_dts(defaults.track_id, dts);
50        Ok(())
51    }

53    fn parse_tfhd(&mut self, moof: BoxHeader, header: BoxHeader) -> Result<FragmentDefaults> {
54        seek_to(&mut self.reader, header.content_start)?;
55        let (_, flags) = full_box(&mut self.reader)?;
56        let track_id = read_u32(&mut self.reader)?;
57        let trex = self
58            .defaults
59            .iter()
60            .find(|defaults| defaults.track_id == track_id)
61            .copied()
62            .unwrap_or_default();
63        let base_offset = if flags & 0x000001 != 0 {
64            read_u64(&mut self.reader)?
65        } else if flags & 0x020000 != 0 {
66            moof.start
67        } else {
68            return Err(Error::Unsupported(
69                "tfhd without an explicit or moof-relative data base",
70            ));
71        };
72        if flags & 0x000002 != 0 {
73            let _sample_description_index = read_u32(&mut self.reader)?;
74        }
75        let duration = if flags & 0x000008 != 0 {
76            read_u32(&mut self.reader)?
77        } else {
78            trex.duration
79        };
80        let size = if flags & 0x000010 != 0 {
81            read_u32(&mut self.reader)?
82        } else {
83            trex.size
84        };
85        let sample_flags = if flags & 0x000020 != 0 {
86            read_u32(&mut self.reader)?
87        } else {
88            trex.flags
89        };
90        Ok(FragmentDefaults {
91            track_id,
92            base_offset,
93            duration,
94            size,
95            flags: sample_flags,
96        })
97    }

99    #[allow(clippy::too_many_arguments)]
100    fn parse_trun(
101        &mut self,
102        header: BoxHeader,
103        defaults: FragmentDefaults,
104        dts: &mut u64,
105        edit_offset: i64,
106        prior_end: Option<u64>,
107        output: &mut Vec<Sample>,
108    ) -> Result<u64> {
109        seek_to(&mut self.reader, header.content_start)?;
110        let (version, flags) = full_box(&mut self.reader)?;
111        if version > 1 {
112            return Err(Error::Unsupported("trun version"));
113        }
114        let count = read_u32(&mut self.reader)?;
115        if count > MAX_SAMPLES_PER_FRAGMENT
116            || output.len().saturating_add(count as usize) > MAX_SAMPLES_PER_FRAGMENT as usize
117        {
118            return Err(Error::Unsupported("excessive samples in one fragment"));
119        }
120        let data_offset = if flags & 0x000001 != 0 {
121            let relative = i64::from(read_i32(&mut self.reader)?);
122            add_signed(defaults.base_offset, relative, "trun data offset")?
123        } else {
124            prior_end.ok_or(Error::Unsupported("first trun has no data offset"))?
125        };
126        let first_sample_flags = if flags & 0x000004 != 0 {
127            Some(read_u32(&mut self.reader)?)
128        } else {
129            None
130        };
131        let mut offset = data_offset;
132        for index in 0..count {
133            let duration = if flags & 0x000100 != 0 {
134                read_u32(&mut self.reader)?
135            } else {
136                defaults.duration
137            };
138            let size = if flags & 0x000200 != 0 {
139                read_u32(&mut self.reader)?
140            } else {
141                defaults.size
142            };
143            let sample_flags = if flags & 0x000400 != 0 {
144                read_u32(&mut self.reader)?
145            } else if index == 0 {
146                first_sample_flags.unwrap_or(defaults.flags)
147            } else {
148                defaults.flags
149            };
150            let composition_offset = if flags & 0x000800 != 0 {
151                if version == 0 {
152                    i64::from(read_u32(&mut self.reader)?)
153                } else {
154                    i64::from(read_i32(&mut self.reader)?)
155                }
156            } else {
157                0
158            };
159            if duration == 0 || size == 0 {
160                return Err(Error::Invalid("sample duration or size is zero"));
161            }
162            let raw_dts = i64::try_from(*dts).map_err(|_| Error::Overflow("sample dts"))?;
163            let edited_dts = raw_dts
164                .checked_add(edit_offset)
165                .ok_or(Error::Overflow("edited dts"))?;
166            let pts = edited_dts
167                .checked_add(composition_offset)
168                .ok_or(Error::Overflow("sample pts"))?;
169            output.push(Sample {
170                track_id: defaults.track_id,
171                offset,
172                size,
173                dts: edited_dts,
174                pts,
175                duration,
176                flags: sample_flags,
177            });
178            offset = offset
179                .checked_add(u64::from(size))
180                .ok_or(Error::Overflow("sample data offset"))?;
181            *dts = dts
182                .checked_add(u64::from(duration))
183                .ok_or(Error::Overflow("decode timestamp"))?;
184        }
185        Ok(offset)
186    }

188    fn running_dts(&self, track_id: u32) -> u64 {
189        self.next_dts
190            .iter()
191            .find_map(|(id, dts)| (*id == track_id).then_some(*dts))
192            .unwrap_or(0)
193    }
194
195    fn set_running_dts(&mut self, track_id: u32, dts: u64) {
196        if let Some((_, value)) = self.next_dts.iter_mut().find(|(id, _)| *id == track_id) {
197            *value = dts;
198        } else {
199            self.next_dts.push((track_id, dts));
200        }
201    }
