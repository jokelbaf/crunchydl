5/// The media role of a parsed track.

7#[non_exhaustive]

9    /// A video track.

11    /// An audio track.

15/// One movie edit converted into the track time base.
16#[derive(Clone, Copy, Debug, PartialEq, Eq)]
17pub struct Edit {
18    /// Presentation duration of the edit.
19    pub duration: u64,
20    /// Starting media time, or `-1` for an empty edit.

22}

24/// Neutral metadata for one parsed AVC or AAC track.
25#[derive(Clone, Debug, PartialEq, Eq)]
26pub struct Track {
27    /// ISO-BMFF track identifier.
28    pub id: u32,
29    /// Video or audio role.
30    pub kind: TrackKind,
31    /// Track media time base.
32    pub time_base: TimeBase,
33    /// Codec and exact codec-private bytes.
34    pub codec: Codec,
35    /// Display dimensions for video.
36    pub dimensions: Option<(u32, u32)>,
37    /// Sampling rate for audio.
38    pub sample_rate: Option<u32>,
39    /// Channel count for audio.
40    pub channels: Option<u16>,
41    /// Declared or fragment-derived duration.
42    pub duration: Duration,
43    /// Ordered edit-list entries, expressed in track ticks.
44    pub edits: Vec<Edit>,
45}
